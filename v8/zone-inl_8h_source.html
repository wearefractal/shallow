00042 inline void* Zone::New(int size) {
00043   ASSERT(ZoneScope::nesting() > 0);
00044   // Round up the requested size to fit the alignment.
00045   size = RoundUp(size, kAlignment);
00046 
00047   // If the allocation size is divisible by 8 then we return an 8-byte aligned
00048   // address.
00049   if (kPointerSize == 4 && kAlignment == 4) {
00050     position_ += ((~size) & 4) & (reinterpret_cast<intptr_t>(position_) & 4);
00051   } else {
00052     ASSERT(kAlignment >= kPointerSize);
00053   }
00054 
00055   // Check if the requested size is available without expanding.
00056   Address result = position_;
00057 
00058   if (size > limit_ - position_) {
00059      result = NewExpand(size);
00060   } else {
00061      position_ += size;
00062   }
00063 
00064   // Check that the result has the proper alignment and return it.
00065   ASSERT(IsAddressAligned(result, kAlignment, 0));
00066   allocation_size_ += size;
00067   return reinterpret_cast<void*>(result);
00068 }
