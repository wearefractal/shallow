00305 int SnapshotByteSource::GetInt() {
00306   // A little unwind to catch the really small ints.
00307   int snapshot_byte = Get();
00308   if ((snapshot_byte & 0x80) == 0) {
00309     return snapshot_byte;
00310   }
00311   int accumulator = (snapshot_byte & 0x7f) << 7;
00312   while (true) {
00313     snapshot_byte = Get();
00314     if ((snapshot_byte & 0x80) == 0) {
00315       return accumulator | snapshot_byte;
00316     }
00317     accumulator = (accumulator | (snapshot_byte & 0x7f)) << 7;
00318   }
00319   UNREACHABLE();
00320   return accumulator;
00321 }
