00497   static Mode GetMode(Code* stub) {
00498     Instr first_instruction = Assembler::instr_at(stub->instruction_start());
00499     Instr second_instruction = Assembler::instr_at(stub->instruction_start() +
00500                                                    Assembler::kInstrSize);
00501 
00502     if (Assembler::IsBranch(first_instruction)) {
00503       return INCREMENTAL;
00504     }
00505 
00506     ASSERT(Assembler::IsTstImmediate(first_instruction));
00507 
00508     if (Assembler::IsBranch(second_instruction)) {
00509       return INCREMENTAL_COMPACTION;
00510     }
00511 
00512     ASSERT(Assembler::IsTstImmediate(second_instruction));
00513 
00514     return STORE_BUFFER_ONLY;
00515   }

00517   static void Patch(Code* stub, Mode mode) {
00518     MacroAssembler masm(NULL,
00519                         stub->instruction_start(),
00520                         stub->instruction_size());
00521     switch (mode) {
00522       case STORE_BUFFER_ONLY:
00523         ASSERT(GetMode(stub) == INCREMENTAL ||
00524                GetMode(stub) == INCREMENTAL_COMPACTION);
00525         PatchBranchIntoNop(&masm, 0);
00526         PatchBranchIntoNop(&masm, Assembler::kInstrSize);
00527         break;
00528       case INCREMENTAL:
00529         ASSERT(GetMode(stub) == STORE_BUFFER_ONLY);
00530         PatchNopIntoBranch(&masm, 0);
00531         break;
00532       case INCREMENTAL_COMPACTION:
00533         ASSERT(GetMode(stub) == STORE_BUFFER_ONLY);
00534         PatchNopIntoBranch(&masm, Assembler::kInstrSize);
00535         break;
00536     }
00537     ASSERT(GetMode(stub) == mode);
00538     CPU::FlushICache(stub->instruction_start(), 2 * Assembler::kInstrSize);
00539   }

00568     // If we have to call into C then we need to save and restore all caller-
00569     // saved registers that were not already preserved.  The scratch registers
00570     // will be restored by other means so we don't bother pushing them here.
00571     void SaveCallerSaveRegisters(MacroAssembler* masm, SaveFPRegsMode mode) {
00572       masm->stm(db_w, sp, (kCallerSaved | lr.bit()) & ~scratch1_.bit());
00573       if (mode == kSaveFPRegs) {
00574         CpuFeatures::Scope scope(VFP3);
00575         masm->sub(sp,
00576                   sp,
00577                   Operand(kDoubleSize * (DwVfpRegister::kNumRegisters - 1)));
00578         // Save all VFP registers except d0.
00579         for (int i = DwVfpRegister::kNumRegisters - 1; i > 0; i--) {
00580           DwVfpRegister reg = DwVfpRegister::from_code(i);
00581           masm->vstr(reg, MemOperand(sp, (i - 1) * kDoubleSize));
00582         }
00583       }
00584     }

00586     inline void RestoreCallerSaveRegisters(MacroAssembler*masm,
00587                                            SaveFPRegsMode mode) {
00588       if (mode == kSaveFPRegs) {
00589         CpuFeatures::Scope scope(VFP3);
00590         // Restore all VFP registers except d0.
00591         for (int i = DwVfpRegister::kNumRegisters - 1; i > 0; i--) {
00592           DwVfpRegister reg = DwVfpRegister::from_code(i);
00593           masm->vldr(reg, MemOperand(sp, (i - 1) * kDoubleSize));
00594         }
00595         masm->add(sp,
00596                   sp,
00597                   Operand(kDoubleSize * (DwVfpRegister::kNumRegisters - 1)));
00598       }
00599       masm->ldm(ia_w, sp, (kCallerSaved | lr.bit()) & ~scratch1_.bit());
