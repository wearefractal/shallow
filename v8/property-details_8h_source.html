00035 // Ecma-262 3rd 8.6.1
00036 enum PropertyAttributes {
00037   NONE              = v8::None,
00038   READ_ONLY         = v8::ReadOnly,
00039   DONT_ENUM         = v8::DontEnum,
00040   DONT_DELETE       = v8::DontDelete,
00041   ABSENT            = 16  // Used in runtime to indicate a property is absent.
00042   // ABSENT can never be stored in or returned from a descriptor's attributes
00043   // bitfield.  It is only used as a return value meaning the attributes of
00044   // a non-existent property.
00045 };

00053 // Type of properties.
00054 // Order of properties is significant.
00055 // Must fit in the BitField PropertyDetails::TypeField.
00056 // A copy of this is in mirror-debugger.js.
00057 enum PropertyType {
00058   NORMAL                    = 0,  // only in slow mode
00059   FIELD                     = 1,  // only in fast mode
00060   CONSTANT_FUNCTION         = 2,  // only in fast mode
00061   CALLBACKS                 = 3,
00062   HANDLER                   = 4,  // only in lookup results, not in descriptors
00063   INTERCEPTOR               = 5,  // only in lookup results, not in descriptors
00064   // All properties before MAP_TRANSITION are real.
00065   MAP_TRANSITION            = 6,  // only in fast mode
00066   ELEMENTS_TRANSITION       = 7,
00067   CONSTANT_TRANSITION       = 8,  // only in fast mode
00068   NULL_DESCRIPTOR           = 9,  // only in fast mode
00069   // There are no IC stubs for NULL_DESCRIPTORS. Therefore,
00070   // NULL_DESCRIPTOR can be used as the type flag for IC stubs for
00071   // nonexistent properties.
00072   NONEXISTENT = NULL_DESCRIPTOR
00073 };

00076 // PropertyDetails captures type and attributes for a property.
00077 // They are used both in property dictionaries and instance descriptors.
00078 class PropertyDetails BASE_EMBEDDED {
00079  public:
00080   PropertyDetails(PropertyAttributes attributes,
00081                   PropertyType type,
00082                   int index = 0) {
00083     ASSERT(TypeField::is_valid(type));
00084     ASSERT(AttributesField::is_valid(attributes));
00085     ASSERT(StorageField::is_valid(index));
00086 
00087     value_ = TypeField::encode(type)
00088         | AttributesField::encode(attributes)
00089         | StorageField::encode(index);
00090 
00091     ASSERT(type == this->type());
00092     ASSERT(attributes == this->attributes());
00093     ASSERT(index == this->index());
00094   }
00095 
00096   // Conversion for storing details as Object*.
00097   explicit inline PropertyDetails(Smi* smi);
00098   inline Smi* AsSmi();
00099 
00100   PropertyType type() { return TypeField::decode(value_); }
00101 
00102   PropertyAttributes attributes() { return AttributesField::decode(value_); }
00103 
00104   int index() { return StorageField::decode(value_); }
00105 
00106   inline PropertyDetails AsDeleted();
00107 
00108   static bool IsValidIndex(int index) {
00109     return StorageField::is_valid(index);
00110   }
00111 
00112   bool IsReadOnly() { return (attributes() & READ_ONLY) != 0; }
00113   bool IsDontDelete() { return (attributes() & DONT_DELETE) != 0; }
00114   bool IsDontEnum() { return (attributes() & DONT_ENUM) != 0; }
00115   bool IsDeleted() { return DeletedField::decode(value_) != 0;}
00116 
00117   // Bit fields in value_ (type, shift, size). Must be public so the
00118   // constants can be embedded in generated code.
00119   class TypeField:       public BitField<PropertyType,       0, 4> {};
00120   class AttributesField: public BitField<PropertyAttributes, 4, 3> {};
00121   class DeletedField:    public BitField<uint32_t,           7, 1> {};
00122   class StorageField:    public BitField<uint32_t,           8, 32-8> {};
00123 
00124   static const int kInitialIndex = 1;
00125 
00126  private:
00127   uint32_t value_;
00128 };
