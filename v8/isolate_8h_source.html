00981   void SetCurrentVMState(StateTag state) {
00982     if (RuntimeProfiler::IsEnabled()) {
00983       // Make sure thread local top is initialized.
00984       ASSERT(thread_local_top_.isolate_ == this);
00985       StateTag current_state = thread_local_top_.current_vm_state_;
00986       if (current_state != JS && state == JS) {
00987         // Non-JS -> JS transition.
00988         RuntimeProfiler::IsolateEnteredJS(this);
00989       } else if (current_state == JS && state != JS) {
00990         // JS -> non-JS transition.
00991         ASSERT(RuntimeProfiler::IsSomeIsolateInJS());
00992         RuntimeProfiler::IsolateExitedJS(this);
00993       } else {
00994         // Other types of state transitions are not interesting to the
00995         // runtime profiler, because they don't affect whether we're
00996         // in JS or not.
00997         ASSERT((current_state == JS) == (state == JS));
00998       }
00999     }
01000     thread_local_top_.current_vm_state_ = state;
01001   }

01351 // Support for checking for stack-overflows in C++ code.

01353  public:

01356   bool HasOverflowed() const {
01357     StackGuard* stack_guard = isolate_->stack_guard();
01358     // Stack has overflowed in C++ code only if stack pointer exceeds the C++
01359     // stack guard and the limits are not set to interrupt values.
