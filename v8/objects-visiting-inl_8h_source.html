00096 void Code::CodeIterateBody(ObjectVisitor* v) {
00097   int mode_mask = RelocInfo::kCodeTargetMask |
00098                   RelocInfo::ModeMask(RelocInfo::EMBEDDED_OBJECT) |
00099                   RelocInfo::ModeMask(RelocInfo::GLOBAL_PROPERTY_CELL) |
00100                   RelocInfo::ModeMask(RelocInfo::EXTERNAL_REFERENCE) |
00101                   RelocInfo::ModeMask(RelocInfo::JS_RETURN) |
00102                   RelocInfo::ModeMask(RelocInfo::DEBUG_BREAK_SLOT) |
00103                   RelocInfo::ModeMask(RelocInfo::RUNTIME_ENTRY);
00104 
00105   // There are two places where we iterate code bodies: here and the
00106   // templated CodeIterateBody (below).  They should be kept in sync.
00107   IteratePointer(v, kRelocationInfoOffset);
00108   IteratePointer(v, kHandlerTableOffset);
00109   IteratePointer(v, kDeoptimizationDataOffset);
00110   IteratePointer(v, kTypeFeedbackInfoOffset);
00111 
00112   RelocIterator it(this, mode_mask);
00113   for (; !it.done(); it.next()) {
00114     it.rinfo()->Visit(v);
00115   }
00116 }
