00203 bool RelocInfo::IsPatchedReturnSequence() {
00204   Instr current_instr = Assembler::instr_at(pc_);
00205   Instr next_instr = Assembler::instr_at(pc_ + Assembler::kInstrSize);
00206 #ifdef USE_BLX
00207   // A patched return sequence is:
00208   //  ldr ip, [pc, #0]
00209   //  blx ip
00210   return ((current_instr & kLdrPCMask) == kLdrPCPattern)
00211           && ((next_instr & kBlxRegMask) == kBlxRegPattern);
00212 #else
00213   // A patched return sequence is:
00214   //  mov lr, pc
00215   //  ldr pc, [pc, #-4]
00216   return (current_instr == kMovLrPc)
00217           && ((next_instr & kLdrPCMask) == kLdrPCPattern);
00218 #endif
00219 }

00228 void RelocInfo::Visit(ObjectVisitor* visitor) {
00229   RelocInfo::Mode mode = rmode();
00230   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00231     visitor->VisitEmbeddedPointer(this);
00232   } else if (RelocInfo::IsCodeTarget(mode)) {
00233     visitor->VisitCodeTarget(this);
00234   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00235     visitor->VisitGlobalPropertyCell(this);
00236   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00237     visitor->VisitExternalReference(this);
00238 #ifdef ENABLE_DEBUGGER_SUPPORT
00239   // TODO(isolates): Get a cached isolate below.
00240   } else if (((RelocInfo::IsJSReturn(mode) &&
00241               IsPatchedReturnSequence()) ||
00242              (RelocInfo::IsDebugBreakSlot(mode) &&
00243               IsPatchedDebugBreakSlotSequence())) &&
00244              Isolate::Current()->debug()->has_break_points()) {
00245     visitor->VisitDebugTarget(this);
00246 #endif
00247   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00248     visitor->VisitRuntimeEntry(this);
00249   }
00250 }

00253 template<typename StaticVisitor>
00254 void RelocInfo::Visit(Heap* heap) {
00255   RelocInfo::Mode mode = rmode();
00256   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00257     StaticVisitor::VisitEmbeddedPointer(heap, this);
00258   } else if (RelocInfo::IsCodeTarget(mode)) {
00259     StaticVisitor::VisitCodeTarget(heap, this);
00260   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00261     StaticVisitor::VisitGlobalPropertyCell(heap, this);
00262   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00263     StaticVisitor::VisitExternalReference(this);
00264 #ifdef ENABLE_DEBUGGER_SUPPORT
00265   } else if (heap->isolate()->debug()->has_break_points() &&
00266              ((RelocInfo::IsJSReturn(mode) &&
00267               IsPatchedReturnSequence()) ||
00268              (RelocInfo::IsDebugBreakSlot(mode) &&
00269               IsPatchedDebugBreakSlotSequence()))) {
00270     StaticVisitor::VisitDebugTarget(heap, this);
00271 #endif
00272   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00273     StaticVisitor::VisitRuntimeEntry(this);
00274   }
00275 }

00332 Address Assembler::target_address_address_at(Address pc) {
00333   Address target_pc = pc;
00334   Instr instr = Memory::int32_at(target_pc);
00335   // If we have a bx instruction, the instruction before the bx is
00336   // what we need to patch.
00337   static const int32_t kBxInstMask = 0x0ffffff0;
00338   static const int32_t kBxInstPattern = 0x012fff10;
00339   if ((instr & kBxInstMask) == kBxInstPattern) {
00340     target_pc -= kInstrSize;
00341     instr = Memory::int32_at(target_pc);
00342   }
00343 
00344 #ifdef USE_BLX
00345   // If we have a blx instruction, the instruction before it is
00346   // what needs to be patched.
00347   if ((instr & kBlxRegMask) == kBlxRegPattern) {
00348     target_pc -= kInstrSize;
00349     instr = Memory::int32_at(target_pc);
00350   }
00351 #endif
00352 
00353   ASSERT(IsLdrPcImmediateOffset(instr));
00354   int offset = instr & 0xfff;  // offset_12 is unsigned
00355   if ((instr & (1 << 23)) == 0) offset = -offset;  // U bit defines offset sign
00356   // Verify that the constant pool comes after the instruction referencing it.
00357   ASSERT(offset >= -4);
00358   return target_pc + offset + 8;
00359 }

00379 void Assembler::set_target_address_at(Address pc, Address target) {
00380   Memory::Address_at(target_address_address_at(pc)) = target;
00381   // Intuitively, we would think it is necessary to flush the instruction cache

00385   // above; the target address is not part of an instruction, it is patched in
00386   // the constant pool and is read via a data access; the instruction accessing
00387   // this address in the constant pool remains unchanged.
