00028 // emulates google3/base/once.h
00029 //
00030 // This header is intended to be included only by v8's internal code. Users
00031 // should not use this directly.
00032 //
00033 // This is basically a portable version of pthread_once().
00034 //
00035 // This header declares:
00036 // * A type called OnceType.
00037 // * A macro V8_DECLARE_ONCE() which declares a (global) variable of type
00038 //   OnceType.
00039 // * A function CallOnce(OnceType* once, void (*init_func)()).
00040 //   This function, when invoked multiple times given the same OnceType object,
00041 //   will invoke init_func on the first call only, and will make sure none of
00042 //   the calls return before that first call to init_func has finished.
00043 //
00044 // Additionally, the following features are supported:
00045 // * A macro V8_ONCE_INIT which is expanded into the expression used to
00046 //   initialize a OnceType. This is only useful when clients embed a OnceType
00047 //   into a structure of their own and want to initialize it statically.
00048 // * The user can provide a parameter which CallOnce() forwards to the
00049 //   user-provided function when it is called. Usage example:
00050 //     CallOnce(&my_once, &MyFunctionExpectingIntArgument, 10);
00051 // * This implementation guarantees that OnceType is a POD (i.e. no static
00052 //   initializer generated).
00053 //
00054 // This implements a way to perform lazy initialization.  It's more efficient
00055 // than using mutexes as no lock is needed if initialization has already
00056 // happened.
00057 //
00058 // Example usage:
00059 //   void Init();
00060 //   V8_DECLARE_ONCE(once_init);
00061 //
00062 //   // Calls Init() exactly once.
00063 //   void InitOnce() {
00064 //     CallOnce(&once_init, &Init);
00065 //   }
00066 //
00067 // Note that if CallOnce() is called before main() has begun, it must
00068 // only be called by the thread that will eventually call main() -- that is,
00069 // the thread that performs dynamic initialization.  In general this is a safe
00070 // assumption since people don't usually construct threads before main() starts,
00071 // but it is technically not guaranteed.  Unfortunately, Win32 provides no way
00072 // whatsoever to statically-initialize its synchronization primitives, so our
00073 // only choice is to assume that dynamic initialization is single-threaded.

00105 inline void CallOnce(OnceType* once, NoArgFunction init_func) {
00106   if (Acquire_Load(once) != ONCE_STATE_DONE) {
