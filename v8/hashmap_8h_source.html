00121 typename TemplateHashMapImpl<P>::Entry* TemplateHashMapImpl<P>::Lookup(
00122     void* key, uint32_t hash, bool insert) {
00123   // Find a matching entry.
00124   Entry* p = Probe(key, hash);
00125   if (p->key != NULL) {
00126     return p;
00127   }
00128 
00129   // No entry found; insert one if necessary.
00130   if (insert) {
00131     p->key = key;
00132     p->value = NULL;

00150 template<class P>
00151 void* TemplateHashMapImpl<P>::Remove(void* key, uint32_t hash) {
00152   // Lookup the entry for the key to remove.
00153   Entry* p = Probe(key, hash);
00154   if (p->key == NULL) {
00155     // Key not found nothing to remove.
00156     return NULL;
00157   }
00158 
00159   void* value = p->value;
00160   // To remove an entry we need to ensure that it does not create an empty
00161   // entry that will cause the search for another entry to stop too soon. If all
00162   // the entries between the entry to remove and the next empty slot have their
00163   // initial position inside this interval, clearing the entry to remove will
00164   // not break the search. If, while searching for the next empty entry, an
00165   // entry is encountered which does not have its initial position between the
00166   // entry to remove and the position looked at, then this entry can be moved to
00167   // the place of the entry to remove without breaking the search for it. The
00168   // entry made vacant by this move is now the entry to remove and the process
00169   // starts over.
00170   // Algorithm from http://en.wikipedia.org/wiki/Open_addressing.
00171 
00172   // This guarantees loop termination as there is at least one empty entry so
00173   // eventually the removed entry will have an empty entry after it.
00174   ASSERT(occupancy_ < capacity_);
00175 
00176   // p is the candidate entry to clear. q is used to scan forwards.
00177   Entry* q = p;  // Start at the entry to remove.
00178   while (true) {
00179     // Move q to the next entry.
00180     q = q + 1;
00181     if (q == map_end()) {
00182       q = map_;
00183     }
00184 
00185     // All entries between p and q have their initial position between p and q
00186     // and the entry p can be cleared without breaking the search for these
00187     // entries.
00188     if (q->key == NULL) {
00189       break;
00190     }
00191 
00192     // Find the initial position for the entry at position q.
00193     Entry* r = map_ + (q->hash & (capacity_ - 1));
00194 
00195     // If the entry at position q has its initial position outside the range
00196     // between p and q it can be moved forward to position p and will still be
00197     // found. There is now a new candidate entry for clearing.
00198     if ((q > p && (r <= p || r > q)) ||
00199         (q < p && (r <= p && r > q))) {
00200       *p = *q;
00201       p = q;
00202     }
00203   }
00204 
00205   // Clear the entry which is allowed to en emptied.
00206   p->key = NULL;
00207   occupancy_--;
00208   return value;
00209 }

00243 template<class P>
00244 typename TemplateHashMapImpl<P>::Entry* TemplateHashMapImpl<P>::Probe(void* key,
00245                                                             uint32_t hash) {
00246   ASSERT(key != NULL);
00247 
00248   ASSERT(IsPowerOf2(capacity_));
00249   Entry* p = map_ + (hash & (capacity_ - 1));
00250   const Entry* end = map_end();
00251   ASSERT(map_ <= p && p < end);
00252 
00253   ASSERT(occupancy_ < capacity_);  // Guarantees loop termination.
00254   while (p->key != NULL && (hash != p->hash || !match_(key, p->key))) {
00255     p++;
00256     if (p >= end) {
00257       p = map_;
00258     }
00259   }
00260 
00261   return p;
00262 }

00278 template<class P>
00279 void TemplateHashMapImpl<P>::Resize() {
00280   Entry* map = map_;
00281   uint32_t n = occupancy_;
00282 
00283   // Allocate larger map.
00284   Initialize(capacity_ * 2);
00285 
00286   // Rehash all current entries.
00287   for (Entry* p = map; n > 0; p++) {
00288     if (p->key != NULL) {
00289       Lookup(p->key, p->hash, true)->value = p->value;
00290       n--;
00291     }
00292   }
00293 
00294   // Delete old map.
00295   P::Delete(map);
00296 }
