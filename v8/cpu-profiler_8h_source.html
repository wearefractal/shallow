00097 class TickSampleEventRecord {
00098  public:
00099   // The parameterless constructor is used when we dequeue data from
00100   // the ticks buffer.
00101   TickSampleEventRecord() { }
00102   explicit TickSampleEventRecord(unsigned order)
00103       : filler(1),
00104         order(order) {
00105     ASSERT(filler != SamplingCircularQueue::kClear);
00106   }
00107 
00108   // The first machine word of a TickSampleEventRecord must not ever
00109   // become equal to SamplingCircularQueue::kClear.  As both order and
00110   // TickSample's first field are not reliable in this sense (order
00111   // can overflow, TickSample can have all fields reset), we are
00112   // forced to use an artificial filler field.
00113   int filler;
00114   unsigned order;
00115   TickSample sample;
00116 
00117   static TickSampleEventRecord* cast(void* value) {
00118     return reinterpret_cast<TickSampleEventRecord*>(value);
00119   }
00120 };
