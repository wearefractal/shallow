00096 void HandleScope::CloseScope() {
00097   ASSERT(isolate_ == Isolate::Current());
00098   v8::ImplementationUtilities::HandleScopeData* current =
00099       isolate_->handle_scope_data();
00100   current->next = prev_next_;
00101   current->level--;
00102   if (current->limit != prev_limit_) {
00103     current->limit = prev_limit_;
00104     DeleteExtensions(isolate_);
00105   }
00106 #ifdef DEBUG
00107   ZapRange(prev_next_, prev_limit_);
00108 #endif
00109 }

00112 template <typename T>
00113 Handle<T> HandleScope::CloseAndEscape(Handle<T> handle_value) {
00114   T* value = *handle_value;
00115   // Throw away all handles in the current scope.
00116   CloseScope();
00117   v8::ImplementationUtilities::HandleScopeData* current =
00118       isolate_->handle_scope_data();
00119   // Allocate one handle in the parent scope.
00120   ASSERT(current->level > 0);
00121   Handle<T> result(CreateHandle<T>(value, isolate_));
00122   // Reinitialize the current scope (so that it's ready
00123   // to be used or closed again).
00124   prev_next_ = current->next;
00125   prev_limit_ = current->limit;
00126   current->level++;
00127   return result;
00128 }

00131 template <typename T>
00132 T** HandleScope::CreateHandle(T* value, Isolate* isolate) {
00133   ASSERT(isolate == Isolate::Current());
00134   v8::ImplementationUtilities::HandleScopeData* current =
00135       isolate->handle_scope_data();
00136 
00137   internal::Object** cur = current->next;
00138   if (cur == current->limit) cur = Extend();
00139   // Update the current next field, set the value in the created
00140   // handle, and return the result.
00141   ASSERT(cur < current->limit);
00142   current->next = cur + 1;
00143 
00144   T** result = reinterpret_cast<T**>(cur);
00145   *result = value;
00146   return result;
00147 }
