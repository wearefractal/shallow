00164 template <class T>
00165 bool IsPropertyDescriptor(T* desc) {
00166   switch (desc->type()) {
00167     case NORMAL:
00168     case FIELD:
00169     case CONSTANT_FUNCTION:
00170     case HANDLER:
00171     case INTERCEPTOR:
00172       return true;
00173     case CALLBACKS: {
00174       Object* callback_object = desc->GetCallbackObject();
00175       // Non-JavaScript (i.e. native) accessors are always a property, otherwise
00176       // either the getter or the setter must be an accessor. Put another way:
00177       // If we only see map transitions and holes in a pair, this is not a
00178       // property.
00179       return (!callback_object->IsAccessorPair() ||
00180               AccessorPair::cast(callback_object)->ContainsAccessor());
00181     }
00182     case MAP_TRANSITION:
00183     case ELEMENTS_TRANSITION:
00184     case CONSTANT_TRANSITION:
00185     case NULL_DESCRIPTOR:
00186       return false;
00187   }
00188   UNREACHABLE();  // keep the compiler happy
00189   return false;
00190 }
