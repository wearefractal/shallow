00047 // ES5 10.2 defines lexical environments with mutable and immutable bindings.
00048 // Immutable bindings have two states, initialized and uninitialized, and
00049 // their state is changed by the InitializeImmutableBinding method. The
00050 // BindingFlags enum represents information if a binding has definitely been
00051 // initialized. A mutable binding does not need to be checked and thus has
00052 // the BindingFlag MUTABLE_IS_INITIALIZED.
00053 //
00054 // There are two possibilities for immutable bindings
00055 //  * 'const' declared variables. They are initialized when evaluating the
00056 //    corresponding declaration statement. They need to be checked for being
00057 //    initialized and thus get the flag IMMUTABLE_CHECK_INITIALIZED.
00058 //  * The function name of a named function literal. The binding is immediately
00059 //    initialized when entering the function and thus does not need to be
00060 //    checked. it gets the BindingFlag IMMUTABLE_IS_INITIALIZED.
00061 // Accessing an uninitialized binding produces the undefined value.
00062 //
00063 // The harmony proposal for block scoped bindings also introduces the
00064 // uninitialized state for mutable bindings.
00065 //  * A 'let' declared variable. They are initialized when evaluating the
00066 //    corresponding declaration statement. They need to be checked for being
00067 //    initialized and thus get the flag MUTABLE_CHECK_INITIALIZED.
00068 //  * A 'var' declared variable. It is initialized immediately upon creation
00069 //    and thus doesn't need to be checked. It gets the flag
00070 //    MUTABLE_IS_INITIALIZED.
00071 //  * Catch bound variables, function parameters and variables introduced by
00072 //    function declarations are initialized immediately and do not need to be
00073 //    checked. Thus they get the flag MUTABLE_IS_INITIALIZED.
00074 // Immutable bindings in harmony mode get the _HARMONY flag variants. Accessing
00075 // an uninitialized binding produces a reference error.
00076 //
00077 // In V8 uninitialized bindings are set to the hole value upon creation and set
00078 // to a different value upon initialization.
00079 enum BindingFlags {
00080   MUTABLE_IS_INITIALIZED,
00081   MUTABLE_CHECK_INITIALIZED,
00082   IMMUTABLE_IS_INITIALIZED,
00083   IMMUTABLE_CHECK_INITIALIZED,
00084   IMMUTABLE_IS_INITIALIZED_HARMONY,
00085   IMMUTABLE_CHECK_INITIALIZED_HARMONY,
00086   MISSING_BINDING
00087 };

00167 // JSFunctions are pairs (context, function code), sometimes also called
00168 // closures. A Context object is used to represent function contexts and
00169 // dynamically pushed 'with' contexts (or 'scopes' in ECMA-262 speak).
00170 //
00171 // At runtime, the contexts build a stack in parallel to the execution
00172 // stack, with the top-most context being the current context. All contexts
00173 // have the following slots:
00174 //
00175 // [ closure   ]  This is the current function. It is the same for all
00176 //                contexts inside a function. It provides access to the
00177 //                incoming context (i.e., the outer context, which may
00178 //                or may not become the current function's context), and
00179 //                it provides access to the functions code and thus it's
00180 //                scope information, which in turn contains the names of
00181 //                statically allocated context slots. The names are needed
00182 //                for dynamic lookups in the presence of 'with' or 'eval'.
00183 //
00184 // [ previous  ]  A pointer to the previous context. It is NULL for
00185 //                function contexts, and non-NULL for 'with' contexts.
00186 //                Used to implement the 'with' statement.
00187 //
00188 // [ extension ]  A pointer to an extension JSObject, or NULL. Used to
00189 //                implement 'with' statements and dynamic declarations
00190 //                (through 'eval'). The object in a 'with' statement is
00191 //                stored in the extension slot of a 'with' context.
00192 //                Dynamically declared variables/functions are also added
00193 //                to lazily allocated extension object. Context::Lookup
00194 //                searches the extension object for properties.
00195 //
00196 // [ global    ]  A pointer to the global object. Provided for quick
00197 //                access to the global object from inside the code (since
00198 //                we always have a context pointer).
00199 //
00200 // In addition, function contexts may have statically allocated context slots
00201 // to store local variables/functions that are accessed from inner functions
00202 // (via static context addresses) or through 'eval' (dynamic context lookups).
00203 // Finally, the global context contains additional slots for fast access to
00204 // global properties.

00400   // Lookup the slot called name, starting with the current context.
00401   // There are three possibilities:
00402   //
00403   // 1) result->IsContext():
00404   //    The binding was found in a context.  *index is always the
00405   //    non-negative slot index.  *attributes is NONE for var and let
