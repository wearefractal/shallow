00035 // Lock-free cache-friendly sampling circular queue for large
00036 // records. Intended for fast transfer of large records between a
00037 // single producer and a single consumer. If the queue is full,
00038 // previous unread records are overwritten. The queue is designed with
00039 // a goal in mind to evade cache lines thrashing by preventing
00040 // simultaneous reads and writes to adjanced memory locations.
00041 //
00042 // IMPORTANT: as a producer never checks for chunks cleanness, it is
00043 // possible that it can catch up and overwrite a chunk that a consumer
00044 // is currently reading, resulting in a corrupt record being read.
00045 class SamplingCircularQueue {
00046  public:
00047   // Executed on the application thread.
00048   SamplingCircularQueue(int record_size_in_bytes,
00049                         int desired_chunk_size_in_bytes,
00050                         int buffer_size_in_chunks);
00051   ~SamplingCircularQueue();
00052 
00053   // Enqueue returns a pointer to a memory location for storing the next
00054   // record.
00055   INLINE(void* Enqueue());
00056 
00057   // Executed on the consumer (analyzer) thread.
00058   // StartDequeue returns a pointer to a memory location for retrieving
00059   // the next record. After the record had been read by a consumer,
00060   // FinishDequeue must be called. Until that moment, subsequent calls
00061   // to StartDequeue will return the same pointer.
00062   void* StartDequeue();
00063   void FinishDequeue();
00064   // Due to a presence of slipping between the producer and the consumer,
00065   // the queue must be notified whether producing has been finished in order
00066   // to process remaining records from the buffer.
00067   void FlushResidualRecords();
00068 
00069   typedef AtomicWord Cell;
00070   // Reserved values for the first cell of a record.
00071   static const Cell kClear = 0;  // Marks clean (processed) chunks.
00072   static const Cell kEnd = -1;   // Marks the end of the buffer.
00073 
00074  private:
00075   struct ProducerPosition {
00076     Cell* enqueue_pos;
00077   };
00078   struct ConsumerPosition {
00079     Cell* dequeue_chunk_pos;
00080     Cell* dequeue_chunk_poll_pos;
00081     Cell* dequeue_pos;
00082     Cell* dequeue_end_pos;
00083   };
00084 
00085   INLINE(void WrapPositionIfNeeded(Cell** pos));
00086 
00087   const int record_size_;
00088   const int chunk_size_in_bytes_;
00089   const int chunk_size_;
00090   const int buffer_size_;
00091   const int producer_consumer_distance_;
00092   Cell* buffer_;
00093   byte* positions_;
00094   ProducerPosition* producer_pos_;
00095   ConsumerPosition* consumer_pos_;
00096 
00097   DISALLOW_COPY_AND_ASSIGN(SamplingCircularQueue);
00098 };
