00072 template<typename T, class P>
00073 void List<T, P>::ResizeAddInternal(const T& element) {
00074   ASSERT(length_ >= capacity_);
00075   // Grow the list capacity by 100%, but make sure to let it grow
00076   // even when the capacity is zero (possible initial case).
00077   int new_capacity = 1 + 2 * capacity_;
00078   // Since the element reference could be an element of the list, copy
00079   // it out of the old backing storage before resizing.
00080   T temp = element;
00081   Resize(new_capacity);
00082   data_[length_++] = temp;
00083 }

00086 template<typename T, class P>
00087 void List<T, P>::Resize(int new_capacity) {
00088   T* new_data = List<T, P>::NewData(new_capacity);
00089   memcpy(new_data, data_, capacity_ * sizeof(T));
00090   List<T, P>::DeleteData(data_);
00091   data_ = new_data;
00092   capacity_ = new_capacity;
00093 }

00210 template <typename T, typename P>
00211 int SortedListBSearch(const List<T>& list, P cmp) {
00212   int low = 0;
00213   int high = list.length() - 1;
00214   while (low <= high) {
00215     int mid = (low + high) / 2;
00216     T mid_elem = list[mid];
00217 
00218     if (cmp(&mid_elem) > 0) {
00219       high = mid - 1;
00220       continue;
00221     }
00222     if (cmp(&mid_elem) < 0) {
00223       low = mid + 1;
00224       continue;
00225     }
00226     // Found the elememt.
00227     return mid;
00228   }
00229   return -1;
00230 }
