00030 // ARM binary. This Simulator allows us to run and debug ARM code generation on

00413 // The simulator has its own stack. Thus it has a different stack limit from
00414 // the C-based native code.  Setting the c_limit to indicate a very small
00415 // stack cause stack overflow errors, since the simulator ignores the input.
00416 // This is unlikely to be an issue in practice, though it might cause testing
00417 // trouble down the line.
00418 class SimulatorStack : public v8::internal::AllStatic {
00419  public:
00420   static inline uintptr_t JsLimitFromCLimit(v8::internal::Isolate* isolate,
00421                                             uintptr_t c_limit) {
00422     return Simulator::current(isolate)->StackLimit();
00423   }
00424 
00425   static inline uintptr_t RegisterCTryCatch(uintptr_t try_catch_address) {
00426     Simulator* sim = Simulator::current(Isolate::Current());
00427     return sim->PushAddress(try_catch_address);
00428   }
00429 
00430   static inline void UnregisterCTryCatch() {
00431     Simulator::current(Isolate::Current())->PopAddress();
00432   }
00433 };
