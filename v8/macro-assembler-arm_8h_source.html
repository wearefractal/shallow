00310   // Push two registers.  Pushes leftmost register first (to highest address).
00311   void Push(Register src1, Register src2, Condition cond = al) {
00312     ASSERT(!src1.is(src2));
00313     if (src1.code() > src2.code()) {
00314       stm(db_w, sp, src1.bit() | src2.bit(), cond);
00315     } else {
00316       str(src1, MemOperand(sp, 4, NegPreIndex), cond);
00317       str(src2, MemOperand(sp, 4, NegPreIndex), cond);
00318     }
00319   }

00632   // These instructions are generated to mark special location in the code,

00640   static inline int GetCodeMarker(Instr instr) {
00641     int dst_reg_offset = 12;
00642     int dst_mask = 0xf << dst_reg_offset;
00643     int src_mask = 0xf;
00644     int dst_reg = (instr & dst_mask) >> dst_reg_offset;
00645     int src_reg = instr & src_mask;
00646     uint32_t non_register_mask = ~(dst_mask | src_mask);
00647     uint32_t mov_mask = al | 13 << 21;
00648 
00649     // Return <n> if we have a mov rn rn, else return -1.
00650     int type = ((instr & non_register_mask) == mov_mask) &&
00651                (dst_reg == src_reg) &&
00652                (FIRST_IC_MARKER <= dst_reg) && (dst_reg < LAST_CODE_MARKER)
00653                    ? src_reg
00654                    : -1;
00655     ASSERT((type == -1) ||
00656            ((FIRST_IC_MARKER <= type) && (type < LAST_CODE_MARKER)));
00657     return type;
00658   }
