00052 // The Zone supports very fast allocation of small chunks of
00053 // memory. The chunks cannot be deallocated individually, but instead
00054 // the Zone supports deallocating all chunks in one fast
00055 // operation. The Zone is used to hold temporary data structures like
00056 // the abstract syntax tree, which is deallocated after compilation.
00057 
00058 // Note: There is no need to initialize the Zone; the first time an
00059 // allocation is attempted, a segment of memory will be requested
00060 // through a call to malloc().
00061 
00062 // Note: The implementation is inherently not thread safe. Do not use
00063 // from multi-threaded code.
00064 
00065 class Zone {
00066  public:
00067   // Allocate 'size' bytes of memory in the Zone; expands the Zone by
00068   // allocating new segments of memory on demand using malloc().
00069   inline void* New(int size);
00070 
00071   template <typename T>
00072   inline T* NewArray(int length);
00073 
00074   // Deletes all objects and free all memory allocated in the Zone. Keeps one
00075   // small (size <= kMaximumKeptSegmentSize) segment around if it finds one.
00076   void DeleteAll();
00077 
00078   // Deletes the last small segment kept around by DeleteAll().
00079   void DeleteKeptSegment();
00080 
00081   // Returns true if more memory has been allocated in zones than
00082   // the limit allows.
00083   inline bool excess_allocation();
00084 
00085   inline void adjust_segment_bytes_allocated(int delta);
00086 
00087   inline Isolate* isolate() { return isolate_; }
00088 
00089   static unsigned allocation_size_;
00090 
00091  private:
00092   friend class Isolate;
00093   friend class ZoneScope;
00094 
00095   // All pointers returned from New() have this alignment.  In addition, if the
00096   // object being allocated has a size that is divisible by 8 then its alignment
00097   // will be 8.
00098   static const int kAlignment = kPointerSize;
00099 
00100   // Never allocate segments smaller than this size in bytes.
00101   static const int kMinimumSegmentSize = 8 * KB;
00102 
00103   // Never allocate segments larger than this size in bytes.
00104   static const int kMaximumSegmentSize = 1 * MB;
00105 
00106   // Never keep segments larger than this size in bytes around.
00107   static const int kMaximumKeptSegmentSize = 64 * KB;
00108 
00109   // Report zone excess when allocation exceeds this limit.
00110   int zone_excess_limit_;
00111 
00112   // The number of bytes allocated in segments.  Note that this number
00113   // includes memory allocated from the OS but not yet allocated from
00114   // the zone.
00115   int segment_bytes_allocated_;
00116 
00117   // Each isolate gets its own zone.
00118   Zone();
00119 
00120   // Expand the Zone to hold at least 'size' more bytes and allocate
00121   // the bytes. Returns the address of the newly allocated chunk of
00122   // memory in the Zone. Should only be called if there isn't enough
00123   // room in the Zone already.
00124   Address NewExpand(int size);
00125 
00126   // Creates a new segment, sets it size, and pushes it to the front
00127   // of the segment chain. Returns the new segment.
00128   Segment* NewSegment(int size);
00129 
00130   // Deletes the given segment. Does not touch the segment chain.
00131   void DeleteSegment(Segment* segment, int size);
00132 
00133   // The free region in the current (front) segment is represented as

00136   Address position_;

00139   int scope_nesting_;
00140 
00141   Segment* segment_head_;
00142   Isolate* isolate_;

00146 // ZoneObject is an abstraction that helps define classes of objects
00147 // allocated in the Zone. Use it as a base class; see ast.h.
00148 class ZoneObject {
00149  public:
00150   // Allocate a new ZoneObject of 'size' bytes in the Zone.
00151   INLINE(void* operator new(size_t size));
00152   INLINE(void* operator new(size_t size, Zone* zone));
00153 
00154   // Ideally, the delete operator should be private instead of
00155   // public, but unfortunately the compiler sometimes synthesizes
00156   // (unused) destructors for classes derived from ZoneObject, which
00157   // require the operator to be visible. MSVC requires the delete
00158   // operator to be public.
00159 
00160   // ZoneObjects should never be deleted individually; use
00161   // Zone::DeleteAll() to delete all zone objects in one go.
00162   void operator delete(void*, size_t) { UNREACHABLE(); }
00163   void operator delete(void* pointer, Zone* zone) { UNREACHABLE(); }
00164 };

00180 // ZoneLists are growable lists with constant-time access to the
00181 // elements. The list itself and all its elements are allocated in the
00182 // Zone. ZoneLists cannot be deleted individually; you can delete all
00183 // objects in the Zone by calling Zone::DeleteAll().
00184 template<typename T>
00185 class ZoneList: public List<T, ZoneListAllocationPolicy> {
00186  public:
00187   INLINE(void* operator new(size_t size));
00188   INLINE(void* operator new(size_t size, Zone* zone));
00189 
00190   // Construct a new ZoneList with the given capacity; the length is
00191   // always zero. The capacity must be non-negative.
00192   explicit ZoneList(int capacity)
00193       : List<T, ZoneListAllocationPolicy>(capacity) { }
00194 
00195   // Construct a new ZoneList by copying the elements of the given ZoneList.
00196   explicit ZoneList(const ZoneList<T>& other)
00197       : List<T, ZoneListAllocationPolicy>(other.length()) {
00198     AddAll(other);
00199   }
00200 
00201   void operator delete(void* pointer) { UNREACHABLE(); }
00202   void operator delete(void* pointer, Zone* zone) { UNREACHABLE(); }
00203 };
