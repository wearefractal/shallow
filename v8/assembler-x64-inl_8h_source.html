00066 void Assembler::emit_code_target(Handle<Code> target,
00067                                  RelocInfo::Mode rmode,
00068                                  unsigned ast_id) {
00069   ASSERT(RelocInfo::IsCodeTarget(rmode));
00070   if (rmode == RelocInfo::CODE_TARGET && ast_id != kNoASTId) {
00071     RecordRelocInfo(RelocInfo::CODE_TARGET_WITH_ID, ast_id);
00072   } else {
00073     RecordRelocInfo(rmode);
00074   }
00075   int current = code_targets_.length();
00076   if (current > 0 && code_targets_.last().is_identical_to(target)) {
00077     // Optimization if we keep jumping to the same code target.
00078     emitl(current - 1);
00079   } else {
00080     code_targets_.Add(target);
00081     emitl(current);
00082   }
00083 }

00207   if (IsInternalReference(rmode_)) {
00208     // absolute code pointer inside code object moves with the code object.
00209     Memory::Address_at(pc_) += static_cast<int32_t>(delta);

00212     Memory::int32_at(pc_) -= static_cast<int32_t>(delta);
00213     CPU::FlushICache(pc_, sizeof(int32_t));

00335 bool RelocInfo::IsPatchedReturnSequence() {
00336   // The recognized call sequence is:
00337   //  movq(kScratchRegister, immediate64); call(kScratchRegister);
00338   // It only needs to be distinguished from a return sequence
00339   //  movq(rsp, rbp); pop(rbp); ret(n); int3 *6
00340   // The 11th byte is int3 (0xCC) in the return sequence and
00341   // REX.WB (0x48+register bit) for the call sequence.
00342 #ifdef ENABLE_DEBUGGER_SUPPORT
00343   return pc_[10] != 0xCC;
00344 #else
00345   return false;
00346 #endif
00347 }

00396 void RelocInfo::Visit(ObjectVisitor* visitor) {
00397   RelocInfo::Mode mode = rmode();
00398   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00399     visitor->VisitEmbeddedPointer(this);
00400     CPU::FlushICache(pc_, sizeof(Address));
00401   } else if (RelocInfo::IsCodeTarget(mode)) {
00402     visitor->VisitCodeTarget(this);
00403   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00404     visitor->VisitGlobalPropertyCell(this);
00405   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00406     visitor->VisitExternalReference(this);
00407     CPU::FlushICache(pc_, sizeof(Address));
00408 #ifdef ENABLE_DEBUGGER_SUPPORT
00409   // TODO(isolates): Get a cached isolate below.
00410   } else if (((RelocInfo::IsJSReturn(mode) &&
00411               IsPatchedReturnSequence()) ||
00412              (RelocInfo::IsDebugBreakSlot(mode) &&
00413               IsPatchedDebugBreakSlotSequence())) &&
00414              Isolate::Current()->debug()->has_break_points()) {
00415     visitor->VisitDebugTarget(this);
00416 #endif
00417   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00418     visitor->VisitRuntimeEntry(this);
00419   }
00420 }

00423 template<typename StaticVisitor>
00424 void RelocInfo::Visit(Heap* heap) {
00425   RelocInfo::Mode mode = rmode();
00426   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00427     StaticVisitor::VisitEmbeddedPointer(heap, this);
00428     CPU::FlushICache(pc_, sizeof(Address));
00429   } else if (RelocInfo::IsCodeTarget(mode)) {
00430     StaticVisitor::VisitCodeTarget(heap, this);
00431   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00432     StaticVisitor::VisitGlobalPropertyCell(heap, this);
00433   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00434     StaticVisitor::VisitExternalReference(this);
00435     CPU::FlushICache(pc_, sizeof(Address));
00436 #ifdef ENABLE_DEBUGGER_SUPPORT
00437   } else if (heap->isolate()->debug()->has_break_points() &&
00438              ((RelocInfo::IsJSReturn(mode) &&
00439               IsPatchedReturnSequence()) ||
00440              (RelocInfo::IsDebugBreakSlot(mode) &&
00441               IsPatchedDebugBreakSlotSequence()))) {

00443 #endif
00444   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00445     StaticVisitor::VisitRuntimeEntry(this);

00461 void Operand::set_sib(ScaleFactor scale, Register index, Register base) {
00462   ASSERT(len_ == 1);
00463   ASSERT(is_uint2(scale));
00464   // Use SIB with no index register only for base rsp or r12. Otherwise we
00465   // would skip the SIB byte entirely.
00466   ASSERT(!index.is(rsp) || base.is(rsp) || base.is(r12));
00467   buf_[1] = (scale << 6) | (index.low_bits() << 3) | base.low_bits();
00468   rex_ |= index.high_bit() << 1 | base.high_bit();
00469   len_ = 2;
00470 }
