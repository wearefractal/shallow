00037 // ----------------------------------------------------------------------------
00038 // A Handle provides a reference to an object that survives relocation by
00039 // the garbage collector.
00040 // Handles are only valid within a HandleScope.
00041 // When a handle is created for an object a cell is allocated in the heap.
00042 
00043 template<typename T>
00044 class Handle {
00045  public:
00046   INLINE(explicit Handle(T** location)) { location_ = location; }
00047   INLINE(explicit Handle(T* obj));
00048   INLINE(Handle(T* obj, Isolate* isolate));
00049 
00050   INLINE(Handle()) : location_(NULL) {}
00051 
00052   // Constructor for handling automatic up casting.
00053   // Ex. Handle<JSFunction> can be passed when Handle<Object> is expected.
00054   template <class S> Handle(Handle<S> handle) {
00055 #ifdef DEBUG
00056     T* a = NULL;
00057     S* b = NULL;
00058     a = b;  // Fake assignment to enforce type checks.
00059     USE(a);
00060 #endif
00061     location_ = reinterpret_cast<T**>(handle.location());
00062   }
00063 
00064   INLINE(T* operator ->() const) { return operator*(); }
00065 
00066   // Check if this handle refers to the exact same object as the other handle.
00067   bool is_identical_to(const Handle<T> other) const {
00068     return operator*() == *other;
00069   }
00070 
00071   // Provides the C++ dereference operator.
00072   INLINE(T* operator*() const);
00073 
00074   // Returns the address to where the raw pointer is stored.
00075   T** location() const {
00076     ASSERT(location_ == NULL ||
00077            reinterpret_cast<Address>(*location_) != kZapValue);
00078     return location_;
00079   }
00080 
00081   template <class S> static Handle<T> cast(Handle<S> that) {
00082     T::cast(*that);
00083     return Handle<T>(reinterpret_cast<T**>(that.location()));
00084   }
00085 
00086   static Handle<T> null() { return Handle<T>(); }
00087   bool is_null() const { return location_ == NULL; }
00088 
00089   // Closes the given scope, but lets this handle escape. See
00090   // implementation in api.h.
00091   inline Handle<T> EscapeFrom(v8::HandleScope* scope);
00092 
00093  private:
00094   T** location_;
00095 };

00098 // A stack-allocated class that governs a number of local handles.
00099 // After a handle scope has been created, all local handles will be
00100 // allocated within that handle scope until either the handle scope is
00101 // deleted or another handle scope is created.  If there is already a
00102 // handle scope and a new one is created, all allocations will take
00103 // place in the new handle scope until it is deleted.  After that,
00104 // new handles will again be allocated in the original handle scope.
00105 //
00106 // After the handle scope of a local handle has been deleted the
00107 // garbage collector will no longer track the object stored in the
00108 // handle and may deallocate it.  The behavior of accessing a handle
00109 // for which the handle scope has been deleted is undefined.
00110 class HandleScope {
00111  public:
00112   inline HandleScope();
00113   explicit inline HandleScope(Isolate* isolate);
00114 
00115   inline ~HandleScope();
00116 
00117   // Counts the number of allocated handles.
00118   static int NumberOfHandles();
00119 
00120   // Creates a new handle with the given value.
00121   template <typename T>
00122   static inline T** CreateHandle(T* value, Isolate* isolate);
00123 
00124   // Deallocates any extensions used by the current scope.
00125   static void DeleteExtensions(Isolate* isolate);
00126 
00127   static Address current_next_address();
00128   static Address current_limit_address();
00129   static Address current_level_address();
00130 
00131   // Closes the HandleScope (invalidating all handles
00132   // created in the scope of the HandleScope) and returns
00133   // a Handle backed by the parent scope holding the
00134   // value of the argument handle.
00135   template <typename T>
00136   Handle<T> CloseAndEscape(Handle<T> handle_value);
00137 
00138   Isolate* isolate() { return isolate_; }
00139 
00140  private:
00141   // Prevent heap allocation or illegal handle scopes.
00142   HandleScope(const HandleScope&);
00143   void operator=(const HandleScope&);
00144   void* operator new(size_t size);
00145   void operator delete(void* size_t);
00146 
00147   inline void CloseScope();
00148 
00149   Isolate* isolate_;
00150   Object** prev_next_;

00152
