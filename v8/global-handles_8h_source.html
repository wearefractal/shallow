00038 // Structure for tracking global handles.
00039 // A single list keeps all the allocated global handles.
00040 // Destroyed handles stay in the list but is added to the free list.
00041 // At GC the destroyed global handles are removed from the free list
00042 // and deallocated.

00044 // An object group is treated like a single JS object: if one of object in
00045 // the group is alive, all objects in the same group are considered alive.
00046 // An object group is used to simulate object relationship in a DOM tree.
00047 class ObjectGroup {
00048  public:
00049   static ObjectGroup* New(Object*** handles,
00050                           size_t length,
00051                           v8::RetainedObjectInfo* info) {
00052     ASSERT(length > 0);
00053     ObjectGroup* group = reinterpret_cast<ObjectGroup*>(
00054         malloc(OFFSET_OF(ObjectGroup, objects_[length])));
00055     group->length_ = length;
00056     group->info_ = info;
00057     CopyWords(group->objects_, handles, static_cast<int>(length));
00058     return group;
00059   }
00060 
00061   void Dispose() {
00062     if (info_ != NULL) info_->Dispose();
00063     free(this);
00064   }
00065 
00066   size_t length_;
00067   v8::RetainedObjectInfo* info_;
00068   Object** objects_[1];  // Variable sized array.
00069 
00070  private:
00071   void* operator new(size_t size);
00072   void operator delete(void* p);
00073   ~ObjectGroup();
00074   DISALLOW_IMPLICIT_CONSTRUCTORS(ObjectGroup);
00075 };

00078 // An implicit references group consists of two parts: a parent object and
00079 // a list of children objects.  If the parent is alive, all the children
00080 // are alive too.
00081 class ImplicitRefGroup {
00082  public:
00083   static ImplicitRefGroup* New(HeapObject** parent,
00084                                Object*** children,
00085                                size_t length) {
00086     ASSERT(length > 0);
00087     ImplicitRefGroup* group = reinterpret_cast<ImplicitRefGroup*>(
00088         malloc(OFFSET_OF(ImplicitRefGroup, children_[length])));
00089     group->parent_ = parent;
00090     group->length_ = length;
00091     CopyWords(group->children_, children, static_cast<int>(length));
00092     return group;
00093   }
00094 
00095   void Dispose() {
00096     free(this);
00097   }
00098 
00099   HeapObject** parent_;
00100   size_t length_;
00101   Object** children_[1];  // Variable sized array.

00103  private:
