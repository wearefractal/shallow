00028 // The routines exported by this module are subtle.  If you use them, even if
00029 // you get the code right, it will depend on careful reasoning about atomicity
00030 // and memory ordering; it will be less readable, and harder to maintain.  If
00031 // you plan to use these routines, you should have a good reason, such as solid
00032 // evidence that performance would otherwise suffer, or there being no
00033 // alternative.  You should assume only properties explicitly guaranteed by the
00034 // specifications in this file.  You are almost certainly _not_ writing code
00035 // just for the x86; if you assume x86 semantics, x86 hardware bugs and
00036 // implementations on other archtectures will cause your code to break.  If you
00037 // do not know what you are doing, avoid these routines, and use a Mutex.
00038 //
00039 // It is incorrect to make direct assignments to/from an atomic variable.
00040 // You should use one of the Load or Store routines.  The NoBarrier
00041 // versions are provided when no barriers are needed:
00042 //   NoBarrier_Store()
00043 //   NoBarrier_Load()
00044 // Although there are currently no compiler enforcement, you are encouraged
00045 // to use these.
00046 //

00074 // Atomically execute:
00075 //      result = *ptr;
00076 //      if (*ptr == old_value)
00077 //        *ptr = new_value;
00078 //      return result;
00079 //
00080 // I.e., replace "*ptr" with "new_value" if "*ptr" used to be "old_value".
00081 // Always return the old value of "*ptr"
00082 //
00083 // This routine implies no memory barriers.

00085                                   Atomic32 old_value,
00086                                   Atomic32 new_value);

00099 // These following lower-level operations are typically useful only to people
00100 // implementing higher-level synchronization operations like spinlocks,
00101 // mutexes, and condition-variables.  They combine CompareAndSwap(), a load, or
00102 // a store with appropriate memory-ordering instructions.  "Acquire" operations
00103 // ensure that no later memory access can be reordered ahead of the operation.
00104 // "Release" operations ensure that no previous memory access can be reordered
00105 // after the operation.  "Barrier" operations have both "Acquire" and "Release"
00106 // semantics.   A MemoryBarrier() has "Barrier" semantics, but does no memory
00107 // access.
00108 Atomic32 Acquire_CompareAndSwap(volatile Atomic32* ptr,
00109                                 Atomic32 old_value,
00110                                 Atomic32 new_value);
00111 Atomic32 Release_CompareAndSwap(volatile Atomic32* ptr,
00112                                 Atomic32 old_value,
00113                                 Atomic32 new_value);
