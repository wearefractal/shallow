00028 // The LazyInstance<Type, Traits> class manages a single instance of Type,
00029 // which will be lazily created on the first time it's accessed.  This class is
00030 // useful for places you would normally use a function-level static, but you
00031 // need to have guaranteed thread-safety.  The Type constructor will only ever
00032 // be called once, even if two threads are racing to create the object.  Get()
00033 // and Pointer() will always return the same, completely initialized instance.
00034 //
00035 // LazyInstance is completely thread safe, assuming that you create it safely.
00036 // The class was designed to be POD initialized, so it shouldn't require a
00037 // static constructor.  It really only makes sense to declare a LazyInstance as
00038 // a global variable using the LAZY_INSTANCE_INITIALIZER initializer.
00039 //
00040 // LazyInstance is similar to Singleton, except it does not have the singleton
00041 // property.  You can have multiple LazyInstance's of the same type, and each
00042 // will manage a unique instance.  It also preallocates the space for Type, as
00043 // to avoid allocating the Type instance on the heap.  This may help with the
00044 // performance of creating the instance, and reducing heap fragmentation.  This
00045 // requires that Type be a complete type so we can determine the size. See
00046 // notes for advanced users below for more explanations.
00047 //
00048 // Example usage:
00049 //   static LazyInstance<MyClass>::type my_instance = LAZY_INSTANCE_INITIALIZER;
00050 //   void SomeMethod() {
00051 //     my_instance.Get().SomeMethod();  // MyClass::SomeMethod()
00052 //
00053 //     MyClass* ptr = my_instance.Pointer();
00054 //     ptr->DoDoDo();  // MyClass::DoDoDo
00055 //   }
00056 //
00057 // Additionally you can override the way your instance is constructed by
00058 // providing your own trait:
00059 // Example usage:
00060 //   struct MyCreateTrait {
00061 //     static void Construct(MyClass* allocated_ptr) {
00062 //       new (allocated_ptr) MyClass(/* extra parameters... */);
00063 //     }
00064 //   };
00065 //   static LazyInstance<MyClass, MyCreateTrait>::type my_instance =
00066 //      LAZY_INSTANCE_INITIALIZER;
00067 //
00068 // WARNINGS:
00069 // - This implementation of LazyInstance is NOT THREAD-SAFE by default. See
00070 //   ThreadSafeInitOnceTrait declared below for that.
00071 // - Lazy initialization comes with a cost. Make sure that you don't use it on
00072 //   critical path. Consider adding your initialization code to a function
00073 //   which is explicitly called once.
00074 //
00075 // Notes for advanced users:
00076 // LazyInstance can actually be used in two different ways:
00077 //
00078 // - "Static mode" which is the default mode since it is the most efficient
00079 //   (no extra heap allocation). In this mode, the instance is statically
00080 //   allocated (stored in the global data section at compile time).
00081 //   The macro LAZY_STATIC_INSTANCE_INITIALIZER (= LAZY_INSTANCE_INITIALIZER)
00082 //   must be used to initialize static lazy instances.
00083 //
00084 // - "Dynamic mode". In this mode, the instance is dynamically allocated and
00085 //   constructed (using new) by default. This mode is useful if you have to
00086 //   deal with some code already allocating the instance for you (e.g.
00087 //   OS::Mutex() which returns a new private OS-dependent subclass of Mutex).
00088 //   The macro LAZY_DYNAMIC_INSTANCE_INITIALIZER must be used to initialize
00089 //   dynamic lazy instances.

00123 struct StaticallyAllocatedInstanceTrait {
