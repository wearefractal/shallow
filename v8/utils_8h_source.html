00055 // X must be a power of 2.  Returns the number of trailing zeros.
00056 inline int WhichPowerOf2(uint32_t x) {
00057   ASSERT(IsPowerOf2(x));
00058   ASSERT(x != 0);
00059   int bits = 0;
00060 #ifdef DEBUG
00061   int original_x = x;
00062 #endif
00063   if (x >= 0x10000) {
00064     bits += 16;
00065     x >>= 16;
00066   }
00067   if (x >= 0x100) {
00068     bits += 8;
00069     x >>= 8;
00070   }
00071   if (x >= 0x10) {
00072     bits += 4;
00073     x >>= 4;
00074   }
00075   switch (x) {
00076     default: UNREACHABLE();
00077     case 8: bits++;  // Fall through.
00078     case 4: bits++;  // Fall through.
00079     case 2: bits++;  // Fall through.
00080     case 1: break;
00081   }
00082   ASSERT_EQ(1 << bits, original_x);
00083   return bits;
00084   return 0;
00085 }

00182 // Returns the smallest power of two which is >= x. If you pass in a
00183 // number that is already a power of two, it is returned as is.
00184 // Implementation is from "Hacker's Delight" by Henry S. Warren, Jr.,
00185 // figure 3-3, page 48, where the function is called clp2.
00186 inline uint32_t RoundUpToPowerOf2(uint32_t x) {
00187   ASSERT(x <= 0x80000000u);
00188   x = x - 1;
00189   x = x | (x >> 1);
00190   x = x | (x >> 2);
00191   x = x | (x >> 4);
00192   x = x | (x >> 8);
00193   x = x | (x >> 16);
00194   return x + 1;
00195 }

00242 // BitField is a help template for encoding and decode bitfield with
00243 // unsigned content.
00244 template<class T, int shift, int size>
00245 class BitField {
00246  public:
00247   // A uint32_t mask of bit field.  To use all bits of a uint32 in a
00248   // bitfield without compiler warnings we have to compute 2^32 without
00249   // using a shift count of 32.
00250   static const uint32_t kMask = ((1U << shift) << size) - (1U << shift);
00251 
00252   // Value for the field with all bits set.
00253   static const T kMax = static_cast<T>((1U << size) - 1);
00254 
00255   // Tells whether the provided value fits into the bit field.
00256   static bool is_valid(T value) {
00257     return (static_cast<uint32_t>(value) & ~static_cast<uint32_t>(kMax)) == 0;
00258   }
00259 
00260   // Returns a uint32_t with the bit field value encoded.
00261   static uint32_t encode(T value) {
00262     ASSERT(is_valid(value));
00263     return static_cast<uint32_t>(value) << shift;
00264   }
00265 
00266   // Returns a uint32_t with the bit field value updated.
00267   static uint32_t update(uint32_t previous, T value) {
00268     return (previous & ~kMask) | encode(value);
00269   }
00270 
00271   // Extracts the bit field from the value.
00272   static T decode(uint32_t value) {
00273     return static_cast<T>((value & kMask) >> shift);
00274   }
00275 };

00283 // Thomas Wang, Integer Hash Functions.
00284 // http://www.concentric.net/~Ttwang/tech/inthash.htm
00285 inline uint32_t ComputeIntegerHash(uint32_t key, uint32_t seed) {
00286   uint32_t hash = key;
00287   hash = hash ^ seed;
00288   hash = ~hash + (hash << 15);  // hash = (hash << 15) - hash - 1;
00289   hash = hash ^ (hash >> 12);
00290   hash = hash + (hash << 2);
00291   hash = hash ^ (hash >> 4);
00292   hash = hash * 2057;  // hash = (hash + (hash << 3)) + (hash << 11);
00293   hash = hash ^ (hash >> 16);
00294   return hash;
00295 }
00296 
00297 
00298 inline uint32_t ComputeLongHash(uint64_t key) {
00299   uint64_t hash = key;
00300   hash = ~hash + (hash << 18);  // hash = (hash << 18) - hash - 1;
00301   hash = hash ^ (hash >> 31);
00302   hash = hash * 21;  // hash = (hash + (hash << 2)) + (hash << 4);
00303   hash = hash ^ (hash >> 11);
00304   hash = hash + (hash << 6);
00305   hash = hash ^ (hash >> 22);
00306   return (uint32_t) hash;
00307 }

00539 /*
00540  * A class that collects values into a backing store.
00541  * Specialized versions of the class can allow access to the backing store
00542  * in different ways.
00543  * There is no guarantee that the backing store is contiguous (and, as a
00544  * consequence, no guarantees that consecutively added elements are adjacent
00545  * in memory). The collector may move elements unless it has guaranteed not
00546  * to.
00547  */
00548 template <typename T, int growth_factor = 2, int max_growth = 1 * MB>
00549 class Collector {
00550  public:
00551   explicit Collector(int initial_capacity = kMinCapacity)
00552       : index_(0), size_(0) {
00553     current_chunk_ = Vector<T>::New(initial_capacity);
00554   }
00555 
00556   virtual ~Collector() {
00557     // Free backing store (in reverse allocation order).
00558     current_chunk_.Dispose();
00559     for (int i = chunks_.length() - 1; i >= 0; i--) {
00560       chunks_.at(i).Dispose();
00561     }
00562   }
00563 
00564   // Add a single element.
00565   inline void Add(T value) {
00566     if (index_ >= current_chunk_.length()) {
00567       Grow(1);
00568     }
00569     current_chunk_[index_] = value;
00570     index_++;
00571     size_++;
00572   }
00573 
00574   // Add a block of contiguous elements and return a Vector backed by the
00575   // memory area.
00576   // A basic Collector will keep this vector valid as long as the Collector
00577   // is alive.
00578   inline Vector<T> AddBlock(int size, T initial_value) {
00579     ASSERT(size > 0);
00580     if (size > current_chunk_.length() - index_) {
00581       Grow(size);
00582     }
00583     T* position = current_chunk_.start() + index_;
00584     index_ += size;
00585     size_ += size;
00586     for (int i = 0; i < size; i++) {
00587       position[i] = initial_value;
00588     }
00589     return Vector<T>(position, size);
00590   }

00593   // Add a contiguous block of elements and return a vector backed
00594   // by the added block.
00595   // A basic Collector will keep this vector valid as long as the Collector
00596   // is alive.
00597   inline Vector<T> AddBlock(Vector<const T> source) {
00598     if (source.length() > current_chunk_.length() - index_) {
00599       Grow(source.length());
00600     }
00601     T* position = current_chunk_.start() + index_;
00602     index_ += source.length();
00603     size_ += source.length();
00604     for (int i = 0; i < source.length(); i++) {
00605       position[i] = source[i];
00606     }

00608   }

00611   // Write the contents of the collector into the provided vector.
00612   void WriteTo(Vector<T> destination) {
00613     ASSERT(size_ <= destination.length());

00615     for (int i = 0; i < chunks_.length(); i++) {
00616       Vector<T> chunk = chunks_.at(i);
00617       for (int j = 0; j < chunk.length(); j++) {

00620       }

00623       destination[position] = current_chunk_[i];

00625     }

00644  protected:

00651   // Creates a new current chunk, and stores the old chunk in the chunks_ list.
00652   void Grow(int min_capacity) {
00653     ASSERT(growth_factor > 1);
00654     int new_capacity;
00655     int current_length = current_chunk_.length();
00656     if (current_length < kMinCapacity) {
00657       // The collector started out as empty.
00658       new_capacity = min_capacity * growth_factor;
00659       if (new_capacity < kMinCapacity) new_capacity = kMinCapacity;
00660     } else {
00661       int growth = current_length * (growth_factor - 1);
00662       if (growth > max_growth) {
00663         growth = max_growth;
00664       }
00665       new_capacity = current_length + growth;
00666       if (new_capacity < min_capacity) {
00667         new_capacity = min_capacity + growth;
00668       }
00669     }
00670     NewChunk(new_capacity);
00671     ASSERT(index_ + min_capacity <= current_chunk_.length());
00672   }

00674   // Before replacing the current chunk, give a subclass the option to move
00675   // some of the current data into the new chunk. The function may update
00676   // the current index_ value to represent data no longer in the current chunk.
00677   // Returns the initial index of the new chunk (after copied data).
00678   virtual void NewChunk(int new_capacity)  {
00679     Vector<T> new_chunk = Vector<T>::New(new_capacity);
00680     if (index_ > 0) {
00681       chunks_.Add(current_chunk_.SubVector(0, index_));
00682     } else {
00683       current_chunk_.Dispose();
00684     }
00685     current_chunk_ = new_chunk;

00691 /*
00692  * A collector that allows sequences of values to be guaranteed to
00693  * stay consecutive.
00694  * If the backing store grows while a sequence is active, the current
00695  * sequence might be moved, but after the sequence is ended, it will
00696  * not move again.
00697  * NOTICE: Blocks allocated using Collector::AddBlock(int) can move
00698  * as well, if inside an active sequence where another element is added.
00699  */
00700 template <typename T, int growth_factor = 2, int max_growth = 1 * MB>
00701 class SequenceCollector : public Collector<T, growth_factor, max_growth> {
00702  public:
00703   explicit SequenceCollector(int initial_capacity)
00704       : Collector<T, growth_factor, max_growth>(initial_capacity),
00705         sequence_start_(kNoSequence) { }
00706 
00707   virtual ~SequenceCollector() {}
00708 
00709   void StartSequence() {
00710     ASSERT(sequence_start_ == kNoSequence);
00711     sequence_start_ = this->index_;
00712   }
00713 
00714   Vector<T> EndSequence() {
00715     ASSERT(sequence_start_ != kNoSequence);
00716     int sequence_start = sequence_start_;
00717     sequence_start_ = kNoSequence;
00718     if (sequence_start == this->index_) return Vector<T>();
00719     return this->current_chunk_.SubVector(sequence_start, this->index_);
00720   }
00721 
00722   // Drops the currently added sequence, and all collected elements in it.
00723   void DropSequence() {
00724     ASSERT(sequence_start_ != kNoSequence);
00725     int sequence_length = this->index_ - sequence_start_;
00726     this->index_ = sequence_start_;
00727     this->size_ -= sequence_length;
00728     sequence_start_ = kNoSequence;
00729   }
00730 
00731   virtual void Reset() {
00732     sequence_start_ = kNoSequence;
00733     this->Collector<T, growth_factor, max_growth>::Reset();
00734   }
00735 
00736  private:
00737   static const int kNoSequence = -1;
00738   int sequence_start_;
00739 
00740   // Move the currently active sequence to the new chunk.
00741   virtual void NewChunk(int new_capacity) {
00742     if (sequence_start_ == kNoSequence) {
00743       // Fall back on default behavior if no sequence has been started.
00744       this->Collector<T, growth_factor, max_growth>::NewChunk(new_capacity);
00745       return;
00746     }
00747     int sequence_length = this->index_ - sequence_start_;
00748     Vector<T> new_chunk = Vector<T>::New(sequence_length + new_capacity);
00749     ASSERT(sequence_length < new_chunk.length());
00750     for (int i = 0; i < sequence_length; i++) {
00751       new_chunk[i] = this->current_chunk_[sequence_start_ + i];
00752     }
00753     if (sequence_start_ > 0) {
00754       this->chunks_.Add(this->current_chunk_.SubVector(0, sequence_start_));
00755     } else {
00756       this->current_chunk_.Dispose();
00757     }
00758     this->current_chunk_ = new_chunk;
00759     this->index_ = sequence_length;
00760     sequence_start_ = 0;
00761   }
00762 };

00765 // Compare ASCII/16bit chars to ASCII/16bit chars.
00766 template <typename lchar, typename rchar>
00767 inline int CompareChars(const lchar* lhs, const rchar* rhs, int chars) {
00768   const lchar* limit = lhs + chars;
00769 #ifdef V8_HOST_CAN_READ_UNALIGNED
00770   if (sizeof(*lhs) == sizeof(*rhs)) {
00771     // Number of characters in a uintptr_t.
00772     static const int kStepSize = sizeof(uintptr_t) / sizeof(*lhs);  // NOLINT
00773     while (lhs <= limit - kStepSize) {
00774       if (*reinterpret_cast<const uintptr_t*>(lhs) !=
00775           *reinterpret_cast<const uintptr_t*>(rhs)) {
00776         break;
00777       }
00778       lhs += kStepSize;
00779       rhs += kStepSize;
00780     }
00781   }
00782 #endif
00783   while (lhs < limit) {
00784     int r = static_cast<int>(*lhs) - static_cast<int>(*rhs);
00785     if (r != 0) return r;
00786     ++lhs;
00787     ++rhs;
00788   }
00789   return 0;
00790 }

00803 // The type-based aliasing rule allows the compiler to assume that pointers of
00804 // different types (for some definition of different) never alias each other.
00805 // Thus the following code does not work:
00806 //
00807 // float f = foo();
00808 // int fbits = *(int*)(&f);
00809 //
00810 // The compiler 'knows' that the int pointer can't refer to f since the types
00811 // don't match, so the compiler may cache f in a register, leaving random data
00812 // in fbits.  Using C++ style casts makes no difference, however a pointer to
00813 // char data is assumed to alias any other pointer.  This is the 'memcpy
00814 // exception'.
00815 //
00816 // Bit_cast uses the memcpy exception to move the bits from a variable of one
00817 // type of a variable of another type.  Of course the end result is likely to
00818 // be implementation dependent.  Most compilers (gcc-4.2 and MSVC 2005)
00819 // will completely optimize BitCast away.
00820 //
00821 // There is an additional use for BitCast.
00822 // Recent gccs will warn when they see casts that may result in breakage due to
00823 // the type-based aliasing rule.  If you have checked that there is no breakage
00824 // you can use BitCast to cast one pointer type to another.  This confuses gcc
00825 // enough that it can no longer see that you have cast one pointer type to
00826 // another thus avoiding the warning.
00827 
00828 // We need different implementations of BitCast for pointer and non-pointer
00829 // values. We use partial specialization of auxiliary struct to work around
00830 // issues with template functions overloading.
00831 template <class Dest, class Source>
00832 struct BitCastHelper {
00833   STATIC_ASSERT(sizeof(Dest) == sizeof(Source));
00834 
00835   INLINE(static Dest cast(const Source& source)) {
00836     Dest dest;
00837     memcpy(&dest, &source, sizeof(dest));
00838     return dest;
00839   }
00840 };
00841 
00842 template <class Dest, class Source>
00843 struct BitCastHelper<Dest, Source*> {
00844   INLINE(static Dest cast(Source* source)) {
00845     return BitCastHelper<Dest, uintptr_t>::
00846         cast(reinterpret_cast<uintptr_t>(source));
00847   }
00848 };
00849 
00850 template <class Dest, class Source>
00851 INLINE(Dest BitCast(const Source& source));
00852 
00853 template <class Dest, class Source>
00854 inline Dest BitCast(const Source& source) {
00855   return BitCastHelper<Dest, Source>::cast(source);
00856 }

00952 // A poor man's version of STL's bitset: A bit set of enums E (without explicit
00953 // values), fitting into an integral type T.
00954 template <class E, class T = int>
00955 class EnumSet {
00956  public:
00957   explicit EnumSet(T bits = 0) : bits_(bits) {}
00958   bool IsEmpty() const { return bits_ == 0; }
00959   bool Contains(E element) const { return (bits_ & Mask(element)) != 0; }
00960   bool ContainsAnyOf(const EnumSet& set) const {
00961     return (bits_ & set.bits_) != 0;
00962   }
00963   void Add(E element) { bits_ |= Mask(element); }
00964   void Add(const EnumSet& set) { bits_ |= set.bits_; }
00965   void Remove(E element) { bits_ &= ~Mask(element); }
00966   void Remove(const EnumSet& set) { bits_ &= ~set.bits_; }
00967   void RemoveAll() { bits_ = 0; }
00968   void Intersect(const EnumSet& set) { bits_ &= set.bits_; }
00969   T ToIntegral() const { return bits_; }
00970   bool operator==(const EnumSet& set) { return bits_ == set.bits_; }
00971 
00972  private:
00973   T Mask(E element) const {
00974     // The strange typing in ASSERT is necessary to avoid stupid warnings, see:
00975     // http://gcc.gnu.org/bugzilla/show_bug.cgi?id=43680
00976     ASSERT(element < static_cast<int>(sizeof(T) * CHAR_BIT));
00977     return 1 << element;
00978   }
00979 
00980   T bits_;
00981 };
