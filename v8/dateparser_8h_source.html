00089     // Read a string of digits as an unsigned number. Cap value at
00090     // kMaxSignificantDigits, but skip remaining digits if the numeral
00091     // is longer.
00092     int ReadUnsignedNumeral() {
00093       int n = 0;
00094       int i = 0;
00095       while (IsAsciiDigit()) {
00096         if (i < kMaxSignificantDigits) n = n * 10 + ch_ - '0';
00097         i++;
00098         Next();
00099       }
00100       return n;
00101     }

00103     // Read a word (sequence of chars. >= 'A'), fill the given buffer with a
00104     // lower-case prefix, and pad any remainder of the buffer with zeroes.
00105     // Return word length.
00106     int ReadWord(uint32_t* prefix, int prefix_size) {
00107       int len;
00108       for (len = 0; IsAsciiAlphaOrAbove(); Next(), len++) {
00109         if (len < prefix_size) prefix[len] = AsciiAlphaToLower(ch_);
00110       }
00111       for (int i = len; i < prefix_size; i++) prefix[i] = 0;
00112       return len;
00113     }

00115     // The skip methods return whether they actually skipped something.
00116     bool Skip(uint32_t c) {
00117       if (ch_ == c) {
00118         Next();
00119         return true;
00120       }
00121       return false;
00122     }

00124     bool SkipWhiteSpace() {
00125       if (unicode_cache_->IsWhiteSpace(ch_)) {
00126         Next();
00127         return true;
00128       }
00129       return false;

00132     bool SkipParentheses() {
00133       if (ch_ != '(') return false;
00134       int balance = 0;
00135       do {
00136         if (ch_ == ')') --balance;
00137         else if (ch_ == '(') ++balance;
00138         Next();
00139       } while (balance > 0 && ch_);
00140       return true;
00141     }
