00028 // This file defines all of the flags.  It is separated into different section,
00029 // for Debug, Release, Logging and Profiling, etc.  To add a new flag, find the
00030 // correct section, and use one of the DEFINE_ macros, without a trailing ';'.
00031 //
00032 // This include does not have a guard, because it is a template-style include,
00033 // which can be included multiple times in different modes.  It expects to have
00034 // a mode defined before it's included.  The modes are FLAG_MODE_... below:
00035 
00036 // We want to declare the names of the variables for the header file.  Normally
00037 // this will just be an extern declaration, but for a readonly flag we let the
00038 // compiler make better optimizations by giving it the value.
00039 #if defined(FLAG_MODE_DECLARE)
00040 #define FLAG_FULL(ftype, ctype, nam, def, cmt) \
00041   extern ctype FLAG_##nam;
00042 #define FLAG_READONLY(ftype, ctype, nam, def, cmt) \
00043   static ctype const FLAG_##nam = def;
00044 #define DEFINE_implication(whenflag, thenflag)
00045 
00046 // We want to supply the actual storage and value for the flag variable in the
00047 // .cc file.  We only do this for writable flags.
00048 #elif defined(FLAG_MODE_DEFINE)
00049 #define FLAG_FULL(ftype, ctype, nam, def, cmt) \
00050   ctype FLAG_##nam = def;
00051 #define FLAG_READONLY(ftype, ctype, nam, def, cmt)
00052 #define DEFINE_implication(whenflag, thenflag)
00053 
00054 // We need to define all of our default values so that the Flag structure can
00055 // access them by pointer.  These are just used internally inside of one .cc,
00056 // for MODE_META, so there is no impact on the flags interface.
00057 #elif defined(FLAG_MODE_DEFINE_DEFAULTS)
00058 #define FLAG_FULL(ftype, ctype, nam, def, cmt) \
00059   static ctype const FLAGDEFAULT_##nam = def;
00060 #define FLAG_READONLY(ftype, ctype, nam, def, cmt)
00061 #define DEFINE_implication(whenflag, thenflag)
00062 
00063 // We want to write entries into our meta data table, for internal parsing and
00064 // printing / etc in the flag parser code.  We only do this for writable flags.
00065 #elif defined(FLAG_MODE_META)
00066 #define FLAG_FULL(ftype, ctype, nam, def, cmt) \
00067   { Flag::TYPE_##ftype, #nam, &FLAG_##nam, &FLAGDEFAULT_##nam, cmt, false },
00068 #define FLAG_READONLY(ftype, ctype, nam, def, cmt)
00069 #define DEFINE_implication(whenflag, thenflag)
00070 
00071 // We produce the code to set flags when it is implied by another flag.
00072 #elif defined(FLAG_MODE_DEFINE_IMPLICATIONS)
00073 #define FLAG_FULL(ftype, ctype, nam, def, cmt)
00074 #define FLAG_READONLY(ftype, ctype, nam, def, cmt)
00075 #define DEFINE_implication(whenflag, thenflag) \
00076   if (FLAG_##whenflag) FLAG_##thenflag = true;
00077 
00078 #else
00079 #error No mode supplied when including flags.defs
00080 #endif

00083 // Structure used to hold a collection of arguments to the JavaScript code.
00084 #define JSARGUMENTS_INIT {{}}
00085 struct JSArguments {
00086 public:
00087   inline int argc() const {
00088     return static_cast<int>(storage_[0]);
00089   }
00090   inline const char** argv() const {
00091     return reinterpret_cast<const char**>(storage_[1]);
00092   }
00093   inline const char*& operator[] (int idx) const {
00094     return argv()[idx];
00095   }
00096   inline JSArguments& operator=(JSArguments args) {
00097     set_argc(args.argc());
00098     set_argv(args.argv());
00099     return *this;
00100   }
00101   static JSArguments Create(int argc, const char** argv) {
00102     JSArguments args;
00103     args.set_argc(argc);
00104     args.set_argv(argv);
00105     return args;
00106   }
00107 private:
00108   void set_argc(int argc) {
00109     storage_[0] = argc;
00110   }
00111   void set_argv(const char** argv) {
00112     storage_[1] = reinterpret_cast<AtomicWord>(argv);
00113   }
00114 public:
00115   // Contains argc and argv. Unfortunately we have to store these two fields
00116   // into a single one to avoid making the initialization macro (which would be
00117   // "{ 0, NULL }") contain a coma.
00118   AtomicWord storage_[2];
00119 };
00120 #endif
