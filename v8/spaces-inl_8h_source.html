00109 HeapObject* HeapObjectIterator::FromCurrentPage() {
00110   while (cur_addr_ != cur_end_) {
00111     if (cur_addr_ == space_->top() && cur_addr_ != space_->limit()) {
00112       cur_addr_ = space_->limit();
00113       continue;
00114     }
00115     HeapObject* obj = HeapObject::FromAddress(cur_addr_);
00116     int obj_size = (size_func_ == NULL) ? obj->Size() : size_func_(obj);
00117     cur_addr_ += obj_size;
00118     ASSERT(cur_addr_ <= cur_end_);
00119     if (!obj->IsFiller()) {
00120       ASSERT_OBJECT_SIZE(obj_size);
00121       return obj;
00122     }
00123   }
00124   return NULL;

00197 MemoryChunk* MemoryChunk::FromAnyPointerAddress(Address addr) {
00198   MemoryChunk* maybe = reinterpret_cast<MemoryChunk*>(
00199       OffsetFrom(addr) & ~Page::kPageAlignmentMask);
00200   if (maybe->owner() != NULL) return maybe;
00201   LargeObjectIterator iterator(HEAP->lo_space());
00202   for (HeapObject* o = iterator.Next(); o != NULL; o = iterator.Next()) {
00203     // Fixed arrays are the only pointer-containing objects in large object
00204     // space.
00205     if (o->IsFixedArray()) {
00206       MemoryChunk* chunk = MemoryChunk::FromAddress(o->address());
00207       if (chunk->Contains(addr)) {
00208         return chunk;
00209       }
00210     }
00211   }
00212   UNREACHABLE();
00213   return NULL;

00248 // Try linear allocation in the page of alloc_info's allocation top.  Does
00249 // not contain slow case logic (e.g. move to the next page or try free list
00250 // allocation) so it can be used by all the allocation functions and for all
00251 // the paged spaces.
00252 HeapObject* PagedSpace::AllocateLinearly(int size_in_bytes) {
00253   Address current_top = allocation_info_.top;
00254   Address new_top = current_top + size_in_bytes;
00255   if (new_top > allocation_info_.limit) return NULL;
00256 
00257   allocation_info_.top = new_top;
00258   return HeapObject::FromAddress(current_top);
00259 }

00262 // Raw allocation.
00263 MaybeObject* PagedSpace::AllocateRaw(int size_in_bytes) {
00264   HeapObject* object = AllocateLinearly(size_in_bytes);
00265   if (object != NULL) {
00266     if (identity() == CODE_SPACE) {
00267       SkipList::Update(object->address(), size_in_bytes);
00268     }
00269     return object;
00270   }
00271 
00272   object = free_list_.Allocate(size_in_bytes);
00273   if (object != NULL) {
00274     if (identity() == CODE_SPACE) {
00275       SkipList::Update(object->address(), size_in_bytes);
00276     }
00277     return object;
00278   }
00279 
00280   object = SlowAllocateRaw(size_in_bytes);
00281   if (object != NULL) {
00282     if (identity() == CODE_SPACE) {
00283       SkipList::Update(object->address(), size_in_bytes);
00284     }
00285     return object;
00286   }
00287 
00288   return Failure::RetryAfterGC(identity());
00289 }

00321 template <typename StringType>
00322 void NewSpace::ShrinkStringAtAllocationBoundary(String* string, int length) {
00323   ASSERT(length <= string->length());
00324   ASSERT(string->IsSeqString());
00325   ASSERT(string->address() + StringType::SizeFor(string->length()) ==
00326          allocation_info_.top);
00327   Address old_top = allocation_info_.top;
00328   allocation_info_.top =
00329       string->address() + StringType::SizeFor(length);
00330   string->set_length(length);
00331   if (Marking::IsBlack(Marking::MarkBitFrom(string))) {
00332     int delta = static_cast<int>(old_top - allocation_info_.top);
00333     MemoryChunk::IncrementLiveBytesFromMutator(string->address(), -delta);
00334   }
