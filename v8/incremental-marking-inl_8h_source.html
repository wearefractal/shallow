00037 bool IncrementalMarking::BaseRecordWrite(HeapObject* obj,
00038                                          Object** slot,
00039                                          Object* value) {
00040   MarkBit value_bit = Marking::MarkBitFrom(HeapObject::cast(value));
00041   if (Marking::IsWhite(value_bit)) {
00042     MarkBit obj_bit = Marking::MarkBitFrom(obj);
00043     if (Marking::IsBlack(obj_bit)) {
00044       BlackToGreyAndUnshift(obj, obj_bit);
00045       RestartIfNotMarking();
00046     }
00047 
00048     // Object is either grey or white.  It will be scanned if survives.
00049     return false;
00050   }
00051   return true;
00052 }

00091 void IncrementalMarking::BlackToGreyAndUnshift(HeapObject* obj,
00092                                                MarkBit mark_bit) {
00093   ASSERT(Marking::MarkBitFrom(obj) == mark_bit);
00094   ASSERT(obj->Size() >= 2*kPointerSize);
00095   ASSERT(IsMarking());
00096   Marking::BlackToGrey(mark_bit);
00097   int obj_size = obj->Size();
00098   MemoryChunk::IncrementLiveBytesFromGC(obj->address(), -obj_size);
00099   bytes_scanned_ -= obj_size;
00100   int64_t old_bytes_rescanned = bytes_rescanned_;
00101   bytes_rescanned_ = old_bytes_rescanned + obj_size;
00102   if ((bytes_rescanned_ >> 20) != (old_bytes_rescanned >> 20)) {
00103     if (bytes_rescanned_ > 2 * heap_->PromotedSpaceSize()) {
00104       // If we have queued twice the heap size for rescanning then we are
00105       // going around in circles, scanning the same objects again and again
00106       // as the program mutates the heap faster than we can incrementally
00107       // trace it.  In this case we switch to non-incremental marking in
00108       // order to finish off this marking phase.
00109       if (FLAG_trace_gc) {
00110         PrintF("Hurrying incremental marking because of lack of progress\n");
00111       }
00112       allocation_marking_factor_ = kMaxAllocationMarkingFactor;
00113     }
00114   }
00115 
00116   marking_deque_.UnshiftGrey(obj);
00117 }
