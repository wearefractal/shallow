00091 // Iterates the elements in every lol and returns the one that matches the
00092 // specified key.  If no matching element is found, then it returns NULL.
00093 template <typename T>
00094 inline LiveObjectList::Element*
00095 LiveObjectList::FindElementFor(T (*GetValue)(LiveObjectList::Element*), T key) {
00096   LiveObjectList* lol = last();
00097   while (lol != NULL) {
00098     Element* elements = lol->elements_;
00099     for (int i = 0; i < lol->obj_count_; i++) {
00100       Element* element = &elements[i];
00101       if (GetValue(element) == key) {
00102         return element;
00103       }
00104     }
00105     lol = lol->prev_;
00106   }
00107   return NULL;
00108 }
