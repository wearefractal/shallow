00048 template <class T, int s> int Mapping<T, s>::get(uchar c, uchar n,
00049     uchar* result) {
00050   CacheEntry entry = entries_[c & kMask];
00051   if (entry.code_point_ == c) {
00052     if (entry.offset_ == 0) {

00063 template <class T, int s> int Mapping<T, s>::CalculateValue(uchar c, uchar n,
00064     uchar* result) {
00065   bool allow_caching = true;
00066   int length = T::Convert(c, n, result, &allow_caching);
00067   if (allow_caching) {
00068     if (length == 1) {
00069       entries_[c & kMask] = CacheEntry(c, result[0] - c);
00070       return 1;
00071     } else {
00072       entries_[c & kMask] = CacheEntry(c, 0);
00073       return 0;
00074     }
00075   } else {
00076     return length;
00077   }
00078 }

00081 unsigned Utf8::Encode(char* str, uchar c, int previous) {
00082   static const int kMask = ~(1 << 6);
00083   if (c <= kMaxOneByteChar) {
00084     str[0] = c;
00085     return 1;
00086   } else if (c <= kMaxTwoByteChar) {
00087     str[0] = 0xC0 | (c >> 6);
00088     str[1] = 0x80 | (c & kMask);
00089     return 2;
00090   } else if (c <= kMaxThreeByteChar) {
00091     if (Utf16::IsTrailSurrogate(c) &&
00092         Utf16::IsLeadSurrogate(previous)) {
00093       const int kUnmatchedSize = kSizeOfUnmatchedSurrogate;
00094       return Encode(str - kUnmatchedSize,
00095                     Utf16::CombineSurrogatePair(previous, c),
00096                     Utf16::kNoPreviousCharacter) - kUnmatchedSize;
00097     }
00098     str[0] = 0xE0 | (c >> 12);
00099     str[1] = 0x80 | ((c >> 6) & kMask);
00100     str[2] = 0x80 | (c & kMask);
00101     return 3;
00102   } else {
00103     str[0] = 0xF0 | (c >> 18);
00104     str[1] = 0x80 | ((c >> 12) & kMask);
00105     str[2] = 0x80 | ((c >> 6) & kMask);
00106     str[3] = 0x80 | (c & kMask);
00107     return 4;
00108   }
00109 }

00112 uchar Utf8::ValueOf(const byte* bytes, unsigned length, unsigned* cursor) {
00113   if (length <= 0) return kBadChar;
00114   byte first = bytes[0];
00115   // Characters between 0000 and 0007F are encoded as a single character
00116   if (first <= kMaxOneByteChar) {
00117     *cursor += 1;
00118     return first;
00119   }
00120   return CalculateValue(bytes, length, cursor);

00123 unsigned Utf8::Length(uchar c, int previous) {
00124   if (c <= kMaxOneByteChar) {
00125     return 1;
00126   } else if (c <= kMaxTwoByteChar) {
00127     return 2;
00128   } else if (c <= kMaxThreeByteChar) {
00129     if (Utf16::IsTrailSurrogate(c) &&
00130         Utf16::IsLeadSurrogate(previous)) {
00131       return kSizeOfUnmatchedSurrogate - kBytesSavedByCombiningSurrogates;
00132     }
00133     return 3;
00134   } else {
00135     return 4;

00168 bool CharacterStream::EncodeNonAsciiCharacter(uchar c, byte* buffer,
00169     unsigned capacity, unsigned& offset) {
00170   unsigned aligned = (offset + 0x3) & ~0x3;
00171   if ((aligned + sizeof(uchar)) > capacity)
00172     return false;
00173   if (offset == aligned) {
00174     IF_LITTLE(*reinterpret_cast<uchar*>(buffer + aligned) = (c << 8) | 0x80);
00175     IF_BIG(*reinterpret_cast<uchar*>(buffer + aligned) = c | (1 << 31));
00176   } else {
00177     buffer[offset] = 0x80;
00178     IF_LITTLE(*reinterpret_cast<uchar*>(buffer + aligned) = c << 8);
00179     IF_BIG(*reinterpret_cast<uchar*>(buffer + aligned) = c);
00180   }
00181   offset = aligned + sizeof(uchar);
00182   return true;
00183 }

00194 uchar CharacterStream::DecodeCharacter(const byte* buffer, unsigned* offset) {
00195   byte b = buffer[*offset];
00196   if (b <= Utf8::kMaxOneByteChar) {
00197     (*offset)++;
00198     return b;
00199   } else {
00200     unsigned aligned = (*offset + 0x3) & ~0x3;
00201     *offset = aligned + sizeof(uchar);
00202     IF_LITTLE(return *reinterpret_cast<const uchar*>(buffer + aligned) >> 8);
00203     IF_BIG(return *reinterpret_cast<const uchar*>(buffer + aligned) &
00204                     ~(1 << 31));
00205   }
00206 }
