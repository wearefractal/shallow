00035 // The ARM ABI does not specify the usage of register r9, which may be reserved

00062 // Callee-saved registers preserved when switching from C to JavaScript
00063 const RegList kCalleeSaved =
00064   1 <<  4 |  //  r4 v1
00065   1 <<  5 |  //  r5 v2
00066   1 <<  6 |  //  r6 v3
00067   1 <<  7 |  //  r7 v4
00068   1 <<  8 |  //  r8 v5 (cp in JavaScript code)
00069   kR9Available <<  9 |  //  r9 v6
00070   1 << 10 |  // r10 v7
00071   1 << 11;   // r11 v8 (fp in JavaScript code)
00072 
00073 // When calling into C++ (only for C++ calls that can't cause a GC).
00074 // The call code will take care of lr, fp, etc.
00075 const RegList kCallerSaved =
00076   1 <<  0 |  // r0
00077   1 <<  1 |  // r1
00078   1 <<  2 |  // r2
00079   1 <<  3 |  // r3
00080   1 <<  9;   // r9
