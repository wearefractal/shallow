00035 class StringAllocator {
00036  public:
00037   virtual ~StringAllocator() {}
00038   // Allocate a number of bytes.
00039   virtual char* allocate(unsigned bytes) = 0;
00040   // Allocate a larger number of bytes and copy the old buffer to the new one.
00041   // bytes is an input and output parameter passing the old size of the buffer
00042   // and returning the new size.  If allocation fails then we return the old
00043   // buffer and do not increase the size.
00044   virtual char* grow(unsigned* bytes) = 0;
00045 };
