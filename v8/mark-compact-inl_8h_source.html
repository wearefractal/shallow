00055 void MarkCompactCollector::MarkObject(HeapObject* obj, MarkBit mark_bit) {
00056   ASSERT(Marking::MarkBitFrom(obj) == mark_bit);
00057   if (!mark_bit.Get()) {
00058     mark_bit.Set();
00059     MemoryChunk::IncrementLiveBytesFromGC(obj->address(), obj->Size());
00060     ProcessNewlyMarkedObject(obj);
00061   }
00062 }

00078 void MarkCompactCollector::SetMark(HeapObject* obj, MarkBit mark_bit) {
00079   ASSERT(!mark_bit.Get());
00080   ASSERT(Marking::MarkBitFrom(obj) == mark_bit);
00081   mark_bit.Set();
00082   MemoryChunk::IncrementLiveBytesFromGC(obj->address(), obj->Size());
00083   if (obj->IsMap()) {
00084     heap_->ClearCacheOnMap(Map::cast(obj));
00085   }
