00045 // Preparsing checks a JavaScript program and emits preparse-data that helps
00046 // a later parsing to be faster.
00047 // See preparse-data-format.h for the data format.
00048 
00049 // The PreParser checks that the syntax follows the grammar for JavaScript,
00050 // and collects some information about the program along the way.
00051 // The grammar check is only performed in order to understand the program
00052 // sufficiently to deduce some information about it, that can be used
00053 // to speed up later parsing. Finding errors is not the goal of pre-parsing,
00054 // rather it is to speed up properly written and correct programs.
00055 // That means that contextual checks (like a label being declared where
00056 // it is used) are generally omitted.

00163   // Used to detect duplicates in object literals. Each of the values
00164   // kGetterProperty, kSetterProperty and kValueProperty represents
00165   // a type of object literal property. When parsing a property, its
00166   // type value is stored in the DuplicateFinder for the property name.
00167   // Values are chosen so that having intersection bits means the there is
00168   // an incompatibility.
00169   // I.e., you can add a getter to a property that already has a setter, since
00170   // kGetterProperty and kSetterProperty doesn't intersect, but not if it
00171   // already has a getter or a value. Adding the getter to an existing
00172   // setter will store the value (kGetterProperty | kSetterProperty), which
00173   // is incompatible with adding any further properties.
00174   enum PropertyType {
00175     kNone = 0,
00176     // Bit patterns representing different object literal property types.
00177     kGetterProperty = 1,
00178     kSetterProperty = 2,
00179     kValueProperty = 7,
00180     // Helper constants.
00181     kValueFlag = 4
00182   };

00344     Expression Parenthesize() {
00345       int type = code_ & 3;
00346       if (type != 0) {
00347         // Identifiers and string literals can be parenthesized.
00348         // They no longer work as labels or directive prologues,
00349         // but are still recognized in other contexts.
00350         return Expression(code_ | kParentesizedExpressionFlag);
00351       }
00352       // For other types of expressions, it's not important to remember
00353       // the parentheses.
00354       return *this;
00355     }
