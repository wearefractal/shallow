00077 // Switches between all the Api tests using the threading support.
00078 // In order to get a surprising but repeatable pattern of thread
00079 // switching it has extra semaphores to control the order in which
00080 // the tests alternate, not relying solely on the big V8 lock.
00081 //
00082 // A test is augmented with calls to ApiTestFuzzer::Fuzz() in its
00083 // callbacks.  This will have no effect when we are not running the
00084 // thread fuzzing test.  In the thread fuzzing test it will
00085 // pseudorandomly select a successor thread and switch execution
00086 // to that thread, suspending the current test.
00087 class ApiTestFuzzer: public v8::internal::Thread {
00088  public:
00089   void CallTest();
00090   explicit ApiTestFuzzer(int num)
00091       : Thread("ApiTestFuzzer"),
00092         test_number_(num),
00093         gate_(v8::internal::OS::CreateSemaphore(0)),
00094         active_(true) {
00095   }
00096   ~ApiTestFuzzer() { delete gate_; }

00098   // The ApiTestFuzzer is also a Thread, so it has a Run method.
00099   virtual void Run();

00135 class RegisterThreadedTest {
00136  public:
00137   explicit RegisterThreadedTest(CcTest::TestFunction* callback,
00138                                 const char* name)
00139       : fuzzer_(NULL), callback_(callback), name_(name) {
00140     prev_ = first_;
00141     first_ = this;
00142     count_++;
00143   }
00144   static int count() { return count_; }
00145   static RegisterThreadedTest* nth(int i) {
00146     CHECK(i < count());
00147     RegisterThreadedTest* current = first_;
00148     while (i > 0) {
00149       i--;
00150       current = current->prev_;
00151     }
00152     return current;
00153   }
00154   CcTest::TestFunction* callback() { return callback_; }
00155   ApiTestFuzzer* fuzzer_;
