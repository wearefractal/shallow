00038 // Atomically execute:
00039 //      result = *ptr;
00040 //      if (*ptr == old_value)
00041 //        *ptr = new_value;
00042 //      return result;
00043 //
00044 // I.e., replace "*ptr" with "new_value" if "*ptr" used to be "old_value".
00045 // Always return the old value of "*ptr"
00046 //
00047 // This routine implies no memory barriers.
00048 inline Atomic32 NoBarrier_CompareAndSwap(volatile Atomic32* ptr,
00049                                          Atomic32 old_value,
00050                                          Atomic32 new_value) {
00051   Atomic32 prev, tmp;
00052   __asm__ __volatile__(".set push\n"
00053                        ".set noreorder\n"
00054                        "1:\n"
00055                        "ll %0, %5\n"  // prev = *ptr
00056                        "bne %0, %3, 2f\n"  // if (prev != old_value) goto 2
00057                        "move %2, %4\n"  // tmp = new_value
00058                        "sc %2, %1\n"  // *ptr = tmp (with atomic check)
00059                        "beqz %2, 1b\n"  // start again on atomic error
00060                        "nop\n"  // delay slot nop
00061                        "2:\n"
00062                        ".set pop\n"
00063                        : "=&r" (prev), "=m" (*ptr), "=&r" (tmp)
00064                        : "Ir" (old_value), "r" (new_value), "m" (*ptr)
00065                        : "memory");
00066   return prev;
00067 }

00069 // Atomically store new_value into *ptr, returning the previous value held in
00070 // *ptr.  This routine implies no memory barriers.
00071 inline Atomic32 NoBarrier_AtomicExchange(volatile Atomic32* ptr,
00072                                          Atomic32 new_value) {
00073   Atomic32 temp, old;

00075                        ".set noreorder\n"
00076                        "1:\n"
00077                        "ll %1, %2\n"  // old = *ptr

00081                        "nop\n"  // delay slot nop

00090 // Atomically increment *ptr by "increment".  Returns the new value of
00091 // *ptr with the increment applied.  This routine implies no memory barriers.
00092 inline Atomic32 NoBarrier_AtomicIncrement(volatile Atomic32* ptr,
00093                                           Atomic32 increment) {
00094   Atomic32 temp, temp2;
00095 
00096   __asm__ __volatile__(".set push\n"
00097                        ".set noreorder\n"
00098                        "1:\n"
00099                        "ll %0, %2\n"  // temp = *ptr
00100                        "addu %1, %0, %3\n"  // temp2 = temp + increment
00101                        "sc %1, %2\n"  // *ptr = temp2 (with atomic check)
00102                        "beqz %1, 1b\n"  // start again on atomic error
00103                        "addu %1, %0, %3\n"  // temp2 = temp + increment
00104                        ".set pop\n"
00105                        : "=&r" (temp), "=&r" (temp2), "=m" (*ptr)
00106                        : "Ir" (increment), "m" (*ptr)
00107                        : "memory");
00108   // temp2 now holds the final value.
00109   return temp2;
00110 }

00120 // "Acquire" operations
00121 // ensure that no later memory access can be reordered ahead of the operation.
00122 // "Release" operations ensure that no previous memory access can be reordered
00123 // after the operation.  "Barrier" operations have both "Acquire" and "Release"
00124 // semantics.   A MemoryBarrier() has "Barrier" semantics, but does no memory
00125 // access.
00126 inline Atomic32 Acquire_CompareAndSwap(volatile Atomic32* ptr,
00127                                        Atomic32 old_value,
00128                                        Atomic32 new_value) {
00129   ATOMICOPS_COMPILER_BARRIER();
00130   Atomic32 res = NoBarrier_CompareAndSwap(ptr, old_value, new_value);
00131   ATOMICOPS_COMPILER_BARRIER();
00132   return res;
