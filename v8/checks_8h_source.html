00175 // Helper function used by the CHECK function when given floating
00176 // point arguments.  Should not be called directly.
00177 inline void CheckEqualsHelper(const char* file,
00178                               int line,
00179                               const char* expected_source,
00180                               double expected,
00181                               const char* value_source,
00182                               double value) {
00183   // Force values to 64 bit memory to truncate 80 bit precision on IA32.
00184   volatile double* exp = new double[1];
00185   *exp = expected;
00186   volatile double* val = new double[1];
00187   *val = value;
00188   if (*exp != *val) {
00189     V8_Fatal(file, line,
00190              "CHECK_EQ(%s, %s) failed\n#   Expected: %f\n#   Found: %f",
00191              expected_source, value_source, *exp, *val);
00192   }
00193   delete[] exp;
00194   delete[] val;
00195 }

00257 // The ASSERT macro is equivalent to CHECK except that it only
00258 // generates code in debug builds.
00259 #ifdef DEBUG
00260 #define ASSERT_RESULT(expr)    CHECK(expr)
00261 #define ASSERT(condition)      CHECK(condition)
00262 #define ASSERT_EQ(v1, v2)      CHECK_EQ(v1, v2)
00263 #define ASSERT_NE(v1, v2)      CHECK_NE(v1, v2)
00264 #define ASSERT_GE(v1, v2)      CHECK_GE(v1, v2)
00265 #define ASSERT_LT(v1, v2)      CHECK_LT(v1, v2)
00266 #define ASSERT_LE(v1, v2)      CHECK_LE(v1, v2)
00267 #define SLOW_ASSERT(condition) CHECK(!FLAG_enable_slow_asserts || (condition))
00268 #else
00269 #define ASSERT_RESULT(expr)    (expr)
00270 #define ASSERT(condition)      ((void) 0)
00271 #define ASSERT_EQ(v1, v2)      ((void) 0)
00272 #define ASSERT_NE(v1, v2)      ((void) 0)
00273 #define ASSERT_GE(v1, v2)      ((void) 0)
00274 #define ASSERT_LT(v1, v2)      ((void) 0)
00275 #define ASSERT_LE(v1, v2)      ((void) 0)
00276 #define SLOW_ASSERT(condition) ((void) 0)
00277 #endif
00278 // Static asserts has no impact on runtime performance, so they can be
00279 // safely enabled in release mode. Moreover, the ((void) 0) expression
00280 // obeys different syntax rules than typedef's, e.g. it can't appear
00281 // inside class declaration, this leads to inconsistency between debug
00282 // and release compilation modes behavior.
