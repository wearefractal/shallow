00102   // Integer32 is an integer that can be represented as a signed
00103   // 32-bit integer. It has to be
00104   // in the range [-2^31, 2^31 - 1]. We also have to check for negative 0
00105   // as it is not an Integer32.
00106   static inline bool IsInt32Double(double value) {
00107     const DoubleRepresentation minus_zero(-0.0);
00108     DoubleRepresentation rep(value);
00109     if (rep.bits == minus_zero.bits) return false;
00110     if (value >= kMinInt && value <= kMaxInt &&
00111         value == static_cast<int32_t>(value)) {
00112       return true;
00113     }
00114     return false;
00115   }
