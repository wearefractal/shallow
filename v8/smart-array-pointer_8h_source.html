00035 // A 'scoped array pointer' that calls DeleteArray on its pointer when the
00036 // destructor is called.
00037 template<typename T>
00038 class SmartArrayPointer {
00039  public:
00040   // Default constructor. Constructs an empty scoped pointer.
00041   inline SmartArrayPointer() : p_(NULL) {}
00042 
00043   // Constructs a scoped pointer from a plain one.
00044   explicit inline SmartArrayPointer(T* ptr) : p_(ptr) {}
00045 
00046   // Copy constructor removes the pointer from the original to avoid double
00047   // freeing.
00048   inline SmartArrayPointer(const SmartArrayPointer<T>& rhs) : p_(rhs.p_) {
00049     const_cast<SmartArrayPointer<T>&>(rhs).p_ = NULL;
00050   }
00051 
00052   // When the destructor of the scoped pointer is executed the plain pointer
00053   // is deleted using DeleteArray.  This implies that you must allocate with
00054   // NewArray.
00055   inline ~SmartArrayPointer() { if (p_) DeleteArray(p_); }
00056 
00057   inline T* operator->() const { return p_; }
00058 
00059   // You can get the underlying pointer out with the * operator.
00060   inline T* operator*() { return p_; }
00061 
00062   // You can use [n] to index as if it was a plain pointer
00063   inline T& operator[](size_t i) {
00064     return p_[i];
00065   }
00066 
00067   // We don't have implicit conversion to a T* since that hinders migration:
00068   // You would not be able to change a method from returning a T* to
00069   // returning an SmartArrayPointer<T> and then get errors wherever it is used.
00070 
00071 
00072   // If you want to take out the plain pointer and don't want it automatically
00073   // deleted then call Detach().  Afterwards, the smart pointer is empty
00074   // (NULL).
00075   inline T* Detach() {
00076     T* temp = p_;
00077     p_ = NULL;
00078     return temp;
00079   }
00080 
00081   // Assignment requires an empty (NULL) SmartArrayPointer as the receiver. Like
00082   // the copy constructor it removes the pointer in the original to avoid
00083   // double freeing.
00084   inline SmartArrayPointer& operator=(const SmartArrayPointer<T>& rhs) {
00085     ASSERT(is_empty());
00086     T* tmp = rhs.p_;  // swap to handle self-assignment
00087     const_cast<SmartArrayPointer<T>&>(rhs).p_ = NULL;
00088     p_ = tmp;
00089     return *this;
00090   }
00091 
00092   inline bool is_empty() { return p_ == NULL; }
00093 
00094  private:
00095   T* p_;
00096 };
