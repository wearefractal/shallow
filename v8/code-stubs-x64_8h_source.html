00510   static Mode GetMode(Code* stub) {
00511     byte first_instruction = stub->instruction_start()[0];
00512     byte second_instruction = stub->instruction_start()[2];
00513 
00514     if (first_instruction == kTwoByteJumpInstruction) {
00515       return INCREMENTAL;
00516     }

00519 
00520     if (second_instruction == kFiveByteJumpInstruction) {
00521       return INCREMENTAL_COMPACTION;

00524     ASSERT(second_instruction == kFiveByteNopInstruction);

00529   static void Patch(Code* stub, Mode mode) {
00530     switch (mode) {

00533                GetMode(stub) == INCREMENTAL_COMPACTION);

00535         stub->instruction_start()[2] = kFiveByteNopInstruction;
00536         break;
00537       case INCREMENTAL:

00539         stub->instruction_start()[0] = kTwoByteJumpInstruction;
00540         break;
00541       case INCREMENTAL_COMPACTION:

00552   // This is a helper class for freeing up 3 scratch registers, where the third
00553   // is always rcx (needed for shift operations).  The input is two registers
00554   // that must be preserved and one scratch register provided by the caller.
00555   class RegisterAllocation {
00556    public:
00557     RegisterAllocation(Register object,
00558                        Register address,
00559                        Register scratch0)
00560         : object_orig_(object),
00561           address_orig_(address),
00562           scratch0_orig_(scratch0),
00563           object_(object),
00564           address_(address),
00565           scratch0_(scratch0) {
00566       ASSERT(!AreAliased(scratch0, object, address, no_reg));
00567       scratch1_ = GetRegThatIsNotRcxOr(object_, address_, scratch0_);
00568       if (scratch0.is(rcx)) {
00569         scratch0_ = GetRegThatIsNotRcxOr(object_, address_, scratch1_);
00570       }
00571       if (object.is(rcx)) {
00572         object_ = GetRegThatIsNotRcxOr(address_, scratch0_, scratch1_);
00573       }
00574       if (address.is(rcx)) {
00575         address_ = GetRegThatIsNotRcxOr(object_, scratch0_, scratch1_);
00576       }
00577       ASSERT(!AreAliased(scratch0_, object_, address_, rcx));
00578     }
00579 
00580     void Save(MacroAssembler* masm) {
00581       ASSERT(!address_orig_.is(object_));
00582       ASSERT(object_.is(object_orig_) || address_.is(address_orig_));
00583       ASSERT(!AreAliased(object_, address_, scratch1_, scratch0_));
00584       ASSERT(!AreAliased(object_orig_, address_, scratch1_, scratch0_));
00585       ASSERT(!AreAliased(object_, address_orig_, scratch1_, scratch0_));
00586       // We don't have to save scratch0_orig_ because it was given to us as
00587       // a scratch register.  But if we had to switch to a different reg then
00588       // we should save the new scratch0_.
00589       if (!scratch0_.is(scratch0_orig_)) masm->push(scratch0_);
00590       if (!rcx.is(scratch0_orig_) &&
00591           !rcx.is(object_orig_) &&
00592           !rcx.is(address_orig_)) {
00593         masm->push(rcx);
00594       }
00595       masm->push(scratch1_);
00596       if (!address_.is(address_orig_)) {
00597         masm->push(address_);
00598         masm->movq(address_, address_orig_);
00599       }
00600       if (!object_.is(object_orig_)) {
00601         masm->push(object_);
00602         masm->movq(object_, object_orig_);
00603       }
00604     }

00606     void Restore(MacroAssembler* masm) {
00607       // These will have been preserved the entire time, so we just need to move
00608       // them back.  Only in one case is the orig_ reg different from the plain
00609       // one, since only one of them can alias with rcx.
00610       if (!object_.is(object_orig_)) {
00611         masm->movq(object_orig_, object_);
00612         masm->pop(object_);
00613       }
00614       if (!address_.is(address_orig_)) {
00615         masm->movq(address_orig_, address_);
00616         masm->pop(address_);
00617       }
00618       masm->pop(scratch1_);
