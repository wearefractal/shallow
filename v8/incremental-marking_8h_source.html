00092   // It's hard to know how much work the incremental marker should do to make
00093   // progress in the face of the mutator creating new work for it.  We start
00094   // of at a moderate rate of work and gradually increase the speed of the
00095   // incremental marker until it completes.
00096   // Do some marking every time this much memory has been allocated.
00097   static const intptr_t kAllocatedThreshold = 65536;
00098   // Start off by marking this many times more memory than has been allocated.
00099   static const intptr_t kInitialAllocationMarkingFactor = 1;
00100   // But if we are promoting a lot of data we need to mark faster to keep up
00101   // with the data that is entering the old space through promotion.
00102   static const intptr_t kFastMarking = 3;
00103   // After this many steps we increase the marking/allocating factor.
00104   static const intptr_t kAllocationMarkingFactorSpeedupInterval = 1024;
00105   // This is how much we increase the marking/allocating factor by.
00106   static const intptr_t kAllocationMarkingFactorSpeedup = 2;
00107   static const intptr_t kMaxAllocationMarkingFactor = 1000;
00108 
00109   void OldSpaceStep(intptr_t allocated) {
00110     Step(allocated * kFastMarking / kInitialAllocationMarkingFactor,
