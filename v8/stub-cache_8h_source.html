00311   // The stub cache has a primary and secondary level.  The two levels have
00312   // different hashing algorithms in order to avoid simultaneous collisions
00313   // in both caches.  Unlike a probing strategy (quadratic or otherwise) the
00314   // update strategy on updates is fairly clear and simple:  Any existing entry
00315   // in the primary cache is moved to the secondary cache, and secondary cache
00316   // entries are overwritten.
00317 
00318   // Hash algorithm for the primary table.  This algorithm is replicated in
00319   // assembler for every architecture.  Returns an index into the table that
00320   // is scaled by 1 << kHeapObjectTagSize.
00321   static int PrimaryOffset(String* name, Code::Flags flags, Map* map) {
00322     // This works well because the heap object tag size and the hash
00323     // shift are equal.  Shifting down the length field to get the
00324     // hash code would effectively throw away two bits of the hash
00325     // code.
00326     STATIC_ASSERT(kHeapObjectTagSize == String::kHashShift);
00327     // Compute the hash of the name (use entire hash field).
00328     ASSERT(name->HasHashCode());
00329     uint32_t field = name->hash_field();
00330     // Using only the low bits in 64-bit mode is unlikely to increase the
00331     // risk of collision even if the heap is spread over an area larger than
00332     // 4Gb (and not at all if it isn't).
00333     uint32_t map_low32bits =
00334         static_cast<uint32_t>(reinterpret_cast<uintptr_t>(map));
00335     // We always set the in_loop bit to zero when generating the lookup code
00336     // so do it here too so the hash codes match.
00337     uint32_t iflags =
00338         (static_cast<uint32_t>(flags) & ~Code::kFlagsNotUsedInLookup);
00339     // Base the offset on a simple combination of name, flags, and map.
00340     uint32_t key = (map_low32bits + field) ^ iflags;
00341     return key & ((kPrimaryTableSize - 1) << kHeapObjectTagSize);
00342   }
00343 
00344   // Hash algorithm for the secondary table.  This algorithm is replicated in
00345   // assembler for every architecture.  Returns an index into the table that
00346   // is scaled by 1 << kHeapObjectTagSize.
00347   static int SecondaryOffset(String* name, Code::Flags flags, int seed) {
00348     // Use the seed from the primary cache in the secondary cache.
00349     uint32_t string_low32bits =
00350         static_cast<uint32_t>(reinterpret_cast<uintptr_t>(name));
00351     // We always set the in_loop bit to zero when generating the lookup code
00352     // so do it here too so the hash codes match.
00353     uint32_t iflags =
00354         (static_cast<uint32_t>(flags) & ~Code::kFlagsNotUsedInLookup);
00355     uint32_t key = (seed - string_low32bits) + iflags;
00356     return key & ((kSecondaryTableSize - 1) << kHeapObjectTagSize);
00357   }
00358 
00359   // Compute the entry for a given offset in exactly the same way as
00360   // we do in generated code.  We generate an hash code that already
00361   // ends in String::kHashShift 0s.  Then we multiply it so it is a multiple
00362   // of sizeof(Entry).  This makes it easier to avoid making mistakes
00363   // in the hashed offset computations.
00364   static Entry* entry(Entry* table, int offset) {
00365     const int multiplier = sizeof(*table) >> String::kHashShift;
00366     return reinterpret_cast<Entry*>(
00367         reinterpret_cast<Address>(table) + offset * multiplier);
00368   }
