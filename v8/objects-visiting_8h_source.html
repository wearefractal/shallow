00033 // This file provides base classes and auxiliary methods for defining
00034 // static object visitors used during GC.
00035 // Visiting HeapObject body with a normal ObjectVisitor requires performing
00036 // two switches on object's instance type to determine object size and layout
00037 // and one or more virtual method calls on visitor itself.
00038 // Static visitor is different: it provides a dispatch table which contains
00039 // pointers to specialized visit functions. Each map has the visitor_id
00040 // field which contains an index of specialized visitor to use.

00128   // For visitors that allow specialization by size calculate VisitorId based
00129   // on size, base visitor id and generic visitor id.
00130   static VisitorId GetVisitorIdForSize(VisitorId base,
00131                                        VisitorId generic,
00132                                        int object_size) {
00133     ASSERT((base == kVisitDataObject) ||
00134            (base == kVisitStruct) ||
00135            (base == kVisitJSObject));
00136     ASSERT(IsAligned(object_size, kPointerSize));
00137     ASSERT(kMinObjectSizeInWords * kPointerSize <= object_size);
00138     ASSERT(object_size < Page::kMaxNonCodeHeapObjectSize);
00139 
00140     const VisitorId specialization = static_cast<VisitorId>(
00141         base + (object_size >> kPointerSizeLog2) - kMinObjectSizeInWords);
00142 
00143     return Min(specialization, generic);
00144   }

00148 template<typename Callback>
00149 class VisitorDispatchTable {
00150  public:
00151   void CopyFrom(VisitorDispatchTable* other) {
00152     // We are not using memcpy to guarantee that during update
00153     // every element of callbacks_ array will remain correct
00154     // pointer (memcpy might be implemented as a byte copying loop).
00155     for (int i = 0; i < StaticVisitorBase::kVisitorIdCount; i++) {
00156       NoBarrier_Store(&callbacks_[i], other->callbacks_[i]);
00157     }
00158   }
00159 
00160   inline Callback GetVisitorById(StaticVisitorBase::VisitorId id) {
00161     return reinterpret_cast<Callback>(callbacks_[id]);
00162   }
00163 
00164   inline Callback GetVisitor(Map* map) {
00165     return reinterpret_cast<Callback>(callbacks_[map->visitor_id()]);
00166   }
00167 
00168   void Register(StaticVisitorBase::VisitorId id, Callback callback) {
00169     ASSERT(id < StaticVisitorBase::kVisitorIdCount);  // id is unsigned.
00170     callbacks_[id] = reinterpret_cast<AtomicWord>(callback);
00171   }

00262 // Base class for visitors used for a linear new space iteration.
00263 // IterateBody returns size of visited object.
00264 // Certain types of objects (i.e. Code objects) are not handled
00265 // by dispatch table of this visitor because they cannot appear
00266 // in the new space.
00267 //
00268 // This class is intended to be used in the following way:
00269 //
00270 //   class SomeVisitor : public StaticNewSpaceVisitor<SomeVisitor> {
00271 //     ...
00272 //   }
00273 //
00274 // This is an example of Curiously recurring template pattern
00275 // (see http://en.wikipedia.org/wiki/Curiously_recurring_template_pattern).
00276 // We use CRTP to guarantee aggressive compile time optimizations (i.e.
00277 // inlining and specialization of StaticVisitor::VisitPointers methods).
00278 template<typename StaticVisitor>
00279 class StaticNewSpaceVisitor : public StaticVisitorBase {
00280  public:
00281   static void Initialize();
00282 
00283   static inline int IterateBody(Map* map, HeapObject* obj) {
00284     return table_.GetVisitor(map)(map, obj);
00285   }
00286 
00287   static inline void VisitPointers(Heap* heap, Object** start, Object** end) {
00288     for (Object** p = start; p < end; p++) StaticVisitor::VisitPointer(heap, p);
00289   }
00290 
00291  private:
00292   static inline int VisitJSFunction(Map* map, HeapObject* object) {
00293     Heap* heap = map->GetHeap();
00294     VisitPointers(heap,
00295                   HeapObject::RawField(object, JSFunction::kPropertiesOffset),
00296                   HeapObject::RawField(object, JSFunction::kCodeEntryOffset));
00297 
00298     // Don't visit code entry. We are using this visitor only during scavenges.
00299 
00300     VisitPointers(
00301         heap,
00302         HeapObject::RawField(object,
00303                              JSFunction::kCodeEntryOffset + kPointerSize),
00304         HeapObject::RawField(object,
00305                              JSFunction::kNonWeakFieldsEndOffset));
00306     return JSFunction::kSize;
00307   }

00309   static inline int VisitByteArray(Map* map, HeapObject* object) {
00310     return reinterpret_cast<ByteArray*>(object)->ByteArraySize();
00311   }

00313   static inline int VisitFixedDoubleArray(Map* map, HeapObject* object) {

00316   }

00318   static inline int VisitJSObject(Map* map, HeapObject* object) {
