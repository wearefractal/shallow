00095 void RelocInfo::apply(intptr_t delta) {
00096   if (IsCodeTarget(rmode_)) {
00097     uint32_t scope1 = (uint32_t) target_address() & ~kImm28Mask;
00098     uint32_t scope2 = reinterpret_cast<uint32_t>(pc_) & ~kImm28Mask;
00099 
00100     if (scope1 != scope2) {
00101       Assembler::JumpLabelToJumpRegister(pc_);
00102     }
00103   }
00104   if (IsInternalReference(rmode_)) {
00105     // Absolute code pointer inside code object moves with the code object.
00106     byte* p = reinterpret_cast<byte*>(pc_);
00107     int count = Assembler::RelocateInternalReference(p, delta);
00108     CPU::FlushICache(p, count * sizeof(uint32_t));
00109   }
00110 }

00119 Address RelocInfo::target_address_address() {
00120   ASSERT(IsCodeTarget(rmode_) ||
00121          rmode_ == RUNTIME_ENTRY ||
00122          rmode_ == EMBEDDED_OBJECT ||
00123          rmode_ == EXTERNAL_REFERENCE);
00124   // Read the address of the word containing the target_address in an
00125   // instruction stream.
00126   // The only architecture-independent user of this function is the serializer.
00127   // The serializer uses it to find out how many raw bytes of instruction to
00128   // output before the next target.
00129   // For an instruction like LUI/ORI where the target bits are mixed into the
00130   // instruction bits, the size of the target will be zero, indicating that the
00131   // serializer should not step forward in memory after a target is resolved
00132   // and written. In this case the target_address_address function should
00133   // return the end of the instructions to be patched, allowing the
00134   // deserializer to deserialize the instructions as raw bytes and put them in
00135   // place, ready to be patched with the target. After jump optimization,
00136   // that is the address of the instruction that follows J/JAL/JR/JALR
00137   // instruction.
00138   return reinterpret_cast<Address>(
00139     pc_ + Assembler::kInstructionsFor32BitConstant * Assembler::kInstrSize);
00140 }

00274 bool RelocInfo::IsPatchedReturnSequence() {
00275   Instr instr0 = Assembler::instr_at(pc_);
00276   Instr instr1 = Assembler::instr_at(pc_ + 1 * Assembler::kInstrSize);
00277   Instr instr2 = Assembler::instr_at(pc_ + 2 * Assembler::kInstrSize);
00278   bool patched_return = ((instr0 & kOpcodeMask) == LUI &&
00279                          (instr1 & kOpcodeMask) == ORI &&
00280                          ((instr2 & kOpcodeMask) == JAL ||
00281                           ((instr2 & kOpcodeMask) == SPECIAL &&
00282                            (instr2 & kFunctionFieldMask) == JALR)));
00283   return patched_return;
00284 }

00293 void RelocInfo::Visit(ObjectVisitor* visitor) {
00294   RelocInfo::Mode mode = rmode();
00295   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00296     visitor->VisitEmbeddedPointer(this);
00297   } else if (RelocInfo::IsCodeTarget(mode)) {
00298     visitor->VisitCodeTarget(this);
00299   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00300     visitor->VisitGlobalPropertyCell(this);
00301   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00302     visitor->VisitExternalReference(this);
00303 #ifdef ENABLE_DEBUGGER_SUPPORT
00304   // TODO(isolates): Get a cached isolate below.
00305   } else if (((RelocInfo::IsJSReturn(mode) &&
00306               IsPatchedReturnSequence()) ||
00307              (RelocInfo::IsDebugBreakSlot(mode) &&
00308              IsPatchedDebugBreakSlotSequence())) &&
00309              Isolate::Current()->debug()->has_break_points()) {
00310     visitor->VisitDebugTarget(this);
00311 #endif
00312   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00313     visitor->VisitRuntimeEntry(this);
00314   }
00315 }

00318 template<typename StaticVisitor>
00319 void RelocInfo::Visit(Heap* heap) {
00320   RelocInfo::Mode mode = rmode();
00321   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00322     StaticVisitor::VisitEmbeddedPointer(heap, this);
00323   } else if (RelocInfo::IsCodeTarget(mode)) {
00324     StaticVisitor::VisitCodeTarget(heap, this);
00325   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00326     StaticVisitor::VisitGlobalPropertyCell(heap, this);
00327   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00328     StaticVisitor::VisitExternalReference(this);
00329 #ifdef ENABLE_DEBUGGER_SUPPORT
00330   } else if (heap->isolate()->debug()->has_break_points() &&
00331              ((RelocInfo::IsJSReturn(mode) &&
00332               IsPatchedReturnSequence()) ||
00333              (RelocInfo::IsDebugBreakSlot(mode) &&
00334               IsPatchedDebugBreakSlotSequence()))) {
00335     StaticVisitor::VisitDebugTarget(heap, this);
00336 #endif
00337   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00338     StaticVisitor::VisitRuntimeEntry(this);
00339   }
00340 }
