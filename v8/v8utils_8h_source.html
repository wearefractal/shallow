00122 // Copies data from |src| to |dst|.  The data spans MUST not overlap.
00123 template <typename T>
00124 inline void CopyWords(T* dst, T* src, int num_words) {
00125   STATIC_ASSERT(sizeof(T) == kPointerSize);
00126   ASSERT(Min(dst, src) + num_words <= Max(dst, src));
00127   ASSERT(num_words > 0);
00128 
00129   // Use block copying memcpy if the segment we're copying is
00130   // enough to justify the extra call/setup overhead.
00131   static const int kBlockCopyLimit = 16;
00132 
00133   if (num_words >= kBlockCopyLimit) {
00134     memcpy(dst, src, num_words * kPointerSize);
00135   } else {
00136     int remaining = num_words;
00137     do {
00138       remaining--;
00139       *dst++ = *src++;
00140     } while (remaining > 0);
00141   }
00142 }

00203 // Copy from ASCII/16bit chars to ASCII/16bit chars.
00204 template <typename sourcechar, typename sinkchar>
00205 inline void CopyChars(sinkchar* dest, const sourcechar* src, int chars) {
00206   sinkchar* limit = dest + chars;
00207 #ifdef V8_HOST_CAN_READ_UNALIGNED
00208   if (sizeof(*dest) == sizeof(*src)) {
00209     if (chars >= static_cast<int>(OS::kMinComplexMemCopy / sizeof(*dest))) {
00210       OS::MemCopy(dest, src, chars * sizeof(*dest));
00211       return;
00212     }
00213     // Number of characters in a uintptr_t.
00214     static const int kStepSize = sizeof(uintptr_t) / sizeof(*dest);  // NOLINT
00215     while (dest <= limit - kStepSize) {
00216       *reinterpret_cast<uintptr_t*>(dest) =
00217           *reinterpret_cast<const uintptr_t*>(src);
00218       dest += kStepSize;
00219       src += kStepSize;
00220     }
00221   }
00222 #endif
00223   while (dest < limit) {
00224     *dest++ = static_cast<sinkchar>(*src++);
00225   }
00226 }
