00286 template <bool seq_ascii>
00287 Handle<Object> JsonParser<seq_ascii>::ParseJsonObject() {
00288   Handle<JSFunction> object_constructor(
00289       isolate()->global_context()->object_function());
00290   Handle<JSObject> json_object =
00291       isolate()->factory()->NewJSObject(object_constructor);
00292   ASSERT_EQ(c0_, '{');
00293 
00294   AdvanceSkipWhitespace();
00295   if (c0_ != '}') {
00296     do {
00297       if (c0_ != '"') return ReportUnexpectedCharacter();
00298       Handle<String> key = ParseJsonSymbol();
00299       if (key.is_null() || c0_ != ':') return ReportUnexpectedCharacter();
00300       AdvanceSkipWhitespace();
00301       Handle<Object> value = ParseJsonValue();
00302       if (value.is_null()) return ReportUnexpectedCharacter();
00303 
00304       uint32_t index;
00305       if (key->AsArrayIndex(&index)) {
00306         JSObject::SetOwnElement(json_object, index, value, kNonStrictMode);
00307       } else if (key->Equals(isolate()->heap()->Proto_symbol())) {
00308         SetPrototype(json_object, value);
00309       } else {
00310         JSObject::SetLocalPropertyIgnoreAttributes(
00311             json_object, key, value, NONE);
00312       }
00313     } while (MatchSkipWhiteSpace(','));
00314     if (c0_ != '}') {
00315       return ReportUnexpectedCharacter();
00316     }
00317   }
00318   AdvanceSkipWhitespace();

00351 template <bool seq_ascii>
00352 Handle<Object> JsonParser<seq_ascii>::ParseJsonNumber() {
00353   bool negative = false;
00354   int beg_pos = position_;
00355   if (c0_ == '-') {
00356     Advance();
00357     negative = true;
00358   }
00359   if (c0_ == '0') {
00360     Advance();
00361     // Prefix zero is only allowed if it's the only digit before
00362     // a decimal point or exponent.
00363     if ('0' <= c0_ && c0_ <= '9') return ReportUnexpectedCharacter();
00364   } else {
00365     int i = 0;
00366     int digits = 0;
00367     if (c0_ < '1' || c0_ > '9') return ReportUnexpectedCharacter();
00368     do {
00369       i = i * 10 + c0_ - '0';
00370       digits++;
00371       Advance();
00372     } while (c0_ >= '0' && c0_ <= '9');
00373     if (c0_ != '.' && c0_ != 'e' && c0_ != 'E' && digits < 10) {
00374       SkipWhitespace();
00375       return Handle<Smi>(Smi::FromInt((negative ? -i : i)), isolate());
00376     }
00377   }
00378   if (c0_ == '.') {
00379     Advance();
00380     if (c0_ < '0' || c0_ > '9') return ReportUnexpectedCharacter();
00381     do {
00382       Advance();
00383     } while (c0_ >= '0' && c0_ <= '9');
00384   }
00385   if (AsciiAlphaToLower(c0_) == 'e') {
00386     Advance();
00387     if (c0_ == '-' || c0_ == '+') Advance();
00388     if (c0_ < '0' || c0_ > '9') return ReportUnexpectedCharacter();
00389     do {
00390       Advance();
00391     } while (c0_ >= '0' && c0_ <= '9');
00392   }
00393   int length = position_ - beg_pos;
00394   double number;
00395   if (seq_ascii) {
00396     Vector<const char> chars(seq_source_->GetChars() +  beg_pos, length);
00397     number = StringToDouble(isolate()->unicode_cache(),
00398                              chars,
00399                              NO_FLAGS,  // Hex, octal or trailing junk.
00400                              OS::nan_value());
00401   } else {
00402     Vector<char> buffer = Vector<char>::New(length);
00403     String::WriteToFlat(*source_, buffer.start(), beg_pos, position_);
00404     Vector<const char> result =
00405         Vector<const char>(reinterpret_cast<const char*>(buffer.start()),
00406         length);
00407     number = StringToDouble(isolate()->unicode_cache(),
00408                              result,
00409                              NO_FLAGS,  // Hex, octal or trailing junk.
00410                              0.0);
00411     buffer.Dispose();
00412   }
00413   SkipWhitespace();
00414   return isolate()->factory()->NewNumber(number);
00415 }

00445 // Scans the rest of a JSON string starting from position_ and writes
00446 // prefix[start..end] along with the scanned characters into a
00447 // sequential string of type StringType.
00448 template <bool seq_ascii>
00449 template <typename StringType, typename SinkChar>
00450 Handle<String> JsonParser<seq_ascii>::SlowScanJsonString(
00451     Handle<String> prefix, int start, int end) {
00452   int count = end - start;
00453   int max_length = count + source_length_ - position_;
00454   int length = Min(max_length, Max(kInitialSpecialStringLength, 2 * count));
00455   Handle<StringType> seq_str = NewRawString<StringType>(isolate()->factory(),
00456                                                         length);
00457   // Copy prefix into seq_str.
00458   SinkChar* dest = seq_str->GetChars();
00459   String::WriteToFlat(*prefix, dest, start, end);
00460 
00461   while (c0_ != '"') {
00462     // Check for control character (0x00-0x1f) or unterminated string (<0).
00463     if (c0_ < 0x20) return Handle<String>::null();
00464     if (count >= length) {
00465       // We need to create a longer sequential string for the result.
00466       return SlowScanJsonString<StringType, SinkChar>(seq_str, 0, count);
00467     }
00468     if (c0_ != '\\') {
00469       // If the sink can contain UC16 characters, or source_ contains only
00470       // ASCII characters, there's no need to test whether we can store the
00471       // character. Otherwise check whether the UC16 source character can fit
00472       // in the ASCII sink.
00473       if (sizeof(SinkChar) == kUC16Size ||
00474           seq_ascii ||
00475           c0_ <= kMaxAsciiCharCode) {
00476         SeqStringSet(seq_str, count++, c0_);
00477         Advance();
00478       } else {
00479         // StringType is SeqAsciiString and we just read a non-ASCII char.
00480         return SlowScanJsonString<SeqTwoByteString, uc16>(seq_str, 0, count);
00481       }
00482     } else {
00483       Advance();  // Advance past the \.
00484       switch (c0_) {
00485         case '"':
00486         case '\\':
00487         case '/':
00488           SeqStringSet(seq_str, count++, c0_);
00489           break;
00490         case 'b':
00491           SeqStringSet(seq_str, count++, '\x08');
00492           break;
00493         case 'f':
00494           SeqStringSet(seq_str, count++, '\x0c');
00495           break;
00496         case 'n':
00497           SeqStringSet(seq_str, count++, '\x0a');
00498           break;
00499         case 'r':
00500           SeqStringSet(seq_str, count++, '\x0d');
00501           break;
00502         case 't':
00503           SeqStringSet(seq_str, count++, '\x09');
00504           break;
00505         case 'u': {
00506           uc32 value = 0;
00507           for (int i = 0; i < 4; i++) {
00508             Advance();
00509             int digit = HexValue(c0_);
00510             if (digit < 0) {
00511               return Handle<String>::null();
00512             }
00513             value = value * 16 + digit;
00514           }
00515           if (sizeof(SinkChar) == kUC16Size || value <= kMaxAsciiCharCode) {
00516             SeqStringSet(seq_str, count++, value);
00517             break;
00518           } else {
00519             // StringType is SeqAsciiString and we just read a non-ASCII char.
00520             position_ -= 6;  // Rewind position_ to \ in \uxxxx.
00521             Advance();

00523                                                               0,
00524                                                               count);
00525           }
00526         }
00527         default:
00528           return Handle<String>::null();
00529       }

00531     }
00532   }
00533   // Shrink seq_string length to count.
00534   if (isolate()->heap()->InNewSpace(*seq_str)) {
00535     isolate()->heap()->new_space()->
00536         template ShrinkStringAtAllocationBoundary<StringType>(
00537             *seq_str, count);

00539     int string_size = StringType::SizeFor(count);
00540     int allocated_string_size = StringType::SizeFor(length);
00541     int delta = allocated_string_size - string_size;
00542     Address start_filler_object = seq_str->address() + string_size;
00543     seq_str->set_length(count);
00544     isolate()->heap()->CreateFillerObjectAt(start_filler_object, delta);
00545   }
00546   ASSERT_EQ('"', c0_);
00547   // Advance past the last '"'.
00548   AdvanceSkipWhitespace();

00568       if (seq_ascii || c0_ <= kMaxAsciiCharCode) {
00569         Advance();

00577                                                       beg_pos,
