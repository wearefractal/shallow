00118 // Values for the condition field as defined in section A3.2
00119 enum Condition {
00120   kNoCondition = -1,
00121 
00122   eq =  0 << 28,                 // Z set            Equal.
00123   ne =  1 << 28,                 // Z clear          Not equal.
00124   cs =  2 << 28,                 // C set            Unsigned higher or same.
00125   cc =  3 << 28,                 // C clear          Unsigned lower.
00126   mi =  4 << 28,                 // N set            Negative.
00127   pl =  5 << 28,                 // N clear          Positive or zero.
00128   vs =  6 << 28,                 // V set            Overflow.
00129   vc =  7 << 28,                 // V clear          No overflow.
00130   hi =  8 << 28,                 // C set, Z clear   Unsigned higher.
00131   ls =  9 << 28,                 // C clear or Z set Unsigned lower or same.
00132   ge = 10 << 28,                 // N == V           Greater or equal.
00133   lt = 11 << 28,                 // N != V           Less than.
00134   gt = 12 << 28,                 // Z clear, N == V  Greater than.
00135   le = 13 << 28,                 // Z set or N != V  Less then or equal
00136   al = 14 << 28,                 //                  Always.
00137 
00138   kSpecialCondition = 15 << 28,  // Special condition (refer to section A3.2.1).
00139   kNumberOfConditions = 16,
00140 
00141   // Aliases.
00142   hs = cs,                       // C set            Unsigned higher or same.
00143   lo = cc                        // C clear          Unsigned lower.
00144 };

00153 // Corresponds to transposing the operands of a comparison.
00154 inline Condition ReverseCondition(Condition cond) {
00155   switch (cond) {
00156     case lo:
00157       return hi;
00158     case hi:
00159       return lo;
00160     case hs:
00161       return ls;
00162     case ls:
00163       return hs;
00164     case lt:
00165       return gt;
00166     case gt:
00167       return lt;
00168     case ge:
00169       return le;
00170     case le:
00171       return ge;
00172     default:

00174   };

00494 // -----------------------------------------------------------------------------
00495 // Instruction abstraction.
00496 
00497 // The class Instruction enables access to individual fields defined in the ARM
00498 // architecture instruction set encoding as described in figure A3-1.
00499 // Note that the Assembler uses typedef int32_t Instr.
00500 //
00501 // Example: Test whether the instruction at ptr does set the condition code
00502 // bits.
00503 //
00504 // bool InstructionSetsConditionCodes(byte* ptr) {
00505 //   Instruction* instr = Instruction::At(ptr);
00506 //   int type = instr->TypeValue();
00507 //   return ((type == 0) || (type == 1)) && instr->HasS();
00508 // }
00509 //
00510 class Instruction {
00511  public:
00512   enum {
00513     kInstrSize = 4,
00514     kInstrSizeLog2 = 2,
00515     kPCReadOffset = 8
00516   };
00517 
00518   // Helper macro to define static accessors.
00519   // We use the cast to char* trick to bypass the strict anti-aliasing rules.
00520   #define DECLARE_STATIC_TYPED_ACCESSOR(return_type, Name)                     \
00521     static inline return_type Name(Instr instr) {                              \
00522       char* temp = reinterpret_cast<char*>(&instr);                            \
00523       return reinterpret_cast<Instruction*>(temp)->Name();                     \
00524     }
00525 
00526   #define DECLARE_STATIC_ACCESSOR(Name) DECLARE_STATIC_TYPED_ACCESSOR(int, Name)
00527 
00528   // Get the raw instruction bits.
00529   inline Instr InstructionBits() const {
00530     return *reinterpret_cast<const Instr*>(this);
00531   }
00532 
00533   // Set the raw instruction bits to value.
00534   inline void SetInstructionBits(Instr value) {
00535     *reinterpret_cast<Instr*>(this) = value;
00536   }
00537 
00538   // Read one particular bit out of the instruction bits.
00539   inline int Bit(int nr) const {
00540     return (InstructionBits() >> nr) & 1;
00541   }
00542 
00543   // Read a bit field's value out of the instruction bits.
00544   inline int Bits(int hi, int lo) const {
00545     return (InstructionBits() >> lo) & ((2 << (hi - lo)) - 1);
00546   }
00547 
00548   // Read a bit field out of the instruction bits.
00549   inline int BitField(int hi, int lo) const {
00550     return InstructionBits() & (((2 << (hi - lo)) - 1) << lo);
00551   }
00552 
00553   // Static support.
00554 
00555   // Read one particular bit out of the instruction bits.
00556   static inline int Bit(Instr instr, int nr) {
00557     return (instr >> nr) & 1;
00558   }
00559 
00560   // Read the value of a bit field out of the instruction bits.
00561   static inline int Bits(Instr instr, int hi, int lo) {
00562     return (instr >> lo) & ((2 << (hi - lo)) - 1);
00563   }
00564 
00565 
00566   // Read a bit field out of the instruction bits.
00567   static inline int BitField(Instr instr, int hi, int lo) {
00568     return instr & (((2 << (hi - lo)) - 1) << lo);
00569   }

00582   //   0xC0810002 ConditionField(instr) will return 0xC.

00725   inline int VFPGlueRegValue(VFPRegPrecision pre, int four_bit, int one_bit) {
00726     if (pre == kSinglePrecision) {
00727       return (Bits(four_bit + 3, four_bit) << 1) | Bit(one_bit);
00728     }
00729     return (Bit(one_bit) << 4) | Bits(four_bit + 3, four_bit);
