00159 void RuntimeProfiler::IsolateEnteredJS(Isolate* isolate) {
00160   Atomic32 new_state = NoBarrier_AtomicIncrement(&state_, 1);
00161   if (new_state == 0) {
00162     // Just incremented from -1 to 0. -1 can only be set by the
00163     // profiler thread before it suspends itself and starts waiting on
00164     // the semaphore.
00165     HandleWakeUp(isolate);
00166   }
00167   ASSERT(new_state >= 0);
00168 }

00173   ASSERT(new_state >= 0);
