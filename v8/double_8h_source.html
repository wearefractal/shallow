00036 // We assume that doubles and uint64_t have the same endianness.
00037 inline uint64_t double_to_uint64(double d) { return BitCast<uint64_t>(d); }

00040 // Helper functions for doubles.
00041 class Double {
00042  public:
00043   static const uint64_t kSignMask = V8_2PART_UINT64_C(0x80000000, 00000000);
00044   static const uint64_t kExponentMask = V8_2PART_UINT64_C(0x7FF00000, 00000000);
00045   static const uint64_t kSignificandMask =
00046       V8_2PART_UINT64_C(0x000FFFFF, FFFFFFFF);
00047   static const uint64_t kHiddenBit = V8_2PART_UINT64_C(0x00100000, 00000000);
00048   static const int kPhysicalSignificandSize = 52;  // Excludes the hidden bit.
00049   static const int kSignificandSize = 53;
00050 
00051   Double() : d64_(0) {}
00052   explicit Double(double d) : d64_(double_to_uint64(d)) {}
00053   explicit Double(uint64_t d64) : d64_(d64) {}
00054   explicit Double(DiyFp diy_fp)
00055     : d64_(DiyFpToUint64(diy_fp)) {}

00057   // The value encoded by this Double must be greater or equal to +0.0.
00058   // It must not be special (infinity, or NaN).
00059   DiyFp AsDiyFp() const {
00060     ASSERT(Sign() > 0);
00061     ASSERT(!IsSpecial());
00062     return DiyFp(Significand(), Exponent());
00063   }
00064 
00065   // The value encoded by this Double must be strictly greater than 0.
00066   DiyFp AsNormalizedDiyFp() const {
00067     ASSERT(value() > 0.0);
00068     uint64_t f = Significand();
00069     int e = Exponent();
00070 
00071     // The current double could be a denormal.
00072     while ((f & kHiddenBit) == 0) {
00073       f <<= 1;
00074       e--;
00075     }
00076     // Do the final shifts in one go.
00077     f <<= DiyFp::kSignificandSize - kSignificandSize;
00078     e -= DiyFp::kSignificandSize - kSignificandSize;
00079     return DiyFp(f, e);
00080   }

00082   // Returns the double's bit as uint64.
00083   uint64_t AsUint64() const {
00084     return d64_;
00085   }

00087   // Returns the next greater double. Returns +infinity on input +infinity.
00088   double NextDouble() const {
00089     if (d64_ == kInfinity) return Double(kInfinity).value();
00090     if (Sign() < 0 && Significand() == 0) {
00091       // -0.0
00092       return 0.0;
00093     }
00094     if (Sign() < 0) {
00095       return Double(d64_ - 1).value();
00096     } else {
00097       return Double(d64_ + 1).value();
00098     }
00099   }

00101   int Exponent() const {
00102     if (IsDenormal()) return kDenormalExponent;
00103 
00104     uint64_t d64 = AsUint64();
00105     int biased_e =
00106         static_cast<int>((d64 & kExponentMask) >> kPhysicalSignificandSize);
00107     return biased_e - kExponentBias;
00108   }
00109 
00110   uint64_t Significand() const {
00111     uint64_t d64 = AsUint64();
00112     uint64_t significand = d64 & kSignificandMask;
00113     if (!IsDenormal()) {
00114       return significand + kHiddenBit;
00115     } else {
00116       return significand;
00117     }
00118   }

00120   // Returns true if the double is a denormal.
00121   bool IsDenormal() const {
00122     uint64_t d64 = AsUint64();
00123     return (d64 & kExponentMask) == 0;
00124   }

00126   // We consider denormals not to be special.
00127   // Hence only Infinity and NaN are special.
00128   bool IsSpecial() const {
00129     uint64_t d64 = AsUint64();
00130     return (d64 & kExponentMask) == kExponentMask;
00131   }

00133   bool IsInfinite() const {
00134     uint64_t d64 = AsUint64();
00135     return ((d64 & kExponentMask) == kExponentMask) &&

00139   int Sign() const {
00140     uint64_t d64 = AsUint64();
00141     return (d64 & kSignMask) == 0? 1: -1;

00144   // Precondition: the value encoded by this Double must be greater or equal
00145   // than +0.0.
00146   DiyFp UpperBoundary() const {
00147     ASSERT(Sign() > 0);
00148     return DiyFp(Significand() * 2 + 1, Exponent() - 1);
00149   }

00151   // Returns the two boundaries of this.
00152   // The bigger boundary (m_plus) is normalized. The lower boundary has the same
00153   // exponent as m_plus.
00154   // Precondition: the value encoded by this Double must be greater than 0.
00155   void NormalizedBoundaries(DiyFp* out_m_minus, DiyFp* out_m_plus) const {
00156     ASSERT(value() > 0.0);
00157     DiyFp v = this->AsDiyFp();
00158     bool significand_is_zero = (v.f() == kHiddenBit);
00159     DiyFp m_plus = DiyFp::Normalize(DiyFp((v.f() << 1) + 1, v.e() - 1));
00160     DiyFp m_minus;
00161     if (significand_is_zero && v.e() != kDenormalExponent) {
00162       // The boundary is closer. Think of v = 1000e10 and v- = 9999e9.
00163       // Then the boundary (== (v - v-)/2) is not just at a distance of 1e9 but
00164       // at a distance of 1e8.
00165       // The only exception is for the smallest normal: the largest denormal is
00166       // at the same distance as its successor.
00167       // Note: denormals have the same exponent as the smallest normals.
00168       m_minus = DiyFp((v.f() << 2) - 1, v.e() - 2);
00169     } else {
00170       m_minus = DiyFp((v.f() << 1) - 1, v.e() - 1);
00171     }
00172     m_minus.set_f(m_minus.f() << (m_minus.e() - m_plus.e()));
00173     m_minus.set_e(m_plus.e());
00174     *out_m_plus = m_plus;
00175     *out_m_minus = m_minus;
00176   }

00178   double value() const { return uint64_to_double(d64_); }

00180   // Returns the significand size for a given order of magnitude.
00181   // If v = f*2^e with 2^p-1 <= f <= 2^p then p+e is v's order of magnitude.
00182   // This function returns the number of significant binary digits v will have
00183   // once its encoded into a double. In almost all cases this is equal to
00184   // kSignificandSize. The only exception are denormals. They start with leading
00185   // zeroes and their effective significand-size is hence smaller.
00186   static int SignificandSizeForOrderOfMagnitude(int order) {
00187     if (order >= (kDenormalExponent + kSignificandSize)) {
00188       return kSignificandSize;
00189     }
00190     if (order <= kDenormalExponent) return 0;
00191     return order - kDenormalExponent;
00192   }

00194  private:
00195   static const int kExponentBias = 0x3FF + kPhysicalSignificandSize;
00196   static const int kDenormalExponent = -kExponentBias + 1;
00197   static const int kMaxExponent = 0x7FF - kExponentBias;
00198   static const uint64_t kInfinity = V8_2PART_UINT64_C(0x7FF00000, 00000000);
00199 
00200   const uint64_t d64_;
00201 
00202   static uint64_t DiyFpToUint64(DiyFp diy_fp) {
00203     uint64_t significand = diy_fp.f();
00204     int exponent = diy_fp.e();
00205     while (significand > kHiddenBit + kSignificandMask) {
00206       significand >>= 1;
00207       exponent++;
00208     }
00209     if (exponent >= kMaxExponent) {
00210       return kInfinity;
00211     }
00212     if (exponent < kDenormalExponent) {
00213       return 0;
00214     }
00215     while (exponent > kDenormalExponent && (significand & kHiddenBit) == 0) {
00216       significand <<= 1;
00217       exponent--;
00218     }
00219     uint64_t biased_exponent;
00220     if (exponent == kDenormalExponent && (significand & kHiddenBit) == 0) {
00221       biased_exponent = 0;
00222     } else {
00223       biased_exponent = static_cast<uint64_t>(exponent + kExponentBias);
00224     }
00225     return (significand & kSignificandMask) |
00226         (biased_exponent << kPhysicalSignificandSize);
00227   }
