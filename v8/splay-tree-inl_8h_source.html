00044 template<typename Config, class Allocator>
00045 bool SplayTree<Config, Allocator>::Insert(const Key& key, Locator* locator) {
00046   if (is_empty()) {
00047     // If the tree is empty, insert the new node.
00048     root_ = new Node(key, Config::NoValue());
00049   } else {
00050     // Splay on the key to move the last node on the search path
00051     // for the key to the root of the tree.
00052     Splay(key);
00053     // Ignore repeated insertions with the same key.
00054     int cmp = Config::Compare(key, root_->key_);
00055     if (cmp == 0) {
00056       locator->bind(root_);
00057       return false;
00058     }
00059     // Insert the new node.
00060     Node* node = new Node(key, Config::NoValue());
00061     InsertInternal(cmp, node);
00062   }
00063   locator->bind(root_);
00064   return true;
00065 }

00069 void SplayTree<Config, Allocator>::InsertInternal(int cmp, Node* node) {
00070   if (cmp > 0) {
00071     node->left_ = root_;
00072     node->right_ = root_->right_;
00073     root_->right_ = NULL;

00075     node->right_ = root_;
00076     node->left_ = root_->left_;
00077     root_->left_ = NULL;
00078   }
00079   root_ = node;

00104 bool SplayTree<Config, Allocator>::FindGreatestLessThan(const Key& key,
00105                                                         Locator* locator) {
00106   if (is_empty())
00107     return false;
00108   // Splay on the key to move the node with the given key or the last
00109   // node on the search path to the top of the tree.
00110   Splay(key);
00111   // Now the result is either the root node or the greatest node in
00112   // the left subtree.

00116     return true;

00119     root_ = root_->left_;

00175 template<typename Config, class Allocator>
00176 bool SplayTree<Config, Allocator>::Move(const Key& old_key,
00177                                         const Key& new_key) {
00178   if (!FindInternal(old_key))
00179     return false;
00180   Node* node_to_move = root_;
00181   RemoveRootNode(old_key);
00182   Splay(new_key);
00183   int cmp = Config::Compare(new_key, root_->key_);
00184   if (cmp == 0) {
00185     // A node with the target key already exists.
00186     delete node_to_move;
00187     return false;
00188   }
00189   node_to_move->key_ = new_key;
00190   InsertInternal(cmp, node_to_move);
00191   return true;
00192 }

00206 template<typename Config, class Allocator>
00207 void SplayTree<Config, Allocator>::RemoveRootNode(const Key& key) {
00208   if (root_->left_ == NULL) {
00209     // No left child, so the new tree is just the right child.
00210     root_ = root_->right_;
00211   } else {
00212     // Left child exists.
00213     Node* right = root_->right_;
00214     // Make the original left child the new root.
00215     root_ = root_->left_;
00216     // Splay to make sure that the new root has an empty right child.
00217     Splay(key);
00218     // Insert the original right child as the right child of the new
00219     // root.
00220     root_->right_ = right;
00221   }
00222 }

00225 template<typename Config, class Allocator>
00226 void SplayTree<Config, Allocator>::Splay(const Key& key) {
00227   if (is_empty())
00228     return;
00229   Node dummy_node(Config::kNoKey, Config::NoValue());
00230   // Create a dummy node.  The use of the dummy node is a bit
00231   // counter-intuitive: The right child of the dummy node will hold
00232   // the L tree of the algorithm.  The left child of the dummy node
00233   // will hold the R tree of the algorithm.  Using a dummy node, left
00234   // and right will always be nodes and we avoid special cases.
00235   Node* dummy = &dummy_node;
00236   Node* left = dummy;
00237   Node* right = dummy;
00238   Node* current = root_;
00239   while (true) {
00240     int cmp = Config::Compare(key, current->key_);
00241     if (cmp < 0) {
00242       if (current->left_ == NULL)
00243         break;
00244       if (Config::Compare(key, current->left_->key_) < 0) {
00245         // Rotate right.
00246         Node* temp = current->left_;
00247         current->left_ = temp->right_;
00248         temp->right_ = current;
00249         current = temp;
00250         if (current->left_ == NULL)
00251           break;
00252       }
00253       // Link right.
00254       right->left_ = current;
00255       right = current;
00256       current = current->left_;
00257     } else if (cmp > 0) {
00258       if (current->right_ == NULL)
00259         break;
00260       if (Config::Compare(key, current->right_->key_) > 0) {
00261         // Rotate left.
00262         Node* temp = current->right_;
00263         current->right_ = temp->left_;
00264         temp->left_ = current;
00265         current = temp;
00266         if (current->right_ == NULL)
00267           break;
00268       }
00269       // Link left.
00270       left->right_ = current;
00271       left = current;
00272       current = current->right_;
00273     } else {
00274       break;
00275     }
00276   }
00277   // Assemble.
00278   left->right_ = current->left_;
00279   right->left_ = current->right_;
00280   current->left_ = dummy->right_;
00281   current->right_ = dummy->left_;
00282   root_ = current;
00283 }

00293 template <typename Config, class Allocator> template <class Callback>
00294 void SplayTree<Config, Allocator>::ForEachNode(Callback* callback) {
00295   // Pre-allocate some space for tiny trees.
00296   List<Node*, Allocator> nodes_to_visit(10);
00297   if (root_ != NULL) nodes_to_visit.Add(root_);
00298   int pos = 0;
00299   while (pos < nodes_to_visit.length()) {
00300     Node* node = nodes_to_visit[pos++];
00301     if (node->left() != NULL) nodes_to_visit.Add(node->left());
00302     if (node->right() != NULL) nodes_to_visit.Add(node->right());
00303     callback->Call(node);
00304   }
00305 }
