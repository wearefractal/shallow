00043 void PromotionQueue::insert(HeapObject* target, int size) {
00044   if (emergency_stack_ != NULL) {
00045     emergency_stack_->Add(Entry(target, size));
00046     return;
00047   }
00048 
00049   if (NewSpacePage::IsAtStart(reinterpret_cast<Address>(rear_))) {
00050     NewSpacePage* rear_page =
00051         NewSpacePage::FromAddress(reinterpret_cast<Address>(rear_));
00052     ASSERT(!rear_page->prev_page()->is_anchor());
00053     rear_ = reinterpret_cast<intptr_t*>(rear_page->prev_page()->area_end());
00054     ActivateGuardIfOnTheSamePage();
00055   }
00056 
00057   if (guard_) {
00058     ASSERT(GetHeadPage() ==
00059            Page::FromAllocationTop(reinterpret_cast<Address>(limit_)));
00060 
00061     if ((rear_ - 2) < limit_) {
00062       RelocateQueueHead();
00063       emergency_stack_->Add(Entry(target, size));
00064       return;
00065     }
00066   }
00067 
00068   *(--rear_) = reinterpret_cast<intptr_t>(target);
00069   *(--rear_) = size;
00070   // Assert no overflow into live objects.
00071 #ifdef DEBUG
00072   SemiSpace::AssertValidRange(HEAP->new_space()->top(),
00073                               reinterpret_cast<Address>(rear_));
00074 #endif
00075 }

00184                                AllocationSpace space,

00208   }
00209 
00210   if (OLD_POINTER_SPACE == space) {
00211     result = old_pointer_space_->AllocateRaw(size_in_bytes);
00212   } else if (OLD_DATA_SPACE == space) {
00213     result = old_data_space_->AllocateRaw(size_in_bytes);
00214   } else if (CODE_SPACE == space) {
00215     result = code_space_->AllocateRaw(size_in_bytes);
00216   } else if (LO_SPACE == space) {
00217     result = lo_space_->AllocateRaw(size_in_bytes, NOT_EXECUTABLE);

00219     result = cell_space_->AllocateRaw(size_in_bytes);

00221     ASSERT(MAP_SPACE == space);
00222     result = map_space_->AllocateRaw(size_in_bytes);

00322 bool Heap::ShouldBePromoted(Address old_address, int object_size) {
00323   // An object should be promoted if:
00324   // - the object has survived a scavenge operation or
00325   // - to space is already 25% full.
00326   NewSpacePage* page = NewSpacePage::FromAddress(old_address);
00327   Address age_mark = new_space_.age_mark();
00328   bool below_mark = page->IsFlagSet(MemoryChunk::NEW_SPACE_BELOW_AGE_MARK) &&
00329       (!page->ContainsLimit(age_mark) || old_address < age_mark);
00330   return below_mark || (new_space_.Size() + object_size) >=
00331                         (new_space_.EffectiveCapacity() >> 2);
00332 }

00358 AllocationSpace Heap::TargetSpaceId(InstanceType type) {
00359   // Heap numbers and sequential strings are promoted to old data space, all
00360   // other object types are promoted to old pointer space.  We do not use
00361   // object->IsHeapNumber() and object->IsSeqString() because we already
00362   // know that object has the heap object tag.
00363 
00364   // These objects are never allocated in new space.
00365   ASSERT(type != MAP_TYPE);
00366   ASSERT(type != CODE_TYPE);
00367   ASSERT(type != ODDBALL_TYPE);
00368   ASSERT(type != JS_GLOBAL_PROPERTY_CELL_TYPE);
00369 
00370   if (type < FIRST_NONSTRING_TYPE) {
00371     // There are four string representations: sequential strings, external
00372     // strings, cons strings, and sliced strings.
00373     // Only the latter two contain non-map-word pointers to heap objects.
00374     return ((type & kIsIndirectStringMask) == kIsIndirectStringTag)
00375         ? OLD_POINTER_SPACE
00376         : OLD_DATA_SPACE;
00377   } else {
00378     return (type <= LAST_DATA_TYPE) ? OLD_DATA_SPACE : OLD_POINTER_SPACE;
00379   }
00380 }

00390 void Heap::MoveBlock(Address dst, Address src, int byte_size) {
00391   ASSERT(IsAligned(byte_size, kPointerSize));
00392 
00393   int size_in_words = byte_size / kPointerSize;
00394 
00395   if ((dst < src) || (dst >= (src + byte_size))) {
00396     Object** src_slot = reinterpret_cast<Object**>(src);
00397     Object** dst_slot = reinterpret_cast<Object**>(dst);
00398     Object** end_slot = src_slot + size_in_words;
00399 
00400     while (src_slot != end_slot) {
00401       *dst_slot++ = *src_slot++;
00402     }
00403   } else {
00404     memmove(dst, src, byte_size);
00405   }

00414 void Heap::ScavengeObject(HeapObject** p, HeapObject* object) {
00415   ASSERT(HEAP->InFromSpace(object));
00416 
00417   // We use the first word (where the map pointer usually is) of a heap
00418   // object to record the forwarding pointer.  A forwarding pointer can
00419   // point to an old space, the code space, or the to space of the new
00420   // generation.
00421   MapWord first_word = object->map_word();
00422 
00423   // If the first word is a forwarding address, the object has already been
00424   // copied.
00425   if (first_word.IsForwardingAddress()) {
00426     HeapObject* dest = first_word.ToForwardingAddress();
00427     ASSERT(HEAP->InFromSpace(*p));
00428     *p = dest;
00429     return;
00430   }
00431 
00432   // Call the slow part of scavenge object.
00433   return ScavengeObjectSlow(p, object);
00434 }

00507 // Calls the FUNCTION_CALL function and retries it up to three times
00508 // to guarantee that any allocations performed during the call will
00509 // succeed if there's enough memory.

00512 // __scope__ in a call to this macro.
00513 
00514 #define CALL_AND_RETRY(ISOLATE, FUNCTION_CALL, RETURN_VALUE, RETURN_EMPTY)\
00515   do {                                                                    \
00516     GC_GREEDY_CHECK();                                                    \
00517     MaybeObject* __maybe_object__ = FUNCTION_CALL;                        \
00518     Object* __object__ = NULL;                                            \
00519     if (__maybe_object__->ToObject(&__object__)) RETURN_VALUE;            \
00520     if (__maybe_object__->IsOutOfMemory()) {                              \
00521       v8::internal::V8::FatalProcessOutOfMemory("CALL_AND_RETRY_0", true);\
00522     }                                                                     \
00523     if (!__maybe_object__->IsRetryAfterGC()) RETURN_EMPTY;                \
00524     ISOLATE->heap()->CollectGarbage(Failure::cast(__maybe_object__)->     \
00525                                     allocation_space(),                   \
00526                                     "allocation failure");                \
00527     __maybe_object__ = FUNCTION_CALL;                                     \

00538     }                                                                     \
00539     if (__maybe_object__->ToObject(&__object__)) RETURN_VALUE;            \
00540     if (__maybe_object__->IsOutOfMemory() ||                              \

00678 MaybeObject* TranscendentalCache::SubCache::Get(double input) {
00679   Converter c;
00680   c.dbl = input;
00681   int hash = Hash(c);
00682   Element e = elements_[hash];
00683   if (e.in[0] == c.integers[0] &&
00684       e.in[1] == c.integers[1]) {
00685     ASSERT(e.output != NULL);
00686     isolate_->counters()->transcendental_cache_hit()->Increment();
00687     return e.output;
00688   }
00689   double answer = Calculate(input);
00690   isolate_->counters()->transcendental_cache_miss()->Increment();
00691   Object* heap_number;
00692   { MaybeObject* maybe_heap_number =
00693         isolate_->heap()->AllocateHeapNumber(answer);
00694     if (!maybe_heap_number->ToObject(&heap_number)) return maybe_heap_number;
00695   }
00696   elements_[hash].in[0] = c.integers[0];
00697   elements_[hash].in[1] = c.integers[1];
00698   elements_[hash].output = heap_number;
00699   return heap_number;
00700 }
