00181 // On MIPS Simulator breakpoints can have different codes:
00182 // - Breaks between 0 and kMaxWatchpointCode are treated as simple watchpoints,
00183 //   the simulator will run through them and print the registers.
00184 // - Breaks between kMaxWatchpointCode and kMaxStopCode are treated as stop()
00185 //   instructions (see Assembler::stop()).
00186 // - Breaks larger than kMaxStopCode are simple breaks, dropping you into the
00187 //   debugger.
00188 const uint32_t kMaxWatchpointCode = 31;
00189 const uint32_t kMaxStopCode = 127;
00190 STATIC_ASSERT(kMaxWatchpointCode < kMaxStopCode);

00476 // Returns the equivalent of !cc.
00477 // Negation of the default kNoCondition (-1) results in a non-default
00478 // no_condition value (-2). As long as tests for no_condition check
00479 // for condition < 0, this will work as expected.
00480 inline Condition NegateCondition(Condition cc) {
00481   ASSERT(cc != cc_always);
00482   return static_cast<Condition>(cc ^ 1);
00483 }

00486 inline Condition ReverseCondition(Condition cc) {
00487   switch (cc) {
00488     case Uless:
00489       return Ugreater;
00490     case Ugreater:
00491       return Uless;
00492     case Ugreater_equal:
00493       return Uless_equal;
00494     case Uless_equal:
00495       return Ugreater_equal;
00496     case less:
00497       return greater;
00498     case greater:
00499       return less;
00500     case greater_equal:
00501       return less_equal;
00502     case less_equal:
00503       return greater_equal;
00504     default:
00505       return cc;
00506   };
00507 }

00602   // Get the raw instruction bits.
