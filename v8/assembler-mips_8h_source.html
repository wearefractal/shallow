00047 // CPU Registers.
00048 //
00049 // 1) We would prefer to use an enum, but enum values are assignment-
00050 // compatible with int, which has caused code-generation bugs.
00051 //
00052 // 2) We would prefer to use a class instead of a struct but we don't like
00053 // the register initialization to depend on the particular initialization
00054 // order (which appears to be different on OS X, Linux, and Windows for the
00055 // installed versions of C++ we tried). Using a struct permits C-style
00056 // "initialization". Also, the Register objects cannot be const as this
00057 // forces initialization stubs in MSVC, making us dependent on initialization
00058 // order.
00059 //
00060 // 3) By not using an enum, we are possibly preventing the compiler from
00061 // doing certain constant folds, which may significantly reduce the
00062 // code generated for some assembly instructions (because they boil down
00063 // to a few constants). If this is a problem, we could change the code
00064 // such that we use an enum in optimized mode, and the struct in debug
00065 // mode. This way we get the compile-time error checking in debug mode
00066 // and best performance in optimized code.

01201   // One trampoline consists of:
01202   // - space for trampoline slots,
01203   // - space for labels.
01204   //
01205   // Space for trampoline slots is equal to slot_count * 2 * kInstrSize.
01206   // Space for trampoline slots preceeds space for labels. Each label is of one
01207   // instruction size, so total amount for labels is equal to

01234         // when this happens.

01239         free_slot_count_--;
01240         next_slot_ += kTrampolineSlotsSize;

01242       return trampoline_slot;
01243     }
