00034 // This "Do It Yourself Floating Point" class implements a floating-point number
00035 // with a uint64 significand and an int exponent. Normalized DiyFp numbers will
00036 // have the most significant bit of the significand set.
00037 // Multiplication and Subtraction do not normalize their results.
00038 // DiyFp are not designed to contain special doubles (NaN and Infinity).
00039 class DiyFp {
00040  public:
00041   static const int kSignificandSize = 64;
00042 
00043   DiyFp() : f_(0), e_(0) {}
00044   DiyFp(uint64_t f, int e) : f_(f), e_(e) {}
00045 
00046   // this = this - other.
00047   // The exponents of both numbers must be the same and the significand of this
00048   // must be bigger than the significand of other.
00049   // The result will not be normalized.
00050   void Subtract(const DiyFp& other) {
00051     ASSERT(e_ == other.e_);
00052     ASSERT(f_ >= other.f_);
00053     f_ -= other.f_;
00054   }
00055 
00056   // Returns a - b.
00057   // The exponents of both numbers must be the same and this must be bigger
00058   // than other. The result will not be normalized.
00059   static DiyFp Minus(const DiyFp& a, const DiyFp& b) {
00060     DiyFp result = a;
00061     result.Subtract(b);
00062     return result;
00063   }
00064 
00065 
00066   // this = this * other.
00067   void Multiply(const DiyFp& other);
00068 
00069   // returns a * b;
00070   static DiyFp Times(const DiyFp& a, const DiyFp& b) {
00071     DiyFp result = a;
00072     result.Multiply(b);
00073     return result;
00074   }
00075 
00076   void Normalize() {
00077     ASSERT(f_ != 0);
00078     uint64_t f = f_;
00079     int e = e_;
00080 
00081     // This method is mainly called for normalizing boundaries. In general
00082     // boundaries need to be shifted by 10 bits. We thus optimize for this case.
00083     const uint64_t k10MSBits = static_cast<uint64_t>(0x3FF) << 54;
00084     while ((f & k10MSBits) == 0) {
00085       f <<= 10;
00086       e -= 10;
00087     }
00088     while ((f & kUint64MSB) == 0) {
00089       f <<= 1;
00090       e--;
00091     }
00092     f_ = f;
00093     e_ = e;
00094   }
00095 
00096   static DiyFp Normalize(const DiyFp& a) {
00097     DiyFp result = a;
00098     result.Normalize();
00099     return result;
00100   }
00101 
00102   uint64_t f() const { return f_; }
00103   int e() const { return e_; }
00104 
00105   void set_f(uint64_t new_value) { f_ = new_value; }
00106   void set_e(int new_value) { e_ = new_value; }
00107 
00108  private:
00109   static const uint64_t kUint64MSB = static_cast<uint64_t>(1) << 63;
00110 
00111   uint64_t f_;
00112   int e_;
00113 };
