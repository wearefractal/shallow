00037 // A splay tree.  The config type parameter encapsulates the different
00038 // configurations of a concrete splay tree:
00039 //
00040 //   typedef Key: the key type
00041 //   typedef Value: the value type
00042 //   static const kNoKey: the dummy key used when no key is set
00043 //   static const kNoValue: the dummy value used to initialize nodes
00044 //   int (Compare)(Key& a, Key& b) -> {-1, 0, 1}: comparison function
00045 //
00046 // The tree is also parameterized by an allocation policy
00047 // (Allocator). The policy is used for allocating lists in the C free
00048 // store or the zone; see zone.h.
00049 
00050 // Forward defined as
00051 // template <typename Config, class Allocator = FreeStoreAllocationPolicy>
00052 //     class SplayTree;
00053 template <typename Config, class Allocator>
