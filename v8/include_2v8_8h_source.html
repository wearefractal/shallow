03831 
03832 template <size_t ptr_size> struct SmiTagging;
03833 
03834 // Smi constants for 32-bit systems.
03835 template <> struct SmiTagging<4> {
03836   static const int kSmiShiftSize = 0;
03837   static const int kSmiValueSize = 31;
03838   static inline int SmiToInt(internal::Object* value) {
03839     int shift_bits = kSmiTagSize + kSmiShiftSize;
03840     // Throw away top 32 bits and shift down (requires >> to be sign extending).
03841     return static_cast<int>(reinterpret_cast<intptr_t>(value)) >> shift_bits;
03842   }
03843 
03844   // For 32-bit systems any 2 bytes aligned pointer can be encoded as smi
03845   // with a plain reinterpret_cast.
03846   static const uintptr_t kEncodablePointerMask = 0x1;
03847   static const int kPointerToSmiShift = 0;
03848 };
03849 
03850 // Smi constants for 64-bit systems.
03851 template <> struct SmiTagging<8> {
03852   static const int kSmiShiftSize = 31;
03853   static const int kSmiValueSize = 32;
03854   static inline int SmiToInt(internal::Object* value) {
03855     int shift_bits = kSmiTagSize + kSmiShiftSize;
03856     // Shift down and throw away top 32 bits.
03857     return static_cast<int>(reinterpret_cast<intptr_t>(value) >> shift_bits);
03858   }
03859 
03860   // To maximize the range of pointers that can be encoded
03861   // in the available 32 bits, we require them to be 8 bytes aligned.
03862   // This gives 2 ^ (32 + 3) = 32G address space covered.
03863   // It might be not enough to cover stack allocated objects on some platforms.
03864   static const int kPointerAlignment = 3;
03865 
03866   static const uintptr_t kEncodablePointerMask =
03867       ~(uintptr_t(0xffffffff) << kPointerAlignment);
03868 
03869   static const int kPointerToSmiShift =
03870       kSmiTagSize + kSmiShiftSize - kPointerAlignment;
03871 };
