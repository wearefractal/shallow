00508   static Mode GetMode(Code* stub) {
00509     byte first_instruction = stub->instruction_start()[0];
00510     byte second_instruction = stub->instruction_start()[2];
00511 
00512     if (first_instruction == kTwoByteJumpInstruction) {
00513       return INCREMENTAL;

00527   static void Patch(Code* stub, Mode mode) {
00528     switch (mode) {
00529       case STORE_BUFFER_ONLY:
00530         ASSERT(GetMode(stub) == INCREMENTAL ||
00531                GetMode(stub) == INCREMENTAL_COMPACTION);
00532         stub->instruction_start()[0] = kTwoByteNopInstruction;
00533         stub->instruction_start()[2] = kFiveByteNopInstruction;
00534         break;
00535       case INCREMENTAL:
00536         ASSERT(GetMode(stub) == STORE_BUFFER_ONLY);
00537         stub->instruction_start()[0] = kTwoByteJumpInstruction;
00538         break;
00539       case INCREMENTAL_COMPACTION:

00541         stub->instruction_start()[0] = kTwoByteNopInstruction;
00542         stub->instruction_start()[2] = kFiveByteJumpInstruction;

00550   // This is a helper class for freeing up 3 scratch registers, where the third
00551   // is always ecx (needed for shift operations).  The input is two registers
00552   // that must be preserved and one scratch register provided by the caller.
00553   class RegisterAllocation {
00554    public:
00555     RegisterAllocation(Register object,
00556                        Register address,
00557                        Register scratch0)
00558         : object_orig_(object),
00559           address_orig_(address),
00560           scratch0_orig_(scratch0),
00561           object_(object),
00562           address_(address),
00563           scratch0_(scratch0) {
00564       ASSERT(!AreAliased(scratch0, object, address, no_reg));
00565       scratch1_ = GetRegThatIsNotEcxOr(object_, address_, scratch0_);
00566       if (scratch0.is(ecx)) {
00567         scratch0_ = GetRegThatIsNotEcxOr(object_, address_, scratch1_);
00568       }
00569       if (object.is(ecx)) {
00570         object_ = GetRegThatIsNotEcxOr(address_, scratch0_, scratch1_);
00571       }
00572       if (address.is(ecx)) {
00573         address_ = GetRegThatIsNotEcxOr(object_, scratch0_, scratch1_);
00574       }
00575       ASSERT(!AreAliased(scratch0_, object_, address_, ecx));
00576     }

00578     void Save(MacroAssembler* masm) {
00579       ASSERT(!address_orig_.is(object_));
00580       ASSERT(object_.is(object_orig_) || address_.is(address_orig_));
00581       ASSERT(!AreAliased(object_, address_, scratch1_, scratch0_));
00582       ASSERT(!AreAliased(object_orig_, address_, scratch1_, scratch0_));
00583       ASSERT(!AreAliased(object_, address_orig_, scratch1_, scratch0_));
00584       // We don't have to save scratch0_orig_ because it was given to us as
00585       // a scratch register.  But if we had to switch to a different reg then
00586       // we should save the new scratch0_.
00587       if (!scratch0_.is(scratch0_orig_)) masm->push(scratch0_);
00588       if (!ecx.is(scratch0_orig_) &&
00589           !ecx.is(object_orig_) &&
00590           !ecx.is(address_orig_)) {
00591         masm->push(ecx);
00592       }
00593       masm->push(scratch1_);
00594       if (!address_.is(address_orig_)) {
00595         masm->push(address_);
00596         masm->mov(address_, address_orig_);
00597       }
00598       if (!object_.is(object_orig_)) {
00599         masm->push(object_);
00600         masm->mov(object_, object_orig_);
00601       }
00602     }

00604     void Restore(MacroAssembler* masm) {
00605       // These will have been preserved the entire time, so we just need to move
00606       // them back.  Only in one case is the orig_ reg different from the plain
00607       // one, since only one of them can alias with ecx.
00608       if (!object_.is(object_orig_)) {
00609         masm->mov(object_orig_, object_);
00610         masm->pop(object_);
00611       }
00612       if (!address_.is(address_orig_)) {
00613         masm->mov(address_orig_, address_);
00614         masm->pop(address_);
00615       }
00616       masm->pop(scratch1_);
00617       if (!ecx.is(scratch0_orig_) &&
00618           !ecx.is(object_orig_) &&
00619           !ecx.is(address_orig_)) {
00620         masm->pop(ecx);
00621       }
00622       if (!scratch0_.is(scratch0_orig_)) masm->pop(scratch0_);
00623     }

00625     // If we have to call into C then we need to save and restore all caller-
00626     // saved registers that were not already preserved.  The caller saved
00627     // registers are eax, ecx and edx.  The three scratch registers (incl. ecx)
00628     // will be restored by other means so we don't bother pushing them here.
00629     void SaveCallerSaveRegisters(MacroAssembler* masm, SaveFPRegsMode mode) {
00630       if (!scratch0_.is(eax) && !scratch1_.is(eax)) masm->push(eax);
00631       if (!scratch0_.is(edx) && !scratch1_.is(edx)) masm->push(edx);
00632       if (mode == kSaveFPRegs) {
00633         CpuFeatures::Scope scope(SSE2);
00634         masm->sub(esp,
00635                   Immediate(kDoubleSize * (XMMRegister::kNumRegisters - 1)));
00636         // Save all XMM registers except XMM0.
00637         for (int i = XMMRegister::kNumRegisters - 1; i > 0; i--) {
00638           XMMRegister reg = XMMRegister::from_code(i);
00639           masm->movdbl(Operand(esp, (i - 1) * kDoubleSize), reg);
00640         }
00641       }
00642     }

00644     inline void RestoreCallerSaveRegisters(MacroAssembler*masm,
00645                                            SaveFPRegsMode mode) {
00646       if (mode == kSaveFPRegs) {
00647         CpuFeatures::Scope scope(SSE2);
00648         // Restore all XMM registers except XMM0.
00649         for (int i = XMMRegister::kNumRegisters - 1; i > 0; i--) {
00650           XMMRegister reg = XMMRegister::from_code(i);
00651           masm->movdbl(reg, Operand(esp, (i - 1) * kDoubleSize));
00652         }
00653         masm->add(esp,
00654                   Immediate(kDoubleSize * (XMMRegister::kNumRegisters - 1)));
00655       }
00656       if (!scratch0_.is(edx) && !scratch1_.is(edx)) masm->pop(edx);
00657       if (!scratch0_.is(eax) && !scratch1_.is(eax)) masm->pop(eax);
00658     }
