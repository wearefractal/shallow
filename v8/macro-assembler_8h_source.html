00075 class FrameScope {
00076  public:
00077   explicit FrameScope(MacroAssembler* masm, StackFrame::Type type)
00078       : masm_(masm), type_(type), old_has_frame_(masm->has_frame()) {
00079     masm->set_has_frame(true);
00080     if (type != StackFrame::MANUAL && type_ != StackFrame::NONE) {
00081       masm->EnterFrame(type);
00082     }
00083   }
00084 
00085   ~FrameScope() {
00086     if (type_ != StackFrame::MANUAL && type_ != StackFrame::NONE) {
00087       masm_->LeaveFrame(type_);
00088     }
00089     masm_->set_has_frame(old_has_frame_);
00090   }
00091 
00092   // Normally we generate the leave-frame code when this object goes
00093   // out of scope.  Sometimes we may need to generate the code somewhere else
00094   // in addition.  Calling this will achieve that, but the object stays in
00095   // scope, the MacroAssembler is still marked as being in a frame scope, and
00096   // the code will be generated again when it goes out of scope.
00097   void GenerateLeaveFrame() {
00098     masm_->LeaveFrame(type_);
00099   }
00100 
00101  private:
00102   MacroAssembler* masm_;
00103   StackFrame::Type type_;
00104   bool old_has_frame_;
00105 };
