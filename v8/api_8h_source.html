00322 // Tracks string usage to help make better decisions when
00323 // externalizing strings.
00324 //
00325 // Implementation note: internally this class only tracks fresh
00326 // strings and keeps a single use counter for them.
00327 class StringTracker {
00328  public:
00329   // Records that the given string's characters were copied to some
00330   // external buffer. If this happens often we should honor
00331   // externalization requests for the string.
00332   void RecordWrite(Handle<String> string) {
00333     Address address = reinterpret_cast<Address>(*string);
00334     Address top = isolate_->heap()->NewSpaceTop();
00335     if (IsFreshString(address, top)) {
00336       IncrementUseCount(top);
00337     }
00338   }
00339 
00340   // Estimates freshness and use frequency of the given string based
00341   // on how close it is to the new space top and the recorded usage
00342   // history.
00343   inline bool IsFreshUnusedString(Handle<String> string) {
00344     Address address = reinterpret_cast<Address>(*string);
00345     Address top = isolate_->heap()->NewSpaceTop();
00346     return IsFreshString(address, top) && IsUseCountLow(top);
00347   }
00348 
00349  private:
00350   StringTracker() : use_count_(0), last_top_(NULL), isolate_(NULL) { }
00351 
00352   static inline bool IsFreshString(Address string, Address top) {
00353     return top - kFreshnessLimit <= string && string <= top;
00354   }
00355 
00356   inline bool IsUseCountLow(Address top) {
00357     if (last_top_ != top) return true;
00358     return use_count_ < kUseLimit;
00359   }
00360 
00361   inline void IncrementUseCount(Address top) {
00362     if (last_top_ != top) {
00363       use_count_ = 0;
00364       last_top_ = top;
00365     }
00366     ++use_count_;
00367   }
00368 
00369   // Single use counter shared by all fresh strings.
00370   int use_count_;
00371 
00372   // Last new space top when the use count above was valid.
00373   Address last_top_;
00374 
00375   Isolate* isolate_;
00376 
00377   // How close to the new space top a fresh string has to be.
00378   static const int kFreshnessLimit = 1024;
00379 
00380   // The number of uses required to consider a string useful.
00381   static const int kUseLimit = 32;
00382 
00383   friend class Isolate;
00384 
00385   DISALLOW_COPY_AND_ASSIGN(StringTracker);
00386 };

00531 void HandleScopeImplementer::DeleteExtensions(internal::Object** prev_limit) {
00532   while (!blocks_.is_empty()) {
00533     internal::Object** block_start = blocks_.last();
00534     internal::Object** block_limit = block_start + kHandleBlockSize;
00535 #ifdef DEBUG
00536     // NoHandleAllocation may make the prev_limit to point inside the block.
00537     if (block_start <= prev_limit && prev_limit <= block_limit) break;
00538 #else
00539     if (prev_limit == block_limit) break;
00540 #endif
00541 
00542     blocks_.RemoveLast();
00543 #ifdef DEBUG
00544     v8::ImplementationUtilities::ZapHandleRange(block_start, block_limit);
00545 #endif
00546     if (spare_ != NULL) {
00547       DeleteArray(spare_);
00548     }
00549     spare_ = block_start;
00550   }
00551   ASSERT((blocks_.is_empty() && prev_limit == NULL) ||
00552          (!blocks_.is_empty() && prev_limit != NULL));
00553 }
