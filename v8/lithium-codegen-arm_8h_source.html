00374   class PushSafepointRegistersScope BASE_EMBEDDED {
00375    public:
00376     PushSafepointRegistersScope(LCodeGen* codegen,
00377                                 Safepoint::Kind kind)
00378         : codegen_(codegen) {
00379       ASSERT(codegen_->expected_safepoint_kind_ == Safepoint::kSimple);
00380       codegen_->expected_safepoint_kind_ = kind;
00381 
00382       switch (codegen_->expected_safepoint_kind_) {
00383         case Safepoint::kWithRegisters:
00384           codegen_->masm_->PushSafepointRegisters();
00385           break;
00386         case Safepoint::kWithRegistersAndDoubles:
00387           codegen_->masm_->PushSafepointRegistersAndDoubles();
00388           break;
00389         default:
00390           UNREACHABLE();
00391       }
00392     }
00393 
00394     ~PushSafepointRegistersScope() {
00395       Safepoint::Kind kind = codegen_->expected_safepoint_kind_;
00396       ASSERT((kind & Safepoint::kWithRegisters) != 0);
00397       switch (kind) {
00398         case Safepoint::kWithRegisters:
00399           codegen_->masm_->PopSafepointRegisters();
00400           break;
00401         case Safepoint::kWithRegistersAndDoubles:
00402           codegen_->masm_->PopSafepointRegistersAndDoubles();
00403           break;
00404         default:
00405           UNREACHABLE();
00406       }
00407       codegen_->expected_safepoint_kind_ = Safepoint::kSimple;
00408     }
00409 
00410    private:
00411     LCodeGen* codegen_;
00412   };
