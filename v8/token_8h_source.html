00036 // TOKEN_LIST takes a list of 3 macros M, all of which satisfy the
00037 // same signature M(name, string, precedence), where name is the
00038 // symbolic token name, string is the corresponding syntactic symbol
00039 // (or NULL, for literals), and precedence is the precedence (or 0).
00040 // The parameters are invoked for token categories as follows:
00041 //
00042 //   T: Non-keyword tokens
00043 //   K: Keyword tokens
00044 
00045 // IGNORE_TOKEN is a convenience macro that can be supplied as
00046 // an argument (at any position) for a TOKEN_LIST call. It does
00047 // nothing with tokens belonging to the respective category.
00048 
00049 #define IGNORE_TOKEN(name, string, precedence)

00184 class Token {
00185  public:
00186   // All token values.
00187 #define T(name, string, precedence) name,
00188   enum Value {
00189     TOKEN_LIST(T, T)
00190     NUM_TOKENS
00191   };
00192 #undef T
00193 
00194   // Returns a string corresponding to the C++ token name
00195   // (e.g. "LT" for the token LT).
00196   static const char* Name(Value tok) {
00197     ASSERT(tok < NUM_TOKENS);  // tok is unsigned
00198     return name_[tok];
00199   }
00200 
00201   // Predicates
00202   static bool IsKeyword(Value tok) {
00203     return token_type[tok] == 'K';
00204   }
00205 
00206   static bool IsAssignmentOp(Value tok) {
00207     return INIT_VAR <= tok && tok <= ASSIGN_MOD;
00208   }
00209 
00210   static bool IsBinaryOp(Value op) {
00211     return COMMA <= op && op <= MOD;
00212   }
00213 
00214   static bool IsCompareOp(Value op) {
00215     return EQ <= op && op <= IN;
00216   }
00217 
00218   static bool IsOrderedRelationalCompareOp(Value op) {
00219     return op == LT || op == LTE || op == GT || op == GTE;
00220   }
00221 
00222   static bool IsEqualityOp(Value op) {
00223     return op == EQ || op == EQ_STRICT;
00224   }
00225 
00226   static Value NegateCompareOp(Value op) {
00227     ASSERT(IsCompareOp(op));
00228     switch (op) {
00229       case EQ: return NE;
00230       case NE: return EQ;
00231       case EQ_STRICT: return NE_STRICT;
00232       case LT: return GTE;
00233       case GT: return LTE;
00234       case LTE: return GT;
00235       case GTE: return LT;
00236       default:
00237         return op;
00238     }
00239   }
00240 
00241   static Value InvertCompareOp(Value op) {
00242     ASSERT(IsCompareOp(op));
00243     switch (op) {
00244       case EQ: return NE;
00245       case NE: return EQ;
00246       case EQ_STRICT: return NE_STRICT;
00247       case LT: return GT;
00248       case GT: return LT;
00249       case LTE: return GTE;
00250       case GTE: return LTE;
00251       default:
00252         return op;
00253     }
00254   }
00255 
00256   static bool IsBitOp(Value op) {
00257     return (BIT_OR <= op && op <= SHR) || op == BIT_NOT;
00258   }
00259 
00260   static bool IsUnaryOp(Value op) {
00261     return (NOT <= op && op <= VOID) || op == ADD || op == SUB;
00262   }
00263 
00264   static bool IsCountOp(Value op) {
00265     return op == INC || op == DEC;
00266   }
00267 
00268   static bool IsShiftOp(Value op) {
00269     return (SHL <= op) && (op <= SHR);
00270   }
00271 
00272   // Returns a string corresponding to the JS token string
00273   // (.e., "<" for the token LT) or NULL if the token doesn't
00274   // have a (unique) string (e.g. an IDENTIFIER).
00275   static const char* String(Value tok) {
00276     ASSERT(tok < NUM_TOKENS);  // tok is unsigned.
00277     return string_[tok];
00278   }
00279 
00280   // Returns the precedence > 0 for binary and compare
00281   // operators; returns 0 otherwise.
00282   static int Precedence(Value tok) {
00283     ASSERT(tok < NUM_TOKENS);  // tok is unsigned.
00284     return precedence_[tok];
00285   }
00286 
00287  private:
00288   static const char* const name_[NUM_TOKENS];
00289   static const char* const string_[NUM_TOKENS];
00290   static const int8_t precedence_[NUM_TOKENS];
00291   static const char token_type[NUM_TOKENS];
00292 };
