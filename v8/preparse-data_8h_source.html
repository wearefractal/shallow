00196   template <typename Char>
00197   static int vector_hash(Vector<const Char> string) {
00198     int hash = 0;
00199     for (int i = 0; i < string.length(); i++) {
00200       int c = static_cast<int>(string[i]);
00201       hash += c;
00202       hash += (hash << 10);
00203       hash ^= (hash >> 6);
00204     }
00205     return hash;
00206   }

00208   static bool vector_compare(void* a, void* b) {
00209     Key* string1 = reinterpret_cast<Key*>(a);
00210     Key* string2 = reinterpret_cast<Key*>(b);
00211     if (string1->is_ascii != string2->is_ascii) return false;
00212     int length = string1->literal_bytes.length();
00213     if (string2->literal_bytes.length() != length) return false;
00214     return memcmp(string1->literal_bytes.start(),
00215                   string2->literal_bytes.start(), length) == 0;
00216   }
