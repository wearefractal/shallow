00056   static v8::Arguments NewArguments(internal::Object** implicit_args,
00057                                     internal::Object** argv, int argc,
00058                                     bool is_construct_call) {
00059     ASSERT(implicit_args[v8::Arguments::kCalleeIndex]->IsJSFunction());
00060     ASSERT(implicit_args[v8::Arguments::kHolderIndex]->IsHeapObject());
00061     // The implicit isolate argument is not tagged and looks like a SMI.
00062     ASSERT(implicit_args[v8::Arguments::kIsolateIndex]->IsSmi());
00063 
00064     return v8::Arguments(implicit_args, argv, argc, is_construct_call);
