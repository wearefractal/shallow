00118 class Utf16 {
00119  public:
00120   static inline bool IsLeadSurrogate(int code) {
00121     if (code == kNoPreviousCharacter) return false;
00122     return (code & 0xfc00) == 0xd800;
00123   }
00124   static inline bool IsTrailSurrogate(int code) {
00125     if (code == kNoPreviousCharacter) return false;
00126     return (code & 0xfc00) == 0xdc00;
00127   }
00128 
00129   static inline int CombineSurrogatePair(uchar lead, uchar trail) {
00130     return 0x10000 + ((lead & 0x3ff) << 10) + (trail & 0x3ff);
00131   }
00132   static const int kNoPreviousCharacter = -1;
00133   static const uchar kMaxNonSurrogateCharCode = 0xffff;
00134   // Encoding a single UTF-16 code unit will produce 1, 2 or 3 bytes
00135   // of UTF-8 data.  The special case where the unit is a surrogate
00136   // trail produces 1 byte net, because the encoding of the pair is
00137   // 4 bytes and the 3 bytes that were used to encode the lead surrogate
00138   // can be reclaimed.
00139   static const int kMaxExtraUtf8BytesForOneUtf16CodeUnit = 3;
00140   // One UTF-16 surrogate is endoded (illegally) as 3 UTF-8 bytes.
00141   // The illegality stems from the surrogate not being part of a pair.
00142   static const int kUtf8BytesToCodeASurrogate = 3;
00143   static inline uchar LeadSurrogate(int char_code) {
00144     return 0xd800 + (((char_code - 0x10000) >> 10) & 0x3ff);
00145   }
00146   static inline uchar TrailSurrogate(int char_code) {
00147     return 0xdc00 + (char_code & 0x3ff);
00148   }
00149 };
