00046 // CPU Registers.
00047 //
00048 // 1) We would prefer to use an enum, but enum values are assignment-
00049 // compatible with int, which has caused code-generation bugs.
00050 //
00051 // 2) We would prefer to use a class instead of a struct but we don't like
00052 // the register initialization to depend on the particular initialization
00053 // order (which appears to be different on OS X, Linux, and Windows for the
00054 // installed versions of C++ we tried). Using a struct permits C-style
00055 // "initialization". Also, the Register objects cannot be const as this
00056 // forces initialization stubs in MSVC, making us dependent on initialization
00057 // order.
00058 //
00059 // 3) By not using an enum, we are possibly preventing the compiler from
00060 // doing certain constant folds, which may significantly reduce the
00061 // code generated for some assembly instructions (because they boil down
00062 // to a few constants). If this is a problem, we could change the code
00063 // such that we use an enum in optimized mode, and the struct in debug
00064 // mode. This way we get the compile-time error checking in debug mode
00065 // and best performance in optimized code.
00066 //
00067 struct Register {
00068   static const int kNumAllocatableRegisters = 6;
00069   static const int kNumRegisters = 8;
00070 
00071   static inline const char* AllocationIndexToString(int index);
00072 
00073   static inline int ToAllocationIndex(Register reg);
00074 
00075   static inline Register FromAllocationIndex(int index);
00076 
00077   static Register from_code(int code) {
00078     ASSERT(code >= 0);
00079     ASSERT(code < kNumRegisters);
00080     Register r = { code };
00081     return r;
00082   }
00083   bool is_valid() const { return 0 <= code_ && code_ < kNumRegisters; }
00084   bool is(Register reg) const { return code_ == reg.code_; }
00085   // eax, ebx, ecx and edx are byte registers, the rest are not.
00086   bool is_byte_register() const { return code_ <= 3; }
00087   int code() const {
00088     ASSERT(is_valid());
00089     return code_;
00090   }
00091   int bit() const {
00092     ASSERT(is_valid());
00093     return 1 << code_;
00094   }
00095 
00096   // Unfortunately we can't make this private in a struct.
00097   int code_;
00098 };

00229 // Returns the equivalent of !cc.
00230 // Negation of the default no_condition (-1) results in a non-default
00231 // no_condition value (-2). As long as tests for no_condition check
00232 // for condition < 0, this will work as expected.
00233 inline Condition NegateCondition(Condition cc) {
00234   return static_cast<Condition>(cc ^ 1);
00235 }

00238 // Corresponds to transposing the operands of a comparison.
00239 inline Condition ReverseCondition(Condition cc) {
00240   switch (cc) {
00241     case below:
00242       return above;
00243     case above:
00244       return below;
00245     case above_equal:
00246       return below_equal;
00247     case below_equal:
00248       return above_equal;
00249     case less:
00250       return greater;
00251     case greater:
00252       return less;
00253     case greater_equal:
00254       return less_equal;
00255     case less_equal:
00256       return greater_equal;
00257     default:
00258       return cc;
00259   };
00260 }

00389 // -----------------------------------------------------------------------------
00390 // A Displacement describes the 32bit immediate field of an instruction which
00391 // may be used together with a Label in order to refer to a yet unknown code
00392 // position. Displacements stored in the instruction stream are used to describe
00393 // the instruction and to chain a list of instructions using the same Label.
00394 // A Displacement contains 2 different fields:
00395 //
00396 // next field: position of next displacement in the chain (0 = end of list)
00397 // type field: instruction type
00398 //
00399 // A next value of null (0) indicates the end of a chain (note that there can
00400 // be no displacement at position zero, because there is always at least one
00401 // instruction byte before the displacement).
00402 //
00403 // Displacement _data field layout
00404 //
00405 // |31.....2|1......0|
00406 // [  next  |  type  |
00407 
00408 class Displacement BASE_EMBEDDED {
00409  public:
00410   enum Type {
00411     UNCONDITIONAL_JUMP,
00412     CODE_RELATIVE,
00413     OTHER
00414   };
00415 
00416   int data() const { return data_; }
00417   Type type() const { return TypeField::decode(data_); }
00418   void next(Label* L) const {
00419     int n = NextField::decode(data_);
00420     n > 0 ? L->link_to(n) : L->Unuse();
00421   }
00422   void link_to(Label* L) { init(L, type()); }
00423 
00424   explicit Displacement(int data) { data_ = data; }
00425 
00426   Displacement(Label* L, Type type) { init(L, type); }
00427 
00428   void print() {
00429     PrintF("%s (%x) ", (type() == UNCONDITIONAL_JUMP ? "jmp" : "[other]"),
00430                        NextField::decode(data_));
00431   }
00432 
00433  private:
00434   int data_;
00435 
00436   class TypeField: public BitField<Type, 0, 2> {};
00437   class NextField: public BitField<int,  2, 32-2> {};
00438 
00439   void init(Label* L, Type type);
00440 };

00444 // CpuFeatures keeps track of which features are supported by the target CPU.
00445 // Supported features must be enabled by a Scope before use.
00446 // Example:
00447 //   if (CpuFeatures::IsSupported(SSE2)) {
00448 //     CpuFeatures::Scope fscope(SSE2);
00449 //     // Generate SSE2 floating point code.
00450 //   } else {
00451 //     // Generate standard x87 floating point code.
00452 //   }
00453 class CpuFeatures : public AllStatic {
00454  public:
00455   // Detect features of the target CPU. Set safe defaults if the serializer
00456   // is enabled (snapshots must be portable).
00457   static void Probe();
00458 
00459   // Check whether a feature is supported by the target CPU.
00460   static bool IsSupported(CpuFeature f) {
00461     ASSERT(initialized_);
00462     if (f == SSE2 && !FLAG_enable_sse2) return false;
00463     if (f == SSE3 && !FLAG_enable_sse3) return false;
00464     if (f == SSE4_1 && !FLAG_enable_sse4_1) return false;
00465     if (f == CMOV && !FLAG_enable_cmov) return false;
00466     if (f == RDTSC && !FLAG_enable_rdtsc) return false;
00467     return (supported_ & (static_cast<uint64_t>(1) << f)) != 0;
00468   }
00469 
00470 #ifdef DEBUG
00471   // Check whether a feature is currently enabled.
00472   static bool IsEnabled(CpuFeature f) {
00473     ASSERT(initialized_);
00474     Isolate* isolate = Isolate::UncheckedCurrent();
00475     if (isolate == NULL) {
00476       // When no isolate is available, work as if we're running in
00477       // release mode.
00478       return IsSupported(f);
00479     }
00480     uint64_t enabled = isolate->enabled_cpu_features();
00481     return (enabled & (static_cast<uint64_t>(1) << f)) != 0;
00482   }
00483 #endif
