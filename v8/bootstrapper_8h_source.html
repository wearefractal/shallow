00038 // A SourceCodeCache uses a FixedArray to store pairs of
00039 // (AsciiString*, JSFunction*), mapping names of native code files
00040 // (runtime.js, etc.) to precompiled functions. Instead of mapping
00041 // names to functions it might make sense to let the JS2C tool
00042 // generate an index for each native JS file.
00043 class SourceCodeCache BASE_EMBEDDED {
00044  public:
00045   explicit SourceCodeCache(Script::Type type): type_(type), cache_(NULL) { }
00046 
00047   void Initialize(bool create_heap_objects) {
00048     cache_ = create_heap_objects ? HEAP->empty_fixed_array() : NULL;
00049   }
00050 
00051   void Iterate(ObjectVisitor* v) {
00052     v->VisitPointer(BitCast<Object**, FixedArray**>(&cache_));
00053   }
00054 
00055   bool Lookup(Vector<const char> name, Handle<SharedFunctionInfo>* handle) {
00056     for (int i = 0; i < cache_->length(); i+=2) {
00057       SeqAsciiString* str = SeqAsciiString::cast(cache_->get(i));
00058       if (str->IsEqualTo(name)) {
00059         *handle = Handle<SharedFunctionInfo>(
00060             SharedFunctionInfo::cast(cache_->get(i + 1)));
00061         return true;
00062       }
00063     }
00064     return false;
00065   }
00066 
00067   void Add(Vector<const char> name, Handle<SharedFunctionInfo> shared) {
00068     HandleScope scope;
00069     int length = cache_->length();
00070     Handle<FixedArray> new_array =
00071         FACTORY->NewFixedArray(length + 2, TENURED);
00072     cache_->CopyTo(0, *new_array, 0, cache_->length());
00073     cache_ = *new_array;
00074     Handle<String> str = FACTORY->NewStringFromAscii(name, TENURED);
00075     cache_->set(length, *str);
00076     cache_->set(length + 1, *shared);
00077     Script::cast(shared->script())->set_type(Smi::FromInt(type_));
00078   }
00079 
00080  private:
00081   Script::Type type_;
00082   FixedArray* cache_;
00083   DISALLOW_COPY_AND_ASSIGN(SourceCodeCache);
00084 };
