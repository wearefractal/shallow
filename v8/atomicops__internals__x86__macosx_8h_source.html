00038 inline Atomic32 NoBarrier_CompareAndSwap(volatile Atomic32* ptr,
00039                                          Atomic32 old_value,
00040                                          Atomic32 new_value) {
00041   Atomic32 prev_value;
00042   do {
00043     if (OSAtomicCompareAndSwap32(old_value, new_value,
00044                                  const_cast<Atomic32*>(ptr))) {
00045       return old_value;
00046     }
00047     prev_value = *ptr;
00048   } while (prev_value == old_value);
00049   return prev_value;
00050 }
