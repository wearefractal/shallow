00037 // StatsCounters is an interface for plugging into external
00038 // counters for monitoring.  Counters can be looked up and

00041 class StatsTable {
00042  public:
00043   // Register an application-defined function where
00044   // counters can be looked up.
00045   void SetCounterFunction(CounterLookupCallback f) {
00046     lookup_function_ = f;
00047   }
00048 
00049   // Register an application-defined function to create
00050   // a histogram for passing to the AddHistogramSample function
00051   void SetCreateHistogramFunction(CreateHistogramCallback f) {
00052     create_histogram_function_ = f;
00053   }
00054 
00055   // Register an application-defined function to add a sample
00056   // to a histogram created with CreateHistogram function
00057   void SetAddHistogramSampleFunction(AddHistogramSampleCallback f) {
00058     add_histogram_sample_function_ = f;
00059   }
00060 
00061   bool HasCounterFunction() const {
00062     return lookup_function_ != NULL;
00063   }
00064 
00065   // Lookup the location of a counter by name.  If the lookup
00066   // is successful, returns a non-NULL pointer for writing the
00067   // value of the counter.  Each thread calling this function
00068   // may receive a different location to store it's counter.
00069   // The return value must not be cached and re-used across
00070   // threads, although a single thread is free to cache it.
00071   int* FindLocation(const char* name) {
00072     if (!lookup_function_) return NULL;
00073     return lookup_function_(name);
00074   }
00075 
00076   // Create a histogram by name. If the create is successful,
00077   // returns a non-NULL pointer for use with AddHistogramSample
00078   // function. min and max define the expected minimum and maximum
00079   // sample values. buckets is the maximum number of buckets
00080   // that the samples will be grouped into.
00081   void* CreateHistogram(const char* name,
00082                         int min,
00083                         int max,
00084                         size_t buckets) {
00085     if (!create_histogram_function_) return NULL;
00086     return create_histogram_function_(name, min, max, buckets);
00087   }
00088 
00089   // Add a sample to a histogram created with the CreateHistogram
00090   // function.
00091   void AddHistogramSample(void* histogram, int sample) {
00092     if (!add_histogram_sample_function_) return;
00093     return add_histogram_sample_function_(histogram, sample);
00094   }
00095 
00096  private:
00097   StatsTable();
00098 
00099   CounterLookupCallback lookup_function_;
00100   CreateHistogramCallback create_histogram_function_;
00101   AddHistogramSampleCallback add_histogram_sample_function_;
00102 
00103   friend class Isolate;
00104 
00105   DISALLOW_COPY_AND_ASSIGN(StatsTable);
00106 };

00108 // StatsCounters are dynamically created values which can be tracked in
00109 // the StatsTable.  They are designed to be lightweight to create and
00110 // easy to use.
00111 //
00112 // Internally, a counter represents a value in a row of a StatsTable.
00113 // The row has a 32bit value for each process/thread in the table and also
00114 // a name (stored in the table metadata).  Since the storage location can be
00115 // thread-specific, this class cannot be shared across threads.
00116 //
00117 // This class is designed to be POD initialized.  It will be registered with
00118 // the counter system on first use.  For example:
00119 //   StatsCounter c = { "c:myctr", NULL, false };
00120 struct StatsCounter {
00121   const char* name_;
00122   int* ptr_;
00123   bool lookup_done_;
00124 
00125   // Sets the counter to a specific value.
00126   void Set(int value) {
00127     int* loc = GetPtr();
00128     if (loc) *loc = value;
00129   }
00130 
00131   // Increments the counter.
00132   void Increment() {
00133     int* loc = GetPtr();
00134     if (loc) (*loc)++;
00135   }
00136 
00137   void Increment(int value) {
00138     int* loc = GetPtr();
00139     if (loc)
00140       (*loc) += value;
00141   }
00142 
00143   // Decrements the counter.
00144   void Decrement() {
00145     int* loc = GetPtr();
00146     if (loc) (*loc)--;
00147   }
00148 
00149   void Decrement(int value) {
00150     int* loc = GetPtr();
00151     if (loc) (*loc) -= value;
00152   }
00153 
00154   // Is this counter enabled?
00155   // Returns false if table is full.
00156   bool Enabled() {
00157     return GetPtr() != NULL;
00158   }
00159 
00160   // Get the internal pointer to the counter. This is used
00161   // by the code generator to emit code that manipulates a
00162   // given counter without calling the runtime system.
00163   int* GetInternalPointer() {
00164     int* loc = GetPtr();
00165     ASSERT(loc != NULL);
00166     return loc;
00167   }
00168 
00169  protected:
00170   // Returns the cached address of this counter location.
00171   int* GetPtr() {
00172     if (lookup_done_)
00173       return ptr_;
00174     lookup_done_ = true;
00175     ptr_ = FindLocationInStatsTable();
00176     return ptr_;
00177   }
00178 
00179  private:
00180   int* FindLocationInStatsTable() const;
00181 };
