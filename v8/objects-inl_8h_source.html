00148 bool Object::IsInstanceOf(FunctionTemplateInfo* expected) {
00149   // There is a constraint on the object; check.
00150   if (!this->IsJSObject()) return false;
00151   // Fetch the constructor function of the object.
00152   Object* cons_obj = JSObject::cast(this)->map()->constructor();
00153   if (!cons_obj->IsJSFunction()) return false;
00154   JSFunction* fun = JSFunction::cast(cons_obj);
00155   // Iterate through the chain of inheriting function templates to
00156   // see if the required one occurs.
00157   for (Object* type = fun->shared()->function_data();
00158        type->IsFunctionTemplateInfo();
00159        type = FunctionTemplateInfo::cast(type)->parent_template()) {
00160     if (type == expected) return true;
00161   }
00162   // Didn't find the required type in the inheritance chain.
00163   return false;
00164 }

00552       length % DeoptimizationInputData::kDeoptEntrySize == 0;

01063 bool Smi::IsValid(intptr_t value) {
01064 #ifdef DEBUG
01065   bool in_range = (value >= kMinValue) && (value <= kMaxValue);
01066 #endif
01067 
01068 #ifdef V8_TARGET_ARCH_X64
01069   // To be representable as a long smi, the value must be a 32-bit integer.
01070   bool result = (value == static_cast<int32_t>(value));
01071 #else
01072   // To be representable as an tagged small integer, the two
01073   // most-significant bits of 'value' must be either 00 or 11 due to
01074   // sign-extension. To check this we add 01 to the two
01075   // most-significant bits, and check if the most-significant bit is 0
01076   //
01077   // CAUTION: The original code below:
01078   // bool result = ((value + 0x40000000) & 0x80000000) == 0;
01079   // may lead to incorrect results according to the C language spec, and
01080   // in fact doesn't work correctly with gcc4.1.1 in some cases: The
01081   // compiler may produce undefined results in case of signed integer
01082   // overflow. The computation must be done w/ unsigned ints.
01083   bool result = (static_cast<uintptr_t>(value + 0x40000000U) < 0x80000000U);
01084 #endif
01085   ASSERT(result == in_range);
01086   return result;
01087 }

01441 int JSObject::GetHeaderSize() {
01442   InstanceType type = map()->instance_type();
01443   // Check for the most common kind of JavaScript object before
01444   // falling into the generic switch. This speeds up the internal
01445   // field operations considerably on average.
01446   if (type == JS_OBJECT_TYPE) return JSObject::kHeaderSize;
01447   switch (type) {
01448     case JS_MODULE_TYPE:
01449       return JSModule::kSize;
01450     case JS_GLOBAL_PROXY_TYPE:
01451       return JSGlobalProxy::kSize;

01453       return JSGlobalObject::kSize;
01454     case JS_BUILTINS_OBJECT_TYPE:
01455       return JSBuiltinsObject::kSize;
01456     case JS_FUNCTION_TYPE:
01457       return JSFunction::kSize;
01458     case JS_VALUE_TYPE:
01459       return JSValue::kSize;
01460     case JS_DATE_TYPE:
01461       return JSDate::kSize;
01462     case JS_ARRAY_TYPE:
01463       return JSArray::kSize;
01464     case JS_WEAK_MAP_TYPE:
01465       return JSWeakMap::kSize;
01466     case JS_REGEXP_TYPE:

01604     WRITE_FIELD(this, offset, filler_value);

01978   switch (GetType(descriptor_number)) {
01979     case MAP_TRANSITION:
01980     case CONSTANT_TRANSITION:

02061 template<typename Shape, typename Key>
02062 int HashTable<Shape, Key>::ComputeCapacity(int at_least_space_for) {
02063   const int kMinCapacity = 32;
02064   int capacity = RoundUpToPowerOf2(at_least_space_for * 2);
02065   if (capacity < kMinCapacity) {
02066     capacity = kMinCapacity;  // Guarantee min capacity.
02067   }
02068   return capacity;

02077 
02078 // Find entry for key otherwise return kNotFound.
02079 template<typename Shape, typename Key>
02080 int HashTable<Shape, Key>::FindEntry(Isolate* isolate, Key key) {
02081   uint32_t capacity = Capacity();
02082   uint32_t entry = FirstProbe(HashTable<Shape, Key>::Hash(key), capacity);
02083   uint32_t count = 1;
02084   // EnsureCapacity will guarantee the hash table is never full.
02085   while (true) {
02086     Object* element = KeyAt(entry);
02087     // Empty entry.
02088     if (element == isolate->heap()->raw_unchecked_undefined_value()) break;
02089     if (element != isolate->heap()->raw_unchecked_the_hole_value() &&
02090         Shape::IsMatch(key, element)) return entry;
02091     entry = NextProbe(entry, count++, capacity);
02092   }
02093   return kNotFound;
02094 }

02747 int HeapObject::SizeFromMap(Map* map) {
02748   int instance_size = map->instance_size();
02749   if (instance_size != kVariableSizeSentinel) return instance_size;
02750   // We can ignore the "symbol" bit becase it is only set for symbols
02751   // and implies a string type.
02752   int instance_type = static_cast<int>(map->instance_type()) & ~kIsSymbolMask;
02753   // Only inline the most frequent cases.
02754   if (instance_type == FIXED_ARRAY_TYPE) {
02755     return FixedArray::BodyDescriptor::SizeOf(map, this);
02756   }
02757   if (instance_type == ASCII_STRING_TYPE) {
02758     return SeqAsciiString::SizeFor(
02759         reinterpret_cast<SeqAsciiString*>(this)->length());
02760   }
02761   if (instance_type == BYTE_ARRAY_TYPE) {
02762     return reinterpret_cast<ByteArray*>(this)->ByteArraySize();
02763   }
02764   if (instance_type == FREE_SPACE_TYPE) {
02765     return reinterpret_cast<FreeSpace*>(this)->size();
02766   }
02767   if (instance_type == STRING_TYPE) {
02768     return SeqTwoByteString::SizeFor(
02769         reinterpret_cast<SeqTwoByteString*>(this)->length());
02770   }
02771   if (instance_type == FIXED_DOUBLE_ARRAY_TYPE) {
02772     return FixedDoubleArray::SizeFor(
02773         reinterpret_cast<FixedDoubleArray*>(this)->length());
02774   }
02775   ASSERT(instance_type == CODE_TYPE);
02776   return reinterpret_cast<Code*>(this)->CodeSize();
02777 }

03700 void SharedFunctionInfo::TryReenableOptimization() {
03701   int tries = opt_reenable_tries();
03702   if (tries == Smi::kMaxValue) {
03703     tries = 0;
03704   }
03705   set_opt_reenable_tries(tries + 1);
03706   // We reenable optimization whenever the number of tries is a large
03707   // enough power of 2.
03708   if (tries >= 4 && (((tries - 1) & tries) == 0)) {
03709     set_optimization_disabled(false);
03710     set_opt_count(0);
03711     code()->set_optimizable(true);
03712   }
03713 }

04467 void StringHasher::AddCharacter(uint32_t c) {
04468   if (c > unibrow::Utf16::kMaxNonSurrogateCharCode) {
04469     AddSurrogatePair(c);  // Not inlined.
04470     return;
04471   }
04472   // Use the Jenkins one-at-a-time hash function to update the hash
04473   // for the given character.
04474   raw_running_hash_ += c;
04475   raw_running_hash_ += (raw_running_hash_ << 10);
04476   raw_running_hash_ ^= (raw_running_hash_ >> 6);
04477   // Incremental array index computation.
04478   if (is_array_index_) {
04479     if (c < '0' || c > '9') {
04480       is_array_index_ = false;
04481     } else {
04482       int d = c - '0';
04483       if (is_first_char_) {
04484         is_first_char_ = false;
04485         if (c == '0' && length_ > 1) {
04486           is_array_index_ = false;
04487           return;
04488         }
04489       }
04490       if (array_index_ > 429496729U - ((d + 2) >> 3)) {
04491         is_array_index_ = false;
04492       } else {
04493         array_index_ = array_index_ * 10 + d;
04494       }
04495     }
04496   }
04497 }

04500 void StringHasher::AddCharacterNoIndex(uint32_t c) {
04501   ASSERT(!is_array_index());
04502   if (c > unibrow::Utf16::kMaxNonSurrogateCharCode) {
04503     AddSurrogatePairNoIndex(c);  // Not inlined.
04504     return;
04505   }
04506   raw_running_hash_ += c;
04507   raw_running_hash_ += (raw_running_hash_ << 10);
04508   raw_running_hash_ ^= (raw_running_hash_ >> 6);
04509 }

04512 uint32_t StringHasher::GetHash() {
04513   // Get the calculated raw hash value and do some more bit ops to distribute
04514   // the hash further. Ensure that we never return zero as the hash value.
04515   uint32_t result = raw_running_hash_;
04516   result += (result << 3);
04517   result ^= (result >> 11);
04518   result += (result << 15);
04519   if ((result & String::kHashBitMask) == 0) {
04520     result = 27;
04521   }
04522   return result;
04523 }

04760 void JSArray::EnsureSize(int required_size) {
04761   ASSERT(HasFastTypeElements());
04762   FixedArray* elts = FixedArray::cast(elements());
04763   const int kArraySizeThatFitsComfortablyInNewSpace = 128;
04764   if (elts->length() < required_size) {
04765     // Doubling in size would be overkill, but leave some slack to avoid
04766     // constantly growing.
04767     Expand(required_size + (required_size >> 3));
04768     // It's a performance benefit to keep a frequently used array in new-space.
04769   } else if (!GetHeap()->new_space()->Contains(elts) &&
04770              required_size < kArraySizeThatFitsComfortablyInNewSpace) {
04771     // Expand will allocate a new backing store in new space even if the size
04772     // we asked for isn't larger than what we had before.
04773     Expand(required_size);
04774   }
04775 }
