00086 // Calculated the stack limit beyond which we will throw stack overflow errors.
00087 // This macro must be called from a C++ method. It relies on being able to take
00088 // the address of "this" to get a value on the current execution stack and then
00089 // calculates the stack limit based on that value.
00090 // NOTE: The check for overflow is not safe as there is no guarantee that the
00091 // running thread has its stack in all memory up to address 0x00000000.
00092 #define GENERATED_CODE_STACK_LIMIT(limit) \
00093   (reinterpret_cast<uintptr_t>(this) >= limit ? \
00094       reinterpret_cast<uintptr_t>(this) - limit : 0)
