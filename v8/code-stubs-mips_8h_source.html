00498   static void PatchBranchIntoNop(MacroAssembler* masm, int pos) {
00499     const unsigned offset = masm->instr_at(pos) & kImm16Mask;
00500     masm->instr_at_put(pos, BNE | (zero_reg.code() << kRsShift) |
00501         (zero_reg.code() << kRtShift) | (offset & kImm16Mask));
00502     ASSERT(Assembler::IsBne(masm->instr_at(pos)));
00503   }

00505   static void PatchNopIntoBranch(MacroAssembler* masm, int pos) {
00506     const unsigned offset = masm->instr_at(pos) & kImm16Mask;
00507     masm->instr_at_put(pos, BEQ | (zero_reg.code() << kRsShift) |
00508         (zero_reg.code() << kRtShift) | (offset & kImm16Mask));
00509     ASSERT(Assembler::IsBeq(masm->instr_at(pos)));
00510   }

00512   static Mode GetMode(Code* stub) {
00513     Instr first_instruction = Assembler::instr_at(stub->instruction_start());
00514     Instr second_instruction = Assembler::instr_at(stub->instruction_start() +
00515                                                    2 * Assembler::kInstrSize);
00516 
00517     if (Assembler::IsBeq(first_instruction)) {
00518       return INCREMENTAL;
00519     }
00520 
00521     ASSERT(Assembler::IsBne(first_instruction));
00522 
00523     if (Assembler::IsBeq(second_instruction)) {
00524       return INCREMENTAL_COMPACTION;
00525     }
00526 
00527     ASSERT(Assembler::IsBne(second_instruction));
00528 
00529     return STORE_BUFFER_ONLY;
00530   }

00532   static void Patch(Code* stub, Mode mode) {
00533     MacroAssembler masm(NULL,
00534                         stub->instruction_start(),
00535                         stub->instruction_size());
00536     switch (mode) {
00537       case STORE_BUFFER_ONLY:
00538         ASSERT(GetMode(stub) == INCREMENTAL ||
00539                GetMode(stub) == INCREMENTAL_COMPACTION);
00540         PatchBranchIntoNop(&masm, 0);
00541         PatchBranchIntoNop(&masm, 2 * Assembler::kInstrSize);
00542         break;
00543       case INCREMENTAL:
00544         ASSERT(GetMode(stub) == STORE_BUFFER_ONLY);
00545         PatchNopIntoBranch(&masm, 0);
00546         break;
00547       case INCREMENTAL_COMPACTION:
00548         ASSERT(GetMode(stub) == STORE_BUFFER_ONLY);
00549         PatchNopIntoBranch(&masm, 2 * Assembler::kInstrSize);
00550         break;
00551     }
00552     ASSERT(GetMode(stub) == mode);
00553     CPU::FlushICache(stub->instruction_start(), 4 * Assembler::kInstrSize);
00554   }

00730   // is floating point registers FPU must be supported. If core registers are
00731   // requested when FPU is supported f12 and f14 will still be scratched. If
00732   // either a0 or a1 is not a number (not smi and not heap number object) the
