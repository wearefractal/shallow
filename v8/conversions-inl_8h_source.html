00054 // The fast double-to-unsigned-int conversion routine does not guarantee
00055 // rounding towards zero, or any reasonable value if the argument is larger
00056 // than what fits in an unsigned 32-bit integer.
00057 inline unsigned int FastD2UI(double x) {
00058   // There is no unsigned version of lrint, so there is no fast path
00059   // in this function as there is in FastD2I. Using lrint doesn't work
00060   // for values of 2^31 and above.
00061 
00062   // Convert "small enough" doubles to uint32_t by fixing the 32
00063   // least significant non-fractional bits in the low 32 bits of the
00064   // double, and reading them from there.
00065   const double k2Pow52 = 4503599627370496.0;
00066   bool negative = x < 0;
00067   if (negative) {
00068     x = -x;
00069   }
00070   if (x < k2Pow52) {
00071     x += k2Pow52;
00072     uint32_t result;
00073     Address mantissa_ptr = reinterpret_cast<Address>(&x);
00074     // Copy least significant 32 bits of mantissa.
00075     memcpy(&result, mantissa_ptr, sizeof(result));
00076     return negative ? ~result + 1 : result;
00077   }
00078   // Large number (outside uint32 range), Infinity or NaN.
00079   return 0x80000000u;  // Return integer indefinite.
00080 }

00090 int32_t DoubleToInt32(double x) {
00091   int32_t i = FastD2I(x);
00092   if (FastI2D(i) == x) return i;
00093   Double d(x);
00094   int exponent = d.Exponent();
00095   if (exponent < 0) {
00096     if (exponent <= -Double::kSignificandSize) return 0;
00097     return d.Sign() * static_cast<int32_t>(d.Significand() >> -exponent);
00098   } else {
00099     if (exponent > 31) return 0;
00100     return d.Sign() * static_cast<int32_t>(d.Significand() << exponent);
00101   }
00102 }

00134 template <int radix_log_2, class Iterator, class EndMark>
00135 double InternalStringToIntDouble(UnicodeCache* unicode_cache,
00136                                  Iterator current,
00137                                  EndMark end,
00138                                  bool negative,
00139                                  bool allow_trailing_junk) {
00140   ASSERT(current != end);
00141 
00142   // Skip leading 0s.
00143   while (*current == '0') {
00144     ++current;
00145     if (current == end) return SignedZero(negative);
00146   }
00147 
00148   int64_t number = 0;
00149   int exponent = 0;
00150   const int radix = (1 << radix_log_2);
00151 
00152   do {
00153     int digit;
00154     if (*current >= '0' && *current <= '9' && *current < '0' + radix) {
00155       digit = static_cast<char>(*current) - '0';
00156     } else if (radix > 10 && *current >= 'a' && *current < 'a' + radix - 10) {
00157       digit = static_cast<char>(*current) - 'a' + 10;
00158     } else if (radix > 10 && *current >= 'A' && *current < 'A' + radix - 10) {
00159       digit = static_cast<char>(*current) - 'A' + 10;
00160     } else {
00161       if (allow_trailing_junk ||
00162           !AdvanceToNonspace(unicode_cache, &current, end)) {
00163         break;
00164       } else {
00165         return JunkStringValue();
00166       }
00167     }
00168 
00169     number = number * radix + digit;
00170     int overflow = static_cast<int>(number >> 53);
00171     if (overflow != 0) {
00172       // Overflow occurred. Need to determine which direction to round the
00173       // result.
00174       int overflow_bits_count = 1;
00175       while (overflow > 1) {
00176         overflow_bits_count++;
00177         overflow >>= 1;
00178       }
00179 
00180       int dropped_bits_mask = ((1 << overflow_bits_count) - 1);
00181       int dropped_bits = static_cast<int>(number) & dropped_bits_mask;
00182       number >>= overflow_bits_count;
00183       exponent = overflow_bits_count;
00184 
00185       bool zero_tail = true;
00186       while (true) {
00187         ++current;
00188         if (current == end || !isDigit(*current, radix)) break;
00189         zero_tail = zero_tail && *current == '0';
00190         exponent += radix_log_2;
00191       }
00192 
00193       if (!allow_trailing_junk &&
00194           AdvanceToNonspace(unicode_cache, &current, end)) {
00195         return JunkStringValue();
00196       }
00197 
00198       int middle_value = (1 << (overflow_bits_count - 1));
00199       if (dropped_bits > middle_value) {
00200         number++;  // Rounding up.
00201       } else if (dropped_bits == middle_value) {
00202         // Rounding to even to consistency with decimals: half-way case rounds
00203         // up if significant part is odd and down otherwise.
00204         if ((number & 1) != 0 || !zero_tail) {
00205           number++;  // Rounding up.
00206         }
00207       }
00208 
00209       // Rounding up may cause overflow.
00210       if ((number & ((int64_t)1 << 53)) != 0) {
00211         exponent++;
00212         number >>= 1;
00213       }
00214       break;
00215     }
00216     ++current;
00217   } while (current != end);
00218 
00219   ASSERT(number < ((int64_t)1 << 53));
00220   ASSERT(static_cast<int64_t>(static_cast<double>(number)) == number);
00221 
00222   if (exponent == 0) {
00223     if (negative) {
00224       if (number == 0) return -0.0;
00225       number = -number;
00226     }
00227     return static_cast<double>(number);
00228   }
00229 
00230   ASSERT(number != 0);
00231   return ldexp(static_cast<double>(negative ? -number : number), exponent);
00232 }

00340       if (buffer_pos <= kMaxSignificantDigits) {
00341         // If the number has more than kMaxSignificantDigits it will be parsed

00361   // The following code causes accumulating rounding error for numbers greater
00362   // than ~2^56. It's explicitly allowed in the spec: "if R is not 2, 4, 8, 10,
00363   // 16, or 32, then mathInt may be an implementation-dependent approximation to
00364   // the mathematical integer value" (15.1.2.2).

00366   int lim_0 = '0' + (radix < 10 ? radix : 10);
00367   int lim_a = 'a' + (radix - 10);
00368   int lim_A = 'A' + (radix - 10);
00369

00371   // first glance. It is structured to use 32-bit multiply-and-add
00372   // loops as long as possible to avoid loosing precision.
00373 
00374   double v = 0.0;
00375   bool done = false;
00376   do {
00377     // Parse the longest part of the string starting at index j
00378     // possible while keeping the multiplier, and thus the part
00379     // itself, within 32 bits.
00380     unsigned int part = 0, multiplier = 1;
00381     while (true) {
00382       int d;
00383       if (*current >= '0' && *current < lim_0) {
00384         d = *current - '0';
00385       } else if (*current >= 'a' && *current < lim_a) {
00386         d = *current - 'a' + 10;
00387       } else if (*current >= 'A' && *current < lim_A) {
00388         d = *current - 'A' + 10;
00389       } else {
00390         done = true;
00391         break;
00392       }
00393

00399       uint32_t m = multiplier * radix;
00400       if (m > kMaximumMultiplier) break;

00402       multiplier = m;
00403       ASSERT(multiplier > part);
00404 
00405       ++current;
00406       if (current == end) {
00407         done = true;
00408         break;
00409       }
00410     }
00411 
00412     // Update the value and skip the part in the string.
00413     v = v * multiplier + part;
00414   } while (!done);
00415 
00416   if (!allow_trailing_junk &&
00417       AdvanceToNonspace(unicode_cache, &current, end)) {
00418     return JunkStringValue();
00419   }
00420 
00421   return negative ? -v : v;
00422 }

00426 // the following operations:
00427 // 1. current == end (other ops are not allowed), current != end.
00428 // 2. *current - gets the current character in the sequence.
00429 // 3. ++current (advances the position).

00433                               EndMark end,

00437   // convention is used:

00439   // 2. If AdvanceToNonspace returned false then current == end.

00602       ++current;
00603       if (current == end) {
00604         if (allow_trailing_junk) {
00605           goto parsing_done;
00606         } else {
00607           return JunkStringValue();
00608         }
00609       }
00610     }
