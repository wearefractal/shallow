00036 template <typename Char>
00037 bool DateParser::Parse(Vector<Char> str,
00038                        FixedArray* out,
00039                        UnicodeCache* unicode_cache) {
00040   ASSERT(out->length() >= OUTPUT_SIZE);
00041   InputReader<Char> in(unicode_cache, str);
00042   DateStringTokenizer<Char> scanner(&in);
00043   TimeZoneComposer tz;
00044   TimeComposer time;
00045   DayComposer day;
00046 
00047   // Specification:
00048   // Accept ES5 ISO 8601 date-time-strings or legacy dates compatible
00049   // with Safari.
00050   // ES5 ISO 8601 dates:
00051   //   [('-'|'+')yy]yyyy[-MM[-DD]][THH:mm[:ss[.sss]][Z|(+|-)hh:mm]]
00052   //   where yyyy is in the range 0000..9999 and
00053   //         +/-yyyyyy is in the range -999999..+999999 -
00054   //           but -000000 is invalid (year zero must be positive),
00055   //         MM is in the range 01..12,
00056   //         DD is in the range 01..31,
00057   //         MM and DD defaults to 01 if missing,,
00058   //         HH is generally in the range 00..23, but can be 24 if mm, ss
00059   //           and sss are zero (or missing), representing midnight at the
00060   //           end of a day,
00061   //         mm and ss are in the range 00..59,
00062   //         sss is in the range 000..999,
00063   //         hh is in the range 00..23,
00064   //         mm, ss, and sss default to 00 if missing, and
00065   //         timezone defaults to Z if missing.
00066   //  Extensions:
00067   //   We also allow sss to have more or less than three digits (but at
00068   //   least one).
00069   //   We allow hh:mm to be specified as hhmm.
00070   // Legacy dates:
00071   //  Any unrecognized word before the first number is ignored.
00072   //  Parenthesized text is ignored.
00073   //  An unsigned number followed by ':' is a time value, and is
00074   //  added to the TimeComposer. A number followed by '::' adds a second
00075   //  zero as well. A number followed by '.' is also a time and must be
00076   //  followed by milliseconds.
00077   //  Any other number is a date component and is added to DayComposer.
00078   //  A month name (or really: any word having the same first three letters
00079   //  as a month name) is recorded as a named month in the Day composer.
00080   //  A word recognizable as a time-zone is recorded as such, as is
00081   //  '(+|-)(hhmm|hh:)'.
00082   //  Legacy dates don't allow extra signs ('+' or '-') or umatched ')'
00083   //  after a number has been read (before the first number, any garbage
00084   //  is allowed).
00085   // Intersection of the two:
00086   //  A string that matches both formats (e.g. 1970-01-01) will be
00087   //  parsed as an ES5 date-time string - which means it will default
00088   //  to UTC time-zone. That's unavoidable if following the ES5
00089   //  specification.
00090   //  After a valid "T" has been read while scanning an ES5 datetime string,
00091   //  the input can no longer be a valid legacy date, since the "T" is a
00092   //  garbage string after a number has been read.
00093 
00094   // First try getting as far as possible with as ES5 Date Time String.
00095   DateToken next_unhandled_token = ParseES5DateTime(&scanner, &day, &time, &tz);
00096   if (next_unhandled_token.IsInvalid()) return false;
00097   bool has_read_number = !day.IsEmpty();
00098   // If there's anything left, continue with the legacy parser.
00099   for (DateToken token = next_unhandled_token;
00100        !token.IsEndOfInput();
00101        token = scanner.Next()) {
00102     if (token.IsNumber()) {
00103       has_read_number = true;
00104       int n = token.number();
00105       if (scanner.SkipSymbol(':')) {
00106         if (scanner.SkipSymbol(':')) {
00107           // n + "::"
00108           if (!time.IsEmpty()) return false;
00109           time.Add(n);
00110           time.Add(0);
00111         } else {
00112           // n + ":"
00113           if (!time.Add(n)) return false;
00114           if (scanner.Peek().IsSymbol('.')) scanner.Next();
00115         }
00116       } else if (scanner.SkipSymbol('.') && time.IsExpecting(n)) {
00117         time.Add(n);
00118         if (!scanner.Peek().IsNumber()) return false;
00119         int n = ReadMilliseconds(scanner.Next());
00120         if (n < 0) return false;
00121         time.AddFinal(n);
00122       } else if (tz.IsExpecting(n)) {
00123         tz.SetAbsoluteMinute(n);
00124       } else if (time.IsExpecting(n)) {
00125         time.AddFinal(n);
00126         // Require end, white space, "Z", "+" or "-" immediately after
00127         // finalizing time.
00128         DateToken peek = scanner.Peek();
00129         if (!peek.IsEndOfInput() &&
00130             !peek.IsWhiteSpace() &&
00131             !peek.IsKeywordZ() &&
00132             !peek.IsAsciiSign()) return false;
00133       } else {
00134         if (!day.Add(n)) return false;
00135         scanner.SkipSymbol('-');
00136       }
00137     } else if (token.IsKeyword()) {
00138       // Parse a "word" (sequence of chars. >= 'A').
00139       KeywordType type = token.keyword_type();
00140       int value = token.keyword_value();
00141       if (type == AM_PM && !time.IsEmpty()) {
00142         time.SetHourOffset(value);
00143       } else if (type == MONTH_NAME) {
00144         day.SetNamedMonth(value);
00145         scanner.SkipSymbol('-');
00146       } else if (type == TIME_ZONE_NAME && has_read_number) {
00147         tz.Set(value);
00148       } else {
00149         // Garbage words are illegal if a number has been read.
00150         if (has_read_number) return false;
00151       }
00152     } else if (token.IsAsciiSign() && (tz.IsUTC() || !time.IsEmpty())) {
00153       // Parse UTC offset (only after UTC or time).
00154       tz.SetSign(token.ascii_sign());
00155       // The following number may be empty.
00156       int n = 0;
00157       if (scanner.Peek().IsNumber()) {
00158         n = scanner.Next().number();
00159       }
00160       has_read_number = true;
00161 
00162       if (scanner.Peek().IsSymbol(':')) {
00163         tz.SetAbsoluteHour(n);
00164         tz.SetAbsoluteMinute(kNone);
00165       } else {
00166         tz.SetAbsoluteHour(n / 100);
00167         tz.SetAbsoluteMinute(n % 100);
00168       }
00169     } else if ((token.IsAsciiSign() || token.IsSymbol(')')) &&
00170                has_read_number) {
00171       // Extra sign or ')' is illegal if a number has been read.
00172       return false;
00173     } else {
00174       // Ignore other characters and whitespace.
00175     }
00176   }
00177 
00178   return day.Write(out) && time.Write(out) && tz.Write(out);
00179 }

00182 template<typename CharType>
00183 DateParser::DateToken DateParser::DateStringTokenizer<CharType>::Scan() {
00184   int pre_pos = in_->position();
00185   if (in_->IsEnd()) return DateToken::EndOfInput();
00186   if (in_->IsAsciiDigit()) {
00187     int n = in_->ReadUnsignedNumeral();
00188     int length = in_->position() - pre_pos;
00189     return DateToken::Number(n, length);
00190   }
00191   if (in_->Skip(':')) return DateToken::Symbol(':');
00192   if (in_->Skip('-')) return DateToken::Symbol('-');
00193   if (in_->Skip('+')) return DateToken::Symbol('+');
00194   if (in_->Skip('.')) return DateToken::Symbol('.');
00195   if (in_->Skip(')')) return DateToken::Symbol(')');
00196   if (in_->IsAsciiAlphaOrAbove()) {
00197     ASSERT(KeywordTable::kPrefixLength == 3);
00198     uint32_t buffer[3] = {0, 0, 0};
00199     int length = in_->ReadWord(buffer, 3);
00200     int index = KeywordTable::Lookup(buffer, length);
00201     return DateToken::Keyword(KeywordTable::GetType(index),
00202                               KeywordTable::GetValue(index),
00203                               length);
00204   }
00205   if (in_->SkipWhiteSpace()) {
00206     return DateToken::WhiteSpace(in_->position() - pre_pos);
00207   }
00208   if (in_->SkipParentheses()) {
00209     return DateToken::Unknown();
00210   }
00211   in_->Next();
00212   return DateToken::Unknown();
00213 }

00216 template <typename Char>
00217 DateParser::DateToken DateParser::ParseES5DateTime(
00218     DateStringTokenizer<Char>* scanner,
00219     DayComposer* day,
00220     TimeComposer* time,
00221     TimeZoneComposer* tz) {
00222   ASSERT(day->IsEmpty());
00223   ASSERT(time->IsEmpty());
00224   ASSERT(tz->IsEmpty());
00225 
00226   // Parse mandatory date string: [('-'|'+')yy]yyyy[':'MM[':'DD]]
00227   if (scanner->Peek().IsAsciiSign()) {
00228     // Keep the sign token, so we can pass it back to the legacy
00229     // parser if we don't use it.
00230     DateToken sign_token = scanner->Next();
00231     if (!scanner->Peek().IsFixedLengthNumber(6)) return sign_token;
00232     int sign = sign_token.ascii_sign();
00233     int year = scanner->Next().number();
00234     if (sign < 0 && year == 0) return sign_token;
00235     day->Add(sign * year);
00236   } else if (scanner->Peek().IsFixedLengthNumber(4)) {
00237     day->Add(scanner->Next().number());
00238   } else {
00239     return scanner->Next();
00240   }
00241   if (scanner->SkipSymbol('-')) {
00242     if (!scanner->Peek().IsFixedLengthNumber(2) ||
00243         !DayComposer::IsMonth(scanner->Peek().number())) return scanner->Next();
00244     day->Add(scanner->Next().number());
00245     if (scanner->SkipSymbol('-')) {
00246       if (!scanner->Peek().IsFixedLengthNumber(2) ||
00247           !DayComposer::IsDay(scanner->Peek().number())) return scanner->Next();
00248       day->Add(scanner->Next().number());
00249     }
00250   }
00251   // Check for optional time string: 'T'HH':'mm[':'ss['.'sss]]Z
00252   if (!scanner->Peek().IsKeywordType(TIME_SEPARATOR)) {
00253     if (!scanner->Peek().IsEndOfInput()) return scanner->Next();
00254   } else {
00255     // ES5 Date Time String time part is present.
00256     scanner->Next();
00257     if (!scanner->Peek().IsFixedLengthNumber(2) ||
00258         !Between(scanner->Peek().number(), 0, 24)) {
00259       return DateToken::Invalid();
00260     }
00261     // Allow 24:00[:00[.000]], but no other time starting with 24.
00262     bool hour_is_24 = (scanner->Peek().number() == 24);
00263     time->Add(scanner->Next().number());
00264     if (!scanner->SkipSymbol(':')) return DateToken::Invalid();
00265     if (!scanner->Peek().IsFixedLengthNumber(2) ||
00266         !TimeComposer::IsMinute(scanner->Peek().number()) ||
00267         (hour_is_24 && scanner->Peek().number() > 0)) {
00268       return DateToken::Invalid();
00269     }
00270     time->Add(scanner->Next().number());
00271     if (scanner->SkipSymbol(':')) {
00272       if (!scanner->Peek().IsFixedLengthNumber(2) ||
00273           !TimeComposer::IsSecond(scanner->Peek().number()) ||
00274           (hour_is_24 && scanner->Peek().number() > 0)) {
00275         return DateToken::Invalid();
00276       }
00277       time->Add(scanner->Next().number());
00278       if (scanner->SkipSymbol('.')) {
00279         if (!scanner->Peek().IsNumber() ||
00280             (hour_is_24 && scanner->Peek().number() > 0)) {
00281           return DateToken::Invalid();
00282         }
00283         // Allow more or less than the mandated three digits.
00284         time->Add(ReadMilliseconds(scanner->Next()));
00285       }
00286     }
00287     // Check for optional timezone designation: 'Z' | ('+'|'-')hh':'mm
00288     if (scanner->Peek().IsKeywordZ()) {
00289       scanner->Next();
00290       tz->Set(0);
00291     } else if (scanner->Peek().IsSymbol('+') ||
00292                scanner->Peek().IsSymbol('-')) {
00293       tz->SetSign(scanner->Next().symbol() == '+' ? 1 : -1);
00294       if (scanner->Peek().IsFixedLengthNumber(4)) {
00295         // hhmm extension syntax.
00296         int hourmin = scanner->Next().number();
00297         int hour = hourmin / 100;
00298         int min = hourmin % 100;
00299         if (!TimeComposer::IsHour(hour) || !TimeComposer::IsMinute(min)) {
00300           return DateToken::Invalid();
00301         }
00302         tz->SetAbsoluteHour(hour);
00303         tz->SetAbsoluteMinute(min);
00304       } else {
00305         // hh:mm standard syntax.
00306         if (!scanner->Peek().IsFixedLengthNumber(2) ||
00307             !TimeComposer::IsHour(scanner->Peek().number())) {
00308           return DateToken::Invalid();
00309         }
00310         tz->SetAbsoluteHour(scanner->Next().number());
00311         if (!scanner->SkipSymbol(':')) return DateToken::Invalid();
00312         if (!scanner->Peek().IsFixedLengthNumber(2) ||
00313             !TimeComposer::IsMinute(scanner->Peek().number())) {
00314           return DateToken::Invalid();
00315         }
00316         tz->SetAbsoluteMinute(scanner->Next().number());
00317       }
00318     }
00319     if (!scanner->Peek().IsEndOfInput()) return DateToken::Invalid();
00320   }
00321   // Successfully parsed ES5 Date Time String. Default to UTC if no TZ given.
00322   if (tz->IsEmpty()) tz->Set(0);
00323   day->set_iso_date();
00324   return DateToken::EndOfInput();
00325 }
