00039 // Class holding constants and methods that apply to all string search variants,
00040 // independently of subject and pattern char size.
00041 class StringSearchBase {
00042  protected:
00043   // Cap on the maximal shift in the Boyer-Moore implementation. By setting a
00044   // limit, we can fix the size of tables. For a needle longer than this limit,
00045   // search will not be optimal, since we only build tables for a suffix
00046   // of the string, but it is a safe approximation.
00047   static const int kBMMaxShift = Isolate::kBMMaxShift;
00048 
00049   // Reduce alphabet to this size.
00050   // One of the tables used by Boyer-Moore and Boyer-Moore-Horspool has size
00051   // proportional to the input alphabet. We reduce the alphabet size by
00052   // equating input characters modulo a smaller alphabet size. This gives
00053   // a potentially less efficient searching, but is a safe approximation.
00054   // For needles using only characters in the same Unicode 256-code point page,
00055   // there is no search speed degradation.
00056   static const int kAsciiAlphabetSize = 128;
00057   static const int kUC16AlphabetSize = Isolate::kUC16AlphabetSize;
00058 
00059   // Bad-char shift table stored in the state. It's length is the alphabet size.
00060   // For patterns below this length, the skip length of Boyer-Moore is too short
00061   // to compensate for the algorithmic overhead compared to simple brute force.
00062   static const int kBMMinPatternLength = 7;
00063 
00064   static inline bool IsAsciiString(Vector<const char>) {
00065     return true;
00066   }
00067 
00068   static inline bool IsAsciiString(Vector<const uc16> string) {
00069     return String::IsAscii(string.start(), string.length());
00070   }
00071 
00072   friend class Isolate;
00073 };

00209 template <typename PatternChar, typename SubjectChar>
00210 int StringSearch<PatternChar, SubjectChar>::SingleCharSearch(
00211     StringSearch<PatternChar, SubjectChar>* search,
00212     Vector<const SubjectChar> subject,
00213     int index) {
00214   ASSERT_EQ(1, search->pattern_.length());
00215   PatternChar pattern_first_char = search->pattern_[0];
00216   int i = index;
00217   if (sizeof(SubjectChar) == 1 && sizeof(PatternChar) == 1) {
00218     const SubjectChar* pos = reinterpret_cast<const SubjectChar*>(
00219         memchr(subject.start() + i,
00220                pattern_first_char,
00221                subject.length() - i));
00222     if (pos == NULL) return -1;
00223     return static_cast<int>(pos - subject.start());
00224   } else {
00225     if (sizeof(PatternChar) > sizeof(SubjectChar)) {
00226       if (static_cast<uc16>(pattern_first_char) > String::kMaxAsciiCharCodeU) {
00227         return -1;
00228       }
00229     }
00230     SubjectChar search_char = static_cast<SubjectChar>(pattern_first_char);
00231     int n = subject.length();
00232     while (i < n) {
00233       if (subject[i++] == search_char) return i - 1;
00234     }
00235     return -1;
00236   }
00237 }

00260 // Simple linear search for short patterns. Never bails out.
00261 template <typename PatternChar, typename SubjectChar>
00262 int StringSearch<PatternChar, SubjectChar>::LinearSearch(
00263     StringSearch<PatternChar, SubjectChar>* search,
00264     Vector<const SubjectChar> subject,
00265     int index) {
00266   Vector<const PatternChar> pattern = search->pattern_;
00267   ASSERT(pattern.length() > 1);
00268   int pattern_length = pattern.length();
00269   PatternChar pattern_first_char = pattern[0];
00270   int i = index;
00271   int n = subject.length() - pattern_length;
00272   while (i <= n) {
00273     if (sizeof(SubjectChar) == 1 && sizeof(PatternChar) == 1) {
00274       const SubjectChar* pos = reinterpret_cast<const SubjectChar*>(
00275           memchr(subject.start() + i,
00276                  pattern_first_char,
00277                  n - i + 1));
00278       if (pos == NULL) return -1;
00279       i = static_cast<int>(pos - subject.start()) + 1;
00280     } else {
00281       if (subject[i++] != pattern_first_char) continue;
00282     }
00283     // Loop extracted to separate function to allow using return to do
00284     // a deeper break.
00285     if (CharCompare(pattern.start() + 1,
00286                     subject.start() + i,
00287                     pattern_length - 1)) {
00288       return i - 1;
00289     }
00290   }
00291   return -1;
00292 }

00298 template <typename PatternChar, typename SubjectChar>
00299 int StringSearch<PatternChar, SubjectChar>::BoyerMooreSearch(
00300     StringSearch<PatternChar, SubjectChar>* search,
00301     Vector<const SubjectChar> subject,
00302     int start_index) {
00303   Vector<const PatternChar> pattern = search->pattern_;
00304   int subject_length = subject.length();
00305   int pattern_length = pattern.length();
00306   // Only preprocess at most kBMMaxShift last characters of pattern.
00307   int start = search->start_;
00308 
00309   int* bad_char_occurence = search->bad_char_table();
00310   int* good_suffix_shift = search->good_suffix_shift_table();
00311 
00312   PatternChar last_char = pattern[pattern_length - 1];
00313   int index = start_index;
00314   // Continue search from i.
00315   while (index <= subject_length - pattern_length) {
00316     int j = pattern_length - 1;
00317     int c;
00318     while (last_char != (c = subject[index + j])) {
00319       int shift =
00320           j - CharOccurrence(bad_char_occurence, c);
00321       index += shift;
00322       if (index > subject_length - pattern_length) {
00323         return -1;
00324       }
00325     }
00326     while (j >= 0 && pattern[j] == (c = subject[index + j])) j--;
00327     if (j < 0) {
00328       return index;
00329     } else if (j < start) {
00330       // we have matched more than our tables allow us to be smart about.
00331       // Fall back on BMH shift.
00332       index += pattern_length - 1
00333           - CharOccurrence(bad_char_occurence,
00334                            static_cast<SubjectChar>(last_char));
00335     } else {
00336       int gs_shift = good_suffix_shift[j + 1];
00337       int bc_occ =
00338           CharOccurrence(bad_char_occurence, c);
00339       int shift = j - bc_occ;
00340       if (gs_shift > shift) {
00341         shift = gs_shift;
00342       }
00343       index += shift;
00344     }
00345   }
00346 
00347   return -1;
00348 }

00351 template <typename PatternChar, typename SubjectChar>
00352 void StringSearch<PatternChar, SubjectChar>::PopulateBoyerMooreTable() {
00353   int pattern_length = pattern_.length();
00354   const PatternChar* pattern = pattern_.start();
00355   // Only look at the last kBMMaxShift characters of pattern (from start_
00356   // to pattern_length).
00357   int start = start_;
00358   int length = pattern_length - start;
00359 
00360   // Biased tables so that we can use pattern indices as table indices,
00361   // even if we only cover the part of the pattern from offset start.
00362   int* shift_table = good_suffix_shift_table();
00363   int* suffix_table = this->suffix_table();
00364 
00365   // Initialize table.
00366   for (int i = start; i < pattern_length; i++) {
00367     shift_table[i] = length;
00368   }
00369   shift_table[pattern_length] = 1;
00370   suffix_table[pattern_length] = pattern_length + 1;
00371 
00372   if (pattern_length <= start) {
00373     return;
00374   }
00375 
00376   // Find suffixes.
00377   PatternChar last_char = pattern[pattern_length - 1];
00378   int suffix = pattern_length + 1;
00379   {
00380     int i = pattern_length;
00381     while (i > start) {
00382       PatternChar c = pattern[i - 1];
00383       while (suffix <= pattern_length && c != pattern[suffix - 1]) {
00384         if (shift_table[suffix] == length) {
00385           shift_table[suffix] = suffix - i;
00386         }
00387         suffix = suffix_table[suffix];
00388       }
00389       suffix_table[--i] = --suffix;
00390       if (suffix == pattern_length) {
00391         // No suffix to extend, so we check against last_char only.
00392         while ((i > start) && (pattern[i - 1] != last_char)) {
00393           if (shift_table[pattern_length] == length) {
00394             shift_table[pattern_length] = pattern_length - i;
00395           }
00396           suffix_table[--i] = pattern_length;
00397         }
00398         if (i > start) {
00399           suffix_table[--i] = --suffix;
00400         }
00401       }
00402     }
00403   }
00404   // Build shift table using suffixes.
00405   if (suffix < pattern_length) {
00406     for (int i = start; i <= pattern_length; i++) {
00407       if (shift_table[i] == length) {
00408         shift_table[i] = suffix - start;
00409       }
00410       if (i == suffix) {
00411         suffix = suffix_table[suffix];
00412       }
00413     }
00414   }
00415 }

00421 template <typename PatternChar, typename SubjectChar>
00422 int StringSearch<PatternChar, SubjectChar>::BoyerMooreHorspoolSearch(
00423     StringSearch<PatternChar, SubjectChar>* search,
00424     Vector<const SubjectChar> subject,
00425     int start_index) {
00426   Vector<const PatternChar> pattern = search->pattern_;
00427   int subject_length = subject.length();
00428   int pattern_length = pattern.length();
00429   int* char_occurrences = search->bad_char_table();
00430   int badness = -pattern_length;
00431 
00432   // How bad we are doing without a good-suffix table.
00433   PatternChar last_char = pattern[pattern_length - 1];
00434   int last_char_shift = pattern_length - 1 -
00435       CharOccurrence(char_occurrences, static_cast<SubjectChar>(last_char));
00436   // Perform search
00437   int index = start_index;  // No matches found prior to this index.
00438   while (index <= subject_length - pattern_length) {
00439     int j = pattern_length - 1;
00440     int subject_char;
00441     while (last_char != (subject_char = subject[index + j])) {
00442       int bc_occ = CharOccurrence(char_occurrences, subject_char);
00443       int shift = j - bc_occ;
00444       index += shift;
00445       badness += 1 - shift;  // at most zero, so badness cannot increase.
00446       if (index > subject_length - pattern_length) {
00447         return -1;
00448       }
00449     }
00450     j--;
00451     while (j >= 0 && pattern[j] == (subject[index + j])) j--;
00452     if (j < 0) {
00453       return index;
00454     } else {
00455       index += last_char_shift;
00456       // Badness increases by the number of characters we have
00457       // checked, and decreases by the number of characters we
00458       // can skip by shifting. It's a measure of how we are doing
00459       // compared to reading each character exactly once.
00460       badness += (pattern_length - j) - last_char_shift;
00461       if (badness > 0) {
00462         search->PopulateBoyerMooreTable();
00463         search->strategy_ = &BoyerMooreSearch;
00464         return BoyerMooreSearch(search, subject, index);
00465       }
00466     }
00467   }
00468   return -1;
00469 }

00472 template <typename PatternChar, typename SubjectChar>
00473 void StringSearch<PatternChar, SubjectChar>::PopulateBoyerMooreHorspoolTable() {
00474   int pattern_length = pattern_.length();
00475 
00476   int* bad_char_occurrence = bad_char_table();
00477 
00478   // Only preprocess at most kBMMaxShift last characters of pattern.
00479   int start = start_;
00480   // Run forwards to populate bad_char_table, so that *last* instance
00481   // of character equivalence class is the one registered.
00482   // Notice: Doesn't include the last character.
00483   int table_size = AlphabetSize();
00484   if (start == 0) {  // All patterns less than kBMMaxShift in length.
00485     memset(bad_char_occurrence,
00486            -1,
00487            table_size * sizeof(*bad_char_occurrence));
00488   } else {
00489     for (int i = 0; i < table_size; i++) {
00490       bad_char_occurrence[i] = start - 1;
00491     }
00492   }
00493   for (int i = start; i < pattern_length - 1; i++) {
00494     PatternChar c = pattern_[i];
00495     int bucket = (sizeof(PatternChar) == 1) ? c : c % AlphabetSize();
00496     bad_char_occurrence[bucket] = i;
00497   }
00498 }

00504 // Simple linear search for short patterns, which bails out if the string
00505 // isn't found very early in the subject. Upgrades to BoyerMooreHorspool.
00506 template <typename PatternChar, typename SubjectChar>
00507 int StringSearch<PatternChar, SubjectChar>::InitialSearch(
00508     StringSearch<PatternChar, SubjectChar>* search,
00509     Vector<const SubjectChar> subject,
00510     int index) {
00511   Vector<const PatternChar> pattern = search->pattern_;
00512   int pattern_length = pattern.length();
00513   // Badness is a count of how much work we have done.  When we have
00514   // done enough work we decide it's probably worth switching to a better
00515   // algorithm.
00516   int badness = -10 - (pattern_length << 2);
00517 
00518   // We know our pattern is at least 2 characters, we cache the first so
00519   // the common case of the first character not matching is faster.
00520   PatternChar pattern_first_char = pattern[0];
00521   for (int i = index, n = subject.length() - pattern_length; i <= n; i++) {
00522     badness++;
00523     if (badness <= 0) {
00524       if (sizeof(SubjectChar) == 1 && sizeof(PatternChar) == 1) {
00525         const SubjectChar* pos = reinterpret_cast<const SubjectChar*>(
00526             memchr(subject.start() + i,
00527                    pattern_first_char,
00528                    n - i + 1));
00529         if (pos == NULL) {
00530           return -1;
00531         }
00532         i = static_cast<int>(pos - subject.start());
00533       } else {
00534         if (subject[i] != pattern_first_char) continue;
00535       }
00536       int j = 1;
00537       do {
00538         if (pattern[j] != subject[i + j]) {
00539           break;
00540         }
00541         j++;
00542       } while (j < pattern_length);
00543       if (j == pattern_length) {
00544         return i;
00545       }
00546       badness += j;
00547     } else {
00548       search->PopulateBoyerMooreHorspoolTable();
00549       search->strategy_ = &BoyerMooreHorspoolSearch;
00550       return BoyerMooreHorspoolSearch(search, subject, i);
00551     }
00552   }
00553   return -1;
00554 }
