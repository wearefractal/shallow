00036 // FuncNameInferrer is a stateful class that is used to perform name
00037 // inference for anonymous functions during static analysis of source code.
00038 // Inference is performed in cases when an anonymous function is assigned
00039 // to a variable or a property (see test-func-name-inference.cc for examples.)
00040 //
00041 // The basic idea is that during parsing of LHSs of certain expressions
00042 // (assignments, declarations, object literals) we collect name strings,
00043 // and during parsing of the RHS, a function literal can be collected. After
00044 // parsing the RHS we can infer a name for function literals that do not have
00045 // a name.
00046 class FuncNameInferrer : public ZoneObject {
00047  public:
00048   explicit FuncNameInferrer(Isolate* isolate);
00049 
00050   // Returns whether we have entered name collection state.
00051   bool IsOpen() const { return !entries_stack_.is_empty(); }
00052 
00053   // Pushes an enclosing the name of enclosing function onto names stack.
00054   void PushEnclosingName(Handle<String> name);
00055 
00056   // Enters name collection state.
00057   void Enter() {
00058     entries_stack_.Add(names_stack_.length());
00059   }
00060 
00061   // Pushes an encountered name onto names stack when in collection state.
00062   void PushLiteralName(Handle<String> name);
00063 
00064   void PushVariableName(Handle<String> name);
00065 
00066   // Adds a function to infer name for.
00067   void AddFunction(FunctionLiteral* func_to_infer) {
00068     if (IsOpen()) {
00069       funcs_to_infer_.Add(func_to_infer);
00070     }
00071   }
00072 
00073   void RemoveLastFunction() {
00074     if (IsOpen() && !funcs_to_infer_.is_empty()) {
00075       funcs_to_infer_.RemoveLast();
00076     }
00077   }
00078 
00079   // Infers a function name and leaves names collection state.
00080   void Infer() {
00081     ASSERT(IsOpen());
00082     if (!funcs_to_infer_.is_empty()) {
00083       InferFunctionsNames();
00084     }
00085   }
00086 
00087   // Leaves names collection state.
00088   void Leave() {
00089     ASSERT(IsOpen());
00090     names_stack_.Rewind(entries_stack_.RemoveLast());
00091   }
00092 
00093  private:
00094   enum NameType {
00095     kEnclosingConstructorName,
00096     kLiteralName,
00097     kVariableName
00098   };
00099   struct Name {
00100     Name(Handle<String> name, NameType type) : name(name), type(type) { }
00101     Handle<String> name;
00102     NameType type;
00103   };
00104 
00105   Isolate* isolate() { return isolate_; }
00106 
00107   // Constructs a full name in dotted notation from gathered names.
00108   Handle<String> MakeNameFromStack();
00109 
00110   // A helper function for MakeNameFromStack.
00111   Handle<String> MakeNameFromStackHelper(int pos, Handle<String> prev);
00112 
00113   // Performs name inferring for added functions.
00114   void InferFunctionsNames();
00115 
00116   Isolate* isolate_;
00117   ZoneList<int> entries_stack_;
00118   ZoneList<Name> names_stack_;
00119   ZoneList<FunctionLiteral*> funcs_to_infer_;
00120 
00121   DISALLOW_COPY_AND_ASSIGN(FuncNameInferrer);
00122 };
