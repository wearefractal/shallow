00049 // The modes possibly affected by apply must be in kApplyMask.
00050 void RelocInfo::apply(intptr_t delta) {
00051   if (rmode_ == RUNTIME_ENTRY || IsCodeTarget(rmode_)) {
00052     int32_t* p = reinterpret_cast<int32_t*>(pc_);
00053     *p -= delta;  // Relocate entry.
00054     CPU::FlushICache(p, sizeof(uint32_t));
00055   } else if (rmode_ == JS_RETURN && IsPatchedReturnSequence()) {
00056     // Special handling of js_return when a break point is set (call
00057     // instruction has been inserted).
00058     int32_t* p = reinterpret_cast<int32_t*>(pc_ + 1);
00059     *p -= delta;  // Relocate entry.
00060     CPU::FlushICache(p, sizeof(uint32_t));
00061   } else if (rmode_ == DEBUG_BREAK_SLOT && IsPatchedDebugBreakSlotSequence()) {
00062     // Special handling of a debug break slot when a break point is set (call
00063     // instruction has been inserted).
00064     int32_t* p = reinterpret_cast<int32_t*>(pc_ + 1);
00065     *p -= delta;  // Relocate entry.
00066     CPU::FlushICache(p, sizeof(uint32_t));
00067   } else if (IsInternalReference(rmode_)) {
00068     // absolute code pointer inside code object moves with the code object.
00069     int32_t* p = reinterpret_cast<int32_t*>(pc_);
00070     *p += delta;  // Relocate entry.
00071     CPU::FlushICache(p, sizeof(uint32_t));
00072   }
00073 }

00221 void RelocInfo::Visit(ObjectVisitor* visitor) {
00222   RelocInfo::Mode mode = rmode();
00223   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00224     visitor->VisitEmbeddedPointer(this);
00225     CPU::FlushICache(pc_, sizeof(Address));
00226   } else if (RelocInfo::IsCodeTarget(mode)) {
00227     visitor->VisitCodeTarget(this);
00228   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00229     visitor->VisitGlobalPropertyCell(this);
00230   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00231     visitor->VisitExternalReference(this);
00232     CPU::FlushICache(pc_, sizeof(Address));
00233 #ifdef ENABLE_DEBUGGER_SUPPORT
00234   // TODO(isolates): Get a cached isolate below.
00235   } else if (((RelocInfo::IsJSReturn(mode) &&
00236               IsPatchedReturnSequence()) ||
00237              (RelocInfo::IsDebugBreakSlot(mode) &&
00238               IsPatchedDebugBreakSlotSequence())) &&
00239              Isolate::Current()->debug()->has_break_points()) {
00240     visitor->VisitDebugTarget(this);
00241 #endif
00242   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00243     visitor->VisitRuntimeEntry(this);
00244   }
00245 }

00248 template<typename StaticVisitor>
00249 void RelocInfo::Visit(Heap* heap) {
00250   RelocInfo::Mode mode = rmode();
00251   if (mode == RelocInfo::EMBEDDED_OBJECT) {
00252     StaticVisitor::VisitEmbeddedPointer(heap, this);
00253     CPU::FlushICache(pc_, sizeof(Address));
00254   } else if (RelocInfo::IsCodeTarget(mode)) {
00255     StaticVisitor::VisitCodeTarget(heap, this);
00256   } else if (mode == RelocInfo::GLOBAL_PROPERTY_CELL) {
00257     StaticVisitor::VisitGlobalPropertyCell(heap, this);
00258   } else if (mode == RelocInfo::EXTERNAL_REFERENCE) {
00259     StaticVisitor::VisitExternalReference(this);
00260     CPU::FlushICache(pc_, sizeof(Address));
00261 #ifdef ENABLE_DEBUGGER_SUPPORT
00262   } else if (heap->isolate()->debug()->has_break_points() &&
00263              ((RelocInfo::IsJSReturn(mode) &&
00264               IsPatchedReturnSequence()) ||
00265              (RelocInfo::IsDebugBreakSlot(mode) &&
00266               IsPatchedDebugBreakSlotSequence()))) {
00267     StaticVisitor::VisitDebugTarget(heap, this);
00268 #endif
00269   } else if (mode == RelocInfo::RUNTIME_ENTRY) {
00270     StaticVisitor::VisitRuntimeEntry(this);
00271   }
