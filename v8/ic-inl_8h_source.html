00041 Address IC::address() const {
00042   // Get the address of the call.
00043   Address result = pc() - Assembler::kCallTargetAddressOffset;
00044 
00045 #ifdef ENABLE_DEBUGGER_SUPPORT
00046   Debug* debug = Isolate::Current()->debug();
00047   // First check if any break points are active if not just return the address
00048   // of the call.
00049   if (!debug->has_break_points()) return result;
00050 
00051   // At least one break point is active perform additional test to ensure that
00052   // break point locations are updated correctly.
00053   if (debug->IsDebugBreak(Assembler::target_address_at(result))) {
00054     // If the call site is a call to debug break then return the address in
00055     // the original code instead of the address in the running code. This will
00056     // cause the original code to be updated and keeps the breakpoint active in
00057     // the running code.
00058     return OriginalCodeAddress();
00059   } else {
00060     // No break point here just return the address of the call.
00061     return result;
00062   }
00063 #else
00064   return result;
00065 #endif
00066 }

00069 Code* IC::GetTargetAtAddress(Address address) {
00070   // Get the target address of the IC.
00071   Address target = Assembler::target_address_at(address);
00072   // Convert target address to the code object. Code::GetCodeFromTargetAddress
00073   // is safe for use during GC where the map might be marked.
00074   Code* result = Code::GetCodeFromTargetAddress(target);
00075   ASSERT(result->is_inline_cache_stub());
00076   return result;
00077 }

00080 void IC::SetTargetAtAddress(Address address, Code* target) {
00081   ASSERT(target->is_inline_cache_stub() || target->is_compare_ic_stub());
00082   Code* old_target = GetTargetAtAddress(address);
00083 #ifdef DEBUG
00084   // STORE_IC and KEYED_STORE_IC use Code::extra_ic_state() to mark
00085   // ICs as strict mode. The strict-ness of the IC must be preserved.
00086   if (old_target->kind() == Code::STORE_IC ||
00087       old_target->kind() == Code::KEYED_STORE_IC) {
00088     ASSERT(Code::GetStrictMode(old_target->extra_ic_state()) ==
00089            Code::GetStrictMode(target->extra_ic_state()));
00090   }
00091 #endif
00092   Assembler::set_target_address_at(address, target->instruction_start());
00093   target->GetHeap()->incremental_marking()->RecordCodeTargetPatch(address,
00094                                                                   target);
00095   PostPatching(address, target, old_target);
00096 }

00099 InlineCacheHolderFlag IC::GetCodeCacheForObject(Object* object,
00100                                                 JSObject* holder) {
00101   if (object->IsJSObject()) {
00102     return GetCodeCacheForObject(JSObject::cast(object), holder);
00103   }
00104   // If the object is a value, we use the prototype map for the cache.
00105   ASSERT(object->IsString() || object->IsNumber() || object->IsBoolean());
00106   return PROTOTYPE_MAP;
00107 }

00110 InlineCacheHolderFlag IC::GetCodeCacheForObject(JSObject* object,
00111                                                 JSObject* holder) {
00112   // Fast-properties and global objects store stubs in their own maps.
00113   // Slow properties objects use prototype's map (unless the property is its own
00114   // when holder == object). It works because slow properties objects having
00115   // the same prototype (or a prototype with the same map) and not having
00116   // the property are interchangeable for such a stub.
00117   if (holder != object &&
00118       !object->HasFastProperties() &&
00119       !object->IsJSGlobalProxy() &&
00120       !object->IsJSGlobalObject()) {
00121     return PROTOTYPE_MAP;
00122   }
00123   return OWN_MAP;
00124 }
