00057 // The LiveObjectList is both a mechanism for tracking a live capture of
00058 // objects in the JS heap, as well as is the data structure which represents
00059 // each of those captures.  Unlike a snapshot, the lol is live.  For example,
00060 // if an object in a captured lol dies and is collected by the GC, the lol
00061 // will reflect that the object is no longer available.  The term
00062 // LiveObjectList (and lol) is used to describe both the mechanism and the
00063 // data structure depending on context of use.
00064 //
00065 // In captured lols, objects are tracked using their address and an object id.
00066 // The object id is unique.  Once assigned to an object, the object id can never
00067 // be assigned to another object.  That is unless all captured lols are deleted
00068 // which allows the user to start over with a fresh set of lols and object ids.
00069 // The uniqueness of the object ids allows the user to track specific objects
00070 // and inspect its longevity while debugging JS code in execution.
00071 //
00072 // The lol comes with utility functions to capture, dump, summarize, and diff
00073 // captured lols amongst other functionality.  These functionality are
00074 // accessible via the v8 debugger interface.

00223 // Helper class for updating the LiveObjectList HeapObject pointers.
00224 class UpdateLiveObjectListVisitor: public ObjectVisitor {
00225  public:
00226   void VisitPointer(Object** p) { UpdatePointer(p); }
00227 
00228   void VisitPointers(Object** start, Object** end) {
00229     // Copy all HeapObject pointers in [start, end).
00230     for (Object** p = start; p < end; p++) UpdatePointer(p);
00231   }
00232 
00233  private:
00234   // Based on Heap::ScavengeObject() but only does forwarding of pointers
00235   // to live new space objects, and not actually keep them alive.
00236   void UpdatePointer(Object** p) {
00237     Object* object = *p;
00238     if (!HEAP->InNewSpace(object)) return;
00239 
00240     HeapObject* heap_obj = HeapObject::cast(object);
00241     ASSERT(HEAP->InFromSpace(heap_obj));
00242 
00243     // We use the first word (where the map pointer usually is) of a heap
00244     // object to record the forwarding pointer.  A forwarding pointer can
00245     // point to an old space, the code space, or the to space of the new
00246     // generation.
00247     MapWord first_word = heap_obj->map_word();
00248 
00249     // If the first word is a forwarding address, the object has already been
00250     // copied.
00251     if (first_word.IsForwardingAddress()) {
00252       *p = first_word.ToForwardingAddress();
00253       return;
00254 
00255     // Else, it's a dead object.
00256     } else {
00257       LiveObjectList::NullifyNonLivePointer(reinterpret_cast<HeapObject**>(p));
00258     }
00259   }
00260 };
