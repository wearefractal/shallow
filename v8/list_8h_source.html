00037 // ----------------------------------------------------------------------------
00038 // The list is a template for very light-weight lists. We are not
00039 // using the STL because we want full control over space and speed of
00040 // the code. This implementation is based on code by Robert Griesemer
00041 // and Rob Pike.
00042 //
00043 // The list is parameterized by the type of its elements (T) and by an
00044 // allocation policy (P). The policy is used for allocating lists in
00045 // the C free store or the zone; see zone.h.
00046 
00047 // Forward defined as
00048 // template <typename T, class P = FreeStoreAllocationPolicy> class List;
00049 template <typename T, class P>
00050 class List {
00051  public:
00052   List() { Initialize(0); }
00053   INLINE(explicit List(int capacity)) { Initialize(capacity); }
00054   INLINE(~List()) { DeleteData(data_); }
00055 
00056   // Deallocates memory used by the list and leaves the list in a consistent
00057   // empty state.
00058   void Free() {
00059     DeleteData(data_);
00060     Initialize(0);
00061   }
00062 
00063   INLINE(void* operator new(size_t size)) {
00064       return P::New(static_cast<int>(size));
00065   }
00066   INLINE(void operator delete(void* p, size_t)) { return P::Delete(p); }
00067 
00068   // Returns a reference to the element at index i.  This reference is
00069   // not safe to use after operations that can change the list's
00070   // backing store (e.g. Add).
00071   inline T& operator[](int i) const {
00072     ASSERT(0 <= i);
00073     ASSERT(i < length_);
00074     return data_[i];
00075   }
00076   inline T& at(int i) const { return operator[](i); }
00077   inline T& last() const { return at(length_ - 1); }
00078   inline T& first() const { return at(0); }
00079 
00080   INLINE(bool is_empty() const) { return length_ == 0; }
00081   INLINE(int length() const) { return length_; }
00082   INLINE(int capacity() const) { return capacity_; }
00083 
00084   Vector<T> ToVector() const { return Vector<T>(data_, length_); }
00085 
00086   Vector<const T> ToConstVector() { return Vector<const T>(data_, length_); }
00087 
00088   // Adds a copy of the given 'element' to the end of the list,
00089   // expanding the list if necessary.
00090   void Add(const T& element);
00091 
00092   // Add all the elements from the argument list to this list.
00093   void AddAll(const List<T, P>& other);
00094 
00095   // Add all the elements from the vector to this list.
00096   void AddAll(const Vector<T>& other);
00097 
00098   // Inserts the element at the specific index.
00099   void InsertAt(int index, const T& element);
00100 
00101   // Added 'count' elements with the value 'value' and returns a
00102   // vector that allows access to the elements.  The vector is valid
00103   // until the next change is made to this list.
00104   Vector<T> AddBlock(T value, int count);
00105 
00106   // Removes the i'th element without deleting it even if T is a
00107   // pointer type; moves all elements above i "down". Returns the
00108   // removed element.  This function's complexity is linear in the
00109   // size of the list.
00110   T Remove(int i);
00111 
00112   // Remove the given element from the list. Returns whether or not
00113   // the input is included in the list in the first place.
00114   bool RemoveElement(const T& elm);
00115 
00116   // Removes the last element without deleting it even if T is a
00117   // pointer type. Returns the removed element.
00118   INLINE(T RemoveLast()) { return Remove(length_ - 1); }
00119 
00120   // Clears the list by setting the length to zero. Even if T is a
00121   // pointer type, clearing the list doesn't delete the entries.
00122   INLINE(void Clear());
00123 
00124   // Drops all but the first 'pos' elements from the list.
00125   INLINE(void Rewind(int pos));
00126 
00127   // Drop the last 'count' elements from the list.
00128   INLINE(void RewindBy(int count)) { Rewind(length_ - count); }

00135   template<class Visitor>
00136   void Iterate(Visitor* visitor);

00139   void Sort(int (*cmp)(const T* x, const T* y));

00144  private:

00149   INLINE(T* NewData(int n))  { return static_cast<T*>(P::New(n * sizeof(T))); }
00150   INLINE(void DeleteData(T* data))  { P::Delete(data); }
