00038 // 0xffff0fc0 is the hard coded address of a function provided by
00039 // the kernel which implements an atomic compare-exchange. On older
00040 // ARM architecture revisions (pre-v6) this may be implemented using
00041 // a syscall. This address is stable, and in active use (hard coded)
00042 // by at least glibc-2.7 and the Android C library.
00043 typedef Atomic32 (*LinuxKernelCmpxchgFunc)(Atomic32 old_value,
00044                                            Atomic32 new_value,
00045                                            volatile Atomic32* ptr);
00046 LinuxKernelCmpxchgFunc pLinuxKernelCmpxchg __attribute__((weak)) =
00047     (LinuxKernelCmpxchgFunc) 0xffff0fc0;

00054 inline Atomic32 NoBarrier_CompareAndSwap(volatile Atomic32* ptr,
00055                                          Atomic32 old_value,
00056                                          Atomic32 new_value) {
00057   Atomic32 prev_value = *ptr;
00058   do {
00059     if (!pLinuxKernelCmpxchg(old_value, new_value,
00060                              const_cast<Atomic32*>(ptr))) {
00061       return old_value;
00062     }
00063     prev_value = *ptr;
00064   } while (prev_value == old_value);
00065   return prev_value;
00066 }

00068 inline Atomic32 NoBarrier_AtomicExchange(volatile Atomic32* ptr,
00069                                          Atomic32 new_value) {
00070   Atomic32 old_value;
00071   do {
00072     old_value = *ptr;
00073   } while (pLinuxKernelCmpxchg(old_value, new_value,
00074                                const_cast<Atomic32*>(ptr)));
00075   return old_value;
00076 }

00083 inline Atomic32 Barrier_AtomicIncrement(volatile Atomic32* ptr,
00084                                         Atomic32 increment) {
00085   for (;;) {
00086     // Atomic exchange the old value with an incremented one.
00087     Atomic32 old_value = *ptr;
00088     Atomic32 new_value = old_value + increment;
00089     if (pLinuxKernelCmpxchg(old_value, new_value,
00090                             const_cast<Atomic32*>(ptr)) == 0) {
00091       // The exchange took place as expected.
00092       return new_value;
00093     }
00094     // Otherwise, *ptr changed mid-loop and we need to retry.
00095   }
00096 }
