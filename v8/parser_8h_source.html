00189 // A BufferedZoneList is an automatically growing list, just like (and backed
00190 // by) a ZoneList, that is optimized for the case of adding and removing
00191 // a single element. The last element added is stored outside the backing list,
00192 // and if no more than one element is ever added, the ZoneList isn't even
00193 // allocated.
00194 // Elements must not be NULL pointers.
00195 template <typename T, int initial_size>
00196 class BufferedZoneList {
00197  public:
00198   BufferedZoneList() : list_(NULL), last_(NULL) {}
00199 
00200   // Adds element at end of list. This element is buffered and can
00201   // be read using last() or removed using RemoveLast until a new Add or until
00202   // RemoveLast or GetList has been called.
00203   void Add(T* value) {
00204     if (last_ != NULL) {
00205       if (list_ == NULL) {
00206         list_ = new ZoneList<T*>(initial_size);
00207       }
00208       list_->Add(last_);
00209     }
00210     last_ = value;
00211   }
00212 
00213   T* last() {
00214     ASSERT(last_ != NULL);
00215     return last_;
00216   }
00217 
00218   T* RemoveLast() {
00219     ASSERT(last_ != NULL);
00220     T* result = last_;
00221     if ((list_ != NULL) && (list_->length() > 0))
00222       last_ = list_->RemoveLast();
00223     else
00224       last_ = NULL;
00225     return result;
00226   }
00227 
00228   T* Get(int i) {
00229     ASSERT((0 <= i) && (i < length()));
00230     if (list_ == NULL) {
00231       ASSERT_EQ(0, i);
00232       return last_;
00233     } else {
00234       if (i == list_->length()) {
00235         ASSERT(last_ != NULL);
00236         return last_;
00237       } else {
00238         return list_->at(i);
00239       }
00240     }
00241   }
00242 
00243   void Clear() {
00244     list_ = NULL;
00245     last_ = NULL;
00246   }
00247 
00248   int length() {
00249     int length = (list_ == NULL) ? 0 : list_->length();
00250     return length + ((last_ == NULL) ? 0 : 1);
00251   }
00252 
00253   ZoneList<T*>* GetList() {
00254     if (list_ == NULL) {
00255       list_ = new ZoneList<T*>(initial_size);
00256     }
00257     if (last_ != NULL) {
00258       list_->Add(last_);
00259       last_ = NULL;
00260     }
00261     return list_;
00262   }
00263 
00264  private:
00265   ZoneList<T*>* list_;
00266   T* last_;
00267 };
