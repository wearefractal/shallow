00076 // Labels represent pc locations; they are typically jump or call targets.
00077 // After declaration, a label can be freely used to denote known or (yet)
00078 // unknown pc location. Assembler::bind() is used to bind a label to the
00079 // current pc. A label can be bound only once.
00080 
00081 class Label BASE_EMBEDDED {
00082  public:
00083   enum Distance {
00084     kNear, kFar
00085   };
00086 
00087   INLINE(Label()) {
00088     Unuse();
00089     UnuseNear();
00090   }
00091 
00092   INLINE(~Label()) {
00093     ASSERT(!is_linked());
00094     ASSERT(!is_near_linked());
00095   }
00096 
00097   INLINE(void Unuse()) { pos_ = 0; }
00098   INLINE(void UnuseNear()) { near_link_pos_ = 0; }
00099 
00100   INLINE(bool is_bound() const) { return pos_ <  0; }
00101   INLINE(bool is_unused() const) { return pos_ == 0 && near_link_pos_ == 0; }
00102   INLINE(bool is_linked() const) { return pos_ >  0; }
00103   INLINE(bool is_near_linked() const) { return near_link_pos_ > 0; }
00104 
00105   // Returns the position of bound or linked labels. Cannot be used
00106   // for unused labels.
00107   int pos() const;
00108   int near_link_pos() const { return near_link_pos_ - 1; }
00109 
00110  private:
00111   // pos_ encodes both the binding state (via its sign)
00112   // and the binding position (via its value) of a label.
00113   //
00114   // pos_ <  0  bound label, pos() returns the jump target position
00115   // pos_ == 0  unused label
00116   // pos_ >  0  linked label, pos() returns the last reference position
00117   int pos_;
00118 
00119   // Behaves like |pos_| in the "> 0" case, but for near jumps to this label.
00120   int near_link_pos_;
00121 
00122   void bind_to(int pos)  {
00123     pos_ = -pos - 1;
00124     ASSERT(is_bound());
00125   }
00126   void link_to(int pos, Distance distance = kFar) {
00127     if (distance == kNear) {
00128       near_link_pos_ = pos + 1;
00129       ASSERT(is_near_linked());
00130     } else {
00131       pos_ = pos + 1;
00132       ASSERT(is_linked());
00133     }
00134   }
00135

00137   friend class RegexpAssembler;
00138   friend class Displacement;
00139   friend class RegExpMacroAssemblerIrregexp;

00150 // Relocation information consists of the address (pc) of the datum
00151 // to which the relocation information applies, the relocation mode
00152 // (rmode), and an optional data field. The relocation mode may be
00153 // "descriptive" and not indicate a need for relocation, but simply
00154 // describe a property of the datum. Such rmodes are useful for GC
00155 // and nice disassembly output.

00700   static void* Redirect(Isolate* isolate,
00701                         void* address,
00702                         Type type = ExternalReference::BUILTIN_CALL) {
00703     ExternalReferenceRedirector* redirector =
00704         reinterpret_cast<ExternalReferenceRedirector*>(
00705             isolate->external_reference_redirector());
00706     if (redirector == NULL) return address;
00707     void* answer = (*redirector)(address, type);
00708     return answer;
00709   }
00710 
00711   static void* Redirect(Isolate* isolate,
00712                         Address address_arg,
00713                         Type type = ExternalReference::BUILTIN_CALL) {
00714     ExternalReferenceRedirector* redirector =
00715         reinterpret_cast<ExternalReferenceRedirector*>(
00716             isolate->external_reference_redirector());
00717     void* address = reinterpret_cast<void*>(address_arg);
00718     void* answer = (redirector == NULL) ?
00719                    address :
00720                    (*redirector)(address, type);
00721     return answer;
