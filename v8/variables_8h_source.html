00037 // The AST refers to variables via VariableProxies - placeholders for the actual
00038 // variables. Variables themselves are never directly referred to from the AST,
00039 // they are maintained by scopes, and referred to from VariableProxies and Slots
00040 // after binding and variable allocation.
00041 
00042 class Variable: public ZoneObject {
00043  public:
00044   enum Kind {
00045     NORMAL,
00046     THIS,
00047     ARGUMENTS
00048   };

00050   enum Location {
00051     // Before and during variable allocation, a variable whose location is
00052     // not yet determined.  After allocation, a variable looked up as a
00053     // property on the global object (and possibly absent).  name() is the
00054     // variable name, index() is invalid.
00055     UNALLOCATED,
00056 
00057     // A slot in the parameter section on the stack.  index() is the
00058     // parameter index, counting left-to-right.  The reciever is index -1;
00059     // the first parameter is index 0.
00060     PARAMETER,
00061 
00062     // A slot in the local section on the stack.  index() is the variable
00063     // index in the stack frame, starting at 0.
00064     LOCAL,
00065 
00066     // An indexed slot in a heap context.  index() is the variable index in
00067     // the context object on the heap, starting at 0.  scope() is the
00068     // corresponding scope.
00069     CONTEXT,
00070 
00071     // A named slot in a heap context.  name() is the variable name in the
00072     // context object on the heap, with lookup starting at the current
00073     // context.  index() is invalid.
00074     LOOKUP
00075   };
