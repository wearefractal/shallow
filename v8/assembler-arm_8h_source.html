00050 // CPU Registers.
00051 //
00052 // 1) We would prefer to use an enum, but enum values are assignment-
00053 // compatible with int, which has caused code-generation bugs.
00054 //
00055 // 2) We would prefer to use a class instead of a struct but we don't like
00056 // the register initialization to depend on the particular initialization
00057 // order (which appears to be different on OS X, Linux, and Windows for the
00058 // installed versions of C++ we tried). Using a struct permits C-style
00059 // "initialization". Also, the Register objects cannot be const as this
00060 // forces initialization stubs in MSVC, making us dependent on initialization
00061 // order.
00062 //
00063 // 3) By not using an enum, we are possibly preventing the compiler from
00064 // doing certain constant folds, which may significantly reduce the
00065 // code generated for some assembly instructions (because they boil down
00066 // to a few constants). If this is a problem, we could change the code
00067 // such that we use an enum in optimized mode, and the struct in debug
00068 // mode. This way we get the compile-time error checking in debug mode
00069 // and best performance in optimized code.

00501 // CpuFeatures keeps track of which features are supported by the target CPU.
00502 // Supported features must be enabled by a Scope before use.
00503 class CpuFeatures : public AllStatic {
00504  public:
00505   // Detect features of the target CPU. Set safe defaults if the serializer
00506   // is enabled (snapshots must be portable).
00507   static void Probe();
00508 
00509   // Check whether a feature is supported by the target CPU.
00510   static bool IsSupported(CpuFeature f) {
00511     ASSERT(initialized_);
00512     if (f == VFP3 && !FLAG_enable_vfp3) return false;
00513     return (supported_ & (1u << f)) != 0;
00514   }
00515 
00516 #ifdef DEBUG
00517   // Check whether a feature is currently enabled.
00518   static bool IsEnabled(CpuFeature f) {
00519     ASSERT(initialized_);
00520     Isolate* isolate = Isolate::UncheckedCurrent();
00521     if (isolate == NULL) {
00522       // When no isolate is available, work as if we're running in
00523       // release mode.
00524       return IsSupported(f);
00525     }
00526     unsigned enabled = static_cast<unsigned>(isolate->enabled_cpu_features());
00527     return (enabled & (1u << f)) != 0;
00528   }
00529 #endif

00531   // Enable a specified feature within a scope.
00532   class Scope BASE_EMBEDDED {
00533 #ifdef DEBUG
00534

00541       isolate_ = Isolate::UncheckedCurrent();

00544         old_enabled_ = static_cast<unsigned>(isolate_->enabled_cpu_features());
00545         isolate_->set_enabled_cpu_features(old_enabled_ | mask);
00546       }
00547     }
00548     ~Scope() {
00549       ASSERT_EQ(Isolate::UncheckedCurrent(), isolate_);
00550       if (isolate_ != NULL) {
00551         isolate_->set_enabled_cpu_features(old_enabled_);
00552       }
00553     }

00555    private:

00557     unsigned old_enabled_;
00558 #else

00630   // into a buffer, with the instructions starting from the beginning and the
00631   // relocation information starting from the end of the buffer. See CodeDesc
00632   // for a detailed comment on the layout (globals.h).

00634   // If the provided buffer is NULL, the assembler allocates and grows its own

00639   // for code generation and assumes its size to be buffer_size. If the buffer

01298   // Prevent contant pool emission until EndBlockConstPool is called.
01299   // Call to this function can be nested but must be followed by an equal
01300   // number of call to EndBlockConstpool.
01301   void StartBlockConstPool() {
01302     if (const_pool_blocked_nesting_++ == 0) {
01303       // Prevent constant pool checks happening by setting the next check to
01304       // the biggest possible offset.
01305       next_buffer_check_ = kMaxInt;
01306     }
01307   }
01308 
01309   // Resume constant pool emission. Need to be called as many time as
01310   // StartBlockConstPool to have an effect.
01311   void EndBlockConstPool() {
01312     if (--const_pool_blocked_nesting_ == 0) {
01313       // Check the constant pool hasn't been blocked for too long.
01314       ASSERT((num_pending_reloc_info_ == 0) ||
01315              (pc_offset() < (first_const_pool_use_ + kMaxDistToPool)));
01316       // Two cases:
01317       //  * no_const_pool_before_ >= next_buffer_check_ and the emission is
01318       //    still blocked
01319       //  * no_const_pool_before_ < next_buffer_check_ and the next emit will
01320       //    trigger a check.
01321       next_buffer_check_ = no_const_pool_before_;
01322     }
01323   }
01324 
01325   bool is_const_pool_blocked() const {
01326     return (const_pool_blocked_nesting_ > 0) ||
01327            (pc_offset() < no_const_pool_before_);
01328   }
