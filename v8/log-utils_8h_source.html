00076   // Implementation of writing to a log file.
00077   int WriteToFile(const char* msg, int length) {
00078     ASSERT(output_handle_ != NULL);
00079     size_t rv = fwrite(msg, 1, length, output_handle_);
00080     ASSERT(static_cast<size_t>(length) == rv);
00081     USE(rv);
00082     fflush(output_handle_);

00084   }

00086   // Whether logging is stopped (e.g. due to insufficient resources).
00087   bool is_stopped_;
00088 
00089   // When logging is active output_handle_ is used to store a pointer to log

00092 
00093   // Used when low-level profiling is active.

00096   // mutex_ is a Mutex used for enforcing exclusive
