00037 // This class implements the following abstract grammar of interfaces
00038 // (i.e. module types):
00039 //   interface ::= UNDETERMINED | VALUE | MODULE(exports)
00040 //   exports ::= {name : interface, ...}
00041 // A frozen module type is one that is fully determined. Unification does not
00042 // allow adding additional exports to frozen interfaces.
00043 // Otherwise, unifying modules merges their exports.
00044 // Undetermined types are unification variables that can be unified freely.
00045 
00046 class Interface : public ZoneObject {
00047  public:
00048   // ---------------------------------------------------------------------------
00049   // Factory methods.
00050 
00051   static Interface* NewValue() {
00052     static Interface value_interface(VALUE + FROZEN);  // Cached.
00053     return &value_interface;
00054   }
00055 
00056   static Interface* NewUnknown() {
00057     return new Interface(NONE);
00058   }
00059 
00060   static Interface* NewModule() {
00061     return new Interface(MODULE);
00062   }
00063 
00064   // ---------------------------------------------------------------------------
00065   // Mutators.
00066 
00067   // Add a name to the list of exports. If it already exists, unify with
00068   // interface, otherwise insert unless this is closed.
00069   void Add(Handle<String> name, Interface* interface, bool* ok) {
00070     DoAdd(name.location(), name->Hash(), interface, ok);
00071   }
00072 
00073   // Unify with another interface. If successful, both interface objects will
00074   // represent the same type, and changes to one are reflected in the other.
00075   void Unify(Interface* that, bool* ok);
00076 
00077   // Determine this interface to be a value interface.
00078   void MakeValue(bool* ok) {
00079     *ok = !IsModule();
00080     if (*ok) Chase()->flags_ |= VALUE;
00081   }
00082 
00083   // Determine this interface to be a module interface.
00084   void MakeModule(bool* ok) {
00085     *ok = !IsValue();
00086     if (*ok) Chase()->flags_ |= MODULE;
00087   }
00088 
00089   // Set associated instance object.
00090   void MakeSingleton(Handle<JSModule> instance, bool* ok) {
00091     *ok = IsModule() && Chase()->instance_.is_null();
00092     if (*ok) Chase()->instance_ = instance;
00093   }
00094 
00095   // Do not allow any further refinements, directly or through unification.
00096   void Freeze(bool* ok) {
00097     *ok = IsValue() || IsModule();
00098     if (*ok) Chase()->flags_ |= FROZEN;
00099   }
00100 
00101   // ---------------------------------------------------------------------------
00102   // Accessors.
00103 
00104   // Check whether this is still a fully undetermined type.
00105   bool IsUnknown() { return Chase()->flags_ == NONE; }
00106 
00107   // Check whether this is a value type.
00108   bool IsValue() { return Chase()->flags_ & VALUE; }
00109 
00110   // Check whether this is a module type.
00111   bool IsModule() { return Chase()->flags_ & MODULE; }
00112 
00113   // Check whether this is closed (i.e. fully determined).
00114   bool IsFrozen() { return Chase()->flags_ & FROZEN; }
00115 
00116   Handle<JSModule> Instance() { return Chase()->instance_; }
00117 
00118   // Look up an exported name. Returns NULL if not (yet) defined.
00119   Interface* Lookup(Handle<String> name);

00121   // ---------------------------------------------------------------------------
00122   // Iterators.
00123 
00124   // Use like:
00125   //   for (auto it = interface->iterator(); !it.done(); it.Advance()) {
00126   //     ... it.name() ... it.interface() ...
00127   //   }
00128   class Iterator {
00129    public:
00130     bool done() const { return entry_ == NULL; }
00131     Handle<String> name() const {
00132       ASSERT(!done());
00133       return Handle<String>(*static_cast<String**>(entry_->key));
00134     }
00135     Interface* interface() const {
00136       ASSERT(!done());
00137       return static_cast<Interface*>(entry_->value);
00138     }
00139     void Advance() { entry_ = exports_->Next(entry_); }
00140 
00141    private:

00148   };

00183   Interface* Chase() {
00184     Interface* result = this;
00185     while (result->forward_ != NULL) result = result->forward_;

00187     return result;
00188   }
