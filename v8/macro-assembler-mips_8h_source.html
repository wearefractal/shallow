00041 // Reserved Register Usage Summary.
00042 //
00043 // Registers t8, t9, and at are reserved for use by the MacroAssembler.
00044 //
00045 // The programmer should know that the MacroAssembler may clobber these three,
00046 // but won't touch other registers except in special cases.
00047 //
00048 // Per the MIPS ABI, register t9 must be used for indirect function call
00049 // via 'jalr t9' or 'jr t9' instructions. This is relied upon by gcc when
00050 // trying to update gp register for position-independent-code. Whenever
00051 // MIPS generated code calls C code, it must be via t9 register.

00449   static inline int GetCodeMarker(Instr instr) {
00450     uint32_t opcode = ((instr & kOpcodeMask));
00451     uint32_t rt = ((instr & kRtFieldMask) >> kRtShift);
00452     uint32_t rs = ((instr & kRsFieldMask) >> kRsShift);
00453     uint32_t sa = ((instr & kSaFieldMask) >> kSaShift);
00454 
00455     // Return <n> if we have a sll zero_reg, zero_reg, n
00456     // else return -1.
00457     bool sllzz = (opcode == SLL &&
00458                   rt == static_cast<uint32_t>(ToNumber(zero_reg)) &&
00459                   rs == static_cast<uint32_t>(ToNumber(zero_reg)));
00460     int type =
00461         (sllzz && FIRST_IC_MARKER <= sa && sa < LAST_CODE_MARKER) ? sa : -1;
00462     ASSERT((type == -1) ||
00463            ((FIRST_IC_MARKER <= type) && (type < LAST_CODE_MARKER)));
00464     return type;
00465   }
