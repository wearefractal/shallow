00064 // Returns the value (0 .. 15) of a hexadecimal character c.
00065 // If c is not a legal hexadecimal character, returns a value < 0.
00066 inline int HexValue(uc32 c) {
00067   c -= '0';
00068   if (static_cast<unsigned>(c) <= 9) return c;
00069   c = (c | 0x20) - ('a' - '0');  // detect 0x11..0x16 and 0x31..0x36.
00070   if (static_cast<unsigned>(c) <= 5) return c + 10;
00071   return -1;
00072 }

00076 // Buffered stream of UTF-16 code units, using an internal UTF-16 buffer.
00077 // A code unit is a 16 bit value representing either a 16 bit code point
00078 // or one part of a surrogate pair that make a single 21 bit code point.
00079 
00080 class Utf16CharacterStream {
00081  public:
00082   Utf16CharacterStream() : pos_(0) { }
00083   virtual ~Utf16CharacterStream() { }
00084 
00085   // Returns and advances past the next UTF-16 code unit in the input
00086   // stream. If there are no more code units, it returns a negative
00087   // value.
00088   inline uc32 Advance() {
00089     if (buffer_cursor_ < buffer_end_ || ReadBlock()) {
00090       pos_++;
00091       return static_cast<uc32>(*(buffer_cursor_++));
00092     }
00093     // Note: currently the following increment is necessary to avoid a
00094     // parser problem! The scanner treats the final kEndOfInput as
00095     // a code unit with a position, and does math relative to that
00096     // position.
00097     pos_++;
00098 
00099     return kEndOfInput;
00100   }
00101 
00102   // Return the current position in the code unit stream.
00103   // Starts at zero.
00104   inline unsigned pos() const { return pos_; }
00105 
00106   // Skips forward past the next code_unit_count UTF-16 code units
00107   // in the input, or until the end of input if that comes sooner.
00108   // Returns the number of code units actually skipped. If less
00109   // than code_unit_count,
00110   inline unsigned SeekForward(unsigned code_unit_count) {
00111     unsigned buffered_chars =
00112         static_cast<unsigned>(buffer_end_ - buffer_cursor_);
00113     if (code_unit_count <= buffered_chars) {
00114       buffer_cursor_ += code_unit_count;
00115       pos_ += code_unit_count;
00116       return code_unit_count;
00117     }
00118     return SlowSeekForward(code_unit_count);
00119   }
00120 
00121   // Pushes back the most recently read UTF-16 code unit (or negative
00122   // value if at end of input), i.e., the value returned by the most recent
00123   // call to Advance.
00124   // Must not be used right after calling SeekForward.
00125   virtual void PushBack(int32_t code_unit) = 0;
00126 
00127  protected:
00128   static const uc32 kEndOfInput = -1;
00129 
00130   // Ensures that the buffer_cursor_ points to the code_unit at
00131   // position pos_ of the input, if possible. If the position
00132   // is at or after the end of the input, return false. If there
00133   // are more code_units available, return true.
00134   virtual bool ReadBlock() = 0;
00135   virtual unsigned SlowSeekForward(unsigned code_unit_count) = 0;
00136 
00137   const uc16* buffer_cursor_;
00138   const uc16* buffer_end_;
00139   unsigned pos_;
00140 };

00173 class LiteralBuffer {
00174  public:
00175   LiteralBuffer() : is_ascii_(true), position_(0), backing_store_() { }
00176 
00177   ~LiteralBuffer() {
00178     if (backing_store_.length() > 0) {
00179       backing_store_.Dispose();
00180     }
00181   }

00183   INLINE(void AddChar(uint32_t code_unit)) {
00184     if (position_ >= backing_store_.length()) ExpandBuffer();
00185     if (is_ascii_) {
00186       if (code_unit < kMaxAsciiCharCodeU) {
00187         backing_store_[position_] = static_cast<byte>(code_unit);
00188         position_ += kASCIISize;
00189         return;
00190       }
00191       ConvertToUtf16();
00192     }
00193     ASSERT(code_unit < 0x10000u);
00194     *reinterpret_cast<uc16*>(&backing_store_[position_]) = code_unit;
00195     position_ += kUC16Size;
00196   }
00197 
00198   bool is_ascii() { return is_ascii_; }
00199 
00200   Vector<const uc16> utf16_literal() {
00201     ASSERT(!is_ascii_);
00202     ASSERT((position_ & 0x1) == 0);
00203     return Vector<const uc16>(
00204         reinterpret_cast<const uc16*>(backing_store_.start()),
00205         position_ >> 1);
00206   }
00207 
00208   Vector<const char> ascii_literal() {
00209     ASSERT(is_ascii_);
00210     return Vector<const char>(
00211         reinterpret_cast<const char*>(backing_store_.start()),
00212         position_);
00213   }
00214 
00215   int length() {
00216     return is_ascii_ ? position_ : (position_ >> 1);
00217   }
00218 
00219   void Reset() {
00220     position_ = 0;
00221     is_ascii_ = true;
00222   }
00223 
00224  private:
00225   static const int kInitialCapacity = 16;
00226   static const int kGrowthFactory = 4;
00227   static const int kMinConversionSlack = 256;
00228   static const int kMaxGrowth = 1 * MB;
00229   inline int NewCapacity(int min_capacity) {
00230     int capacity = Max(min_capacity, backing_store_.length());
00231     int new_capacity = Min(capacity * kGrowthFactory, capacity + kMaxGrowth);
00232     return new_capacity;
00233   }
00234 
00235   void ExpandBuffer() {
00236     Vector<byte> new_store = Vector<byte>::New(NewCapacity(kInitialCapacity));
00237     memcpy(new_store.start(), backing_store_.start(), position_);
00238     backing_store_.Dispose();
00239     backing_store_ = new_store;
00240   }
00241 
00242   void ConvertToUtf16() {
00243     ASSERT(is_ascii_);
00244     Vector<byte> new_store;
00245     int new_content_size = position_ * kUC16Size;
00246     if (new_content_size >= backing_store_.length()) {
00247       // Ensure room for all currently read code units as UC16 as well
00248       // as the code unit about to be stored.
00249       new_store = Vector<byte>::New(NewCapacity(new_content_size));
00250     } else {
00251       new_store = backing_store_;
00252     }
00253     char* src = reinterpret_cast<char*>(backing_store_.start());
00254     uc16* dst = reinterpret_cast<uc16*>(new_store.start());
00255     for (int i = position_ - 1; i >= 0; i--) {
00256       dst[i] = src[i];
00257     }
00258     if (new_store.start() != backing_store_.start()) {
00259       backing_store_.Dispose();
00260       backing_store_ = new_store;
00261     }
00262     position_ = new_content_size;
00263     is_ascii_ = false;
00264   }
00265 
00266   bool is_ascii_;
00267   int position_;
00268   Vector<byte> backing_store_;
00269 
00270   DISALLOW_COPY_AND_ASSIGN(LiteralBuffer);
